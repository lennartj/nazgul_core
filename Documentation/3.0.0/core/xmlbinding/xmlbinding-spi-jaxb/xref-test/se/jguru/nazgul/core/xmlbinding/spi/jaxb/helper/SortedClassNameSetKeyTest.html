24  package se.jguru.nazgul.core.xmlbinding.spi.jaxb.helper;
25  
26  import org.junit.Assert;
27  import org.junit.Before;
28  import org.junit.Test;
29  import se.jguru.nazgul.core.xmlbinding.spi.jaxb.helper.types.Account;
30  import se.jguru.nazgul.core.xmlbinding.spi.jaxb.helper.types.Person;
31  import se.jguru.nazgul.core.xmlbinding.spi.jaxb.transport.EntityTransporter;
32  
33  import java.util.SortedSet;
34  import java.util.TreeSet;
35  
36  /**
37   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
38   */
39  public class SortedClassNameSetKeyTest {
40  
41      // Shared state
42      private SortedSet<String> threeClassNames;
43      private SortedClassNameSetKey unitUnderTest;
44  
45      @Before
46      public void setupSharedState() {
47  
48          threeClassNames = new TreeSet<String>();
49          threeClassNames.add(Person.class.getName());
50          threeClassNames.add(Account.class.getName());
51          threeClassNames.add(EntityTransporter.class.getName());
52  
53          unitUnderTest = new SortedClassNameSetKey(threeClassNames);
54      }
55  
56      @Test(expected = NullPointerException.class)
57      public void validateExceptionOnNullClassNameSet() {
58  
59          // Act & Assert
60          new SortedClassNameSetKey(null);
61      }
62  
63      @Test
64      public void validateEqualityAndHashcode() {
65  
66          // Assemble
67          final SortedSet<String> someClassNames = new TreeSet<String>();
68          someClassNames.add(Person.class.getName());
69          someClassNames.add(Account.class.getName());
70  
71          final SortedClassNameSetKey unitUnderTest2 = new SortedClassNameSetKey(someClassNames);
72  
73          // Act
74          final boolean result = unitUnderTest.equals(unitUnderTest2);
75          final int hashcode1 = unitUnderTest.hashCode();
76          final int hashcode2 = unitUnderTest2.hashCode();
77  
78          // Assert
79          Assert.assertEquals(result, unitUnderTest.toString().equals(unitUnderTest2.toString()));
80          Assert.assertEquals(hashcode1, unitUnderTest.toString().hashCode());
81          Assert.assertEquals(hashcode2, unitUnderTest2.toString().hashCode());
82  
83          Assert.assertFalse(unitUnderTest.equals(null));
84          Assert.assertFalse(unitUnderTest.equals("fooBar"));
85      }
86  
87      @Test
88      public void validateComparability() {
89  
90          // Assemble
91          final SortedSet<String> someClassNames = new TreeSet<String>();
92          someClassNames.add(Person.class.getName());
93          someClassNames.add(Account.class.getName());
94  
95          final SortedClassNameSetKey unitUnderTest2 = new SortedClassNameSetKey(threeClassNames);
96          final SortedClassNameSetKey unitUnderTest3 = new SortedClassNameSetKey(someClassNames);
97  
98          // Act
99          final int result1 = unitUnderTest.compareTo(unitUnderTest2);
100         final int result2 = unitUnderTest.compareTo(unitUnderTest3);
101 
102         // Assert
103         Assert.assertEquals(Integer.MIN_VALUE, unitUnderTest.compareTo(null));
104         Assert.assertEquals(0, result1);
105         Assert.assertEquals(unitUnderTest.toString().compareTo(unitUnderTest2.toString()), result1);
106         Assert.assertEquals(unitUnderTest.toString().compareTo(unitUnderTest3.toString()), result2);
107 
108         Assert.assertEquals(unitUnderTest, unitUnderTest2);
109         Assert.assertNotSame(unitUnderTest, unitUnderTest2);
110     }
111 
112     @Test
113     public void validateContainability() {
114 
115         // Assemble
116         final SortedSet<String> someClassNames = new TreeSet<String>();
117         someClassNames.add(Person.class.getName());
118         someClassNames.add(Account.class.getName());
119 
120         final SortedSet<String> someOtherClassNames = new TreeSet<String>();
121         someOtherClassNames.add(Person.class.getName());
122         someOtherClassNames.add(Account.class.getName());
123         someOtherClassNames.add(String.class.getName());
124 
125         final SortedClassNameSetKey unitUnderTest2 = new SortedClassNameSetKey(someOtherClassNames);
126 
127         // Act
128         final boolean result1 = unitUnderTest.containsAll(someClassNames);
129         final boolean result2 = unitUnderTest.containsAll(someOtherClassNames);
130         final boolean result4 = unitUnderTest2.containsAll(someClassNames);
131 
132         // Assert
133         Assert.assertTrue(result1);
134         Assert.assertFalse(result2);
135         Assert.assertTrue(result4);
136     }
137 }
