188     @Test
189     public void validateJaxbSchemaGeneration() throws Exception {
190 
191         // Assemble
192         final JAXBContext ctx1 = JAXBContext.newInstance(ThreePartCereal.class);
193         final JAXBContext ctx2 = JAXBContext.newInstance(ThreePartCerealWithNillableAnnotationElement.class);
194         final String requiredTargetNamespace = "targetNamespace=\"http://cereal\"";
195         final String xPathToDiffElement = "/schema[1]/complexType[1]/sequence[1]/element[3]";
196 
197         // Act
198         final List<String> schema1 = generateSchemas(ctx1);
199         final List<String> schema2 = generateSchemas(ctx2);
200 
201         final List<String> relevantSchema1 = CollectionAlgorithms.filter(schema1, new Filter<String>() {
202             @Override
203             public boolean accept(String candidate) {
204                 return candidate.contains(requiredTargetNamespace);
205             }
206         });
207 
208         final List<String> relevantSchema2 = CollectionAlgorithms.filter(schema2, new Filter<String>() {
209             @Override
210             public boolean accept(String candidate) {
211                 return candidate.contains(requiredTargetNamespace);
212             }
213         });
214 
215         // Assert
216         Assert.assertEquals(1, relevantSchema1.size());
217         Assert.assertEquals(1, relevantSchema2.size());
218 
219         final String stringForm1Schema = relevantSchema1.get(0);
220         final String stringForm2Schema = relevantSchema2.get(0);
221 
222         final List<String> linesNotInSchema2 = new ArrayList<String>();
223         final BufferedReader schema1Reader = new BufferedReader(new StringReader(stringForm1Schema));
224         String aLine = null;
225         while ((aLine = schema1Reader.readLine()) != null) {
226             if (!stringForm2Schema.contains(aLine.trim())) {
227                 linesNotInSchema2.add(aLine.trim());
228             }
229         }
230 
231         // 0: <xs:complexType name="threePartCereal">
232         // 1: <xs:element ref="ns3:colourIngredient" minOccurs="0"/>
233         Assert.assertTrue(linesNotInSchema2.get(1).contains(
234                 "<xs:element ref=\"ns3:colourIngredient\" minOccurs=\"0\"/>"));
235 
236         final Diff diff = XmlTestUtils.compareXmlIgnoringWhitespace(relevantSchema1.get(0), relevantSchema2.get(0));
237         Assert.assertFalse(diff.similar());
238 
239         final SortedMap<String, List<Difference>> xpath2DiffMap = XmlTestUtils.getXPathLocationToDifferenceMap(diff);
240         final List<Difference> elementDifference = xpath2DiffMap.get(xPathToDiffElement);
241 
242         Assert.assertEquals(6, elementDifference.size());
243         Assert.assertEquals("number of element attributes", elementDifference.get(0).getDescription());
244         for (final Difference currentDifference : elementDifference) {
245             switch (currentDifference.getId()) {
246                 case 11:
247                     // Different number of element attributes generated:
248                     //
249                     // 0: <xs:element ref="ns3:colourIngredient" minOccurs="0"/>
250                     // 1: <xs:element name="colourIngredient" type="xs:string" form="qualified" nillable="true" minOccurs="0"/>
251                     //
252                     Assert.assertEquals(2, Integer.parseInt(currentDifference.getControlNodeDetail().getValue()));
253                     Assert.assertEquals(5, Integer.parseInt(currentDifference.getTestNodeDetail().getValue()));
254                     break;
255 
256                 default:
257                     // Anything else, typically attribute name changes.
258                     break;
259             }
260         }
261     }

267     private List<String> generateSchemas(final JAXBContext ctx) {
268 
269         final List<String> toReturn = new ArrayList<String>();
270         final List<ByteArrayOutputStream> schemaSnippets = new ArrayList<ByteArrayOutputStream>();
271 
272         try {
273             ctx.generateSchema(new SchemaOutputResolver() {
274                 @Override
275                 public Result createOutput(final String namespaceUri, final String suggestedFileName)
276                         throws IOException {
277 
278                     // Create the result ByteArrayOutputStream
279                     ByteArrayOutputStream out = new ByteArrayOutputStream();
280                     schemaSnippets.add(out);
281 
282                     // Target the result to the generated ByteArrayOutputStream.
283                     StreamResult streamResult = new StreamResult(out);
284                     streamResult.setSystemId("");
285 
286 
287                     // All done.
288                     return streamResult;
289                 }
290             });
291         } catch (IOException e) {
292             throw new IllegalArgumentException("Could not acquire Schema snippets.", e);
293         }
294 
295         // Convert and return
296         return CollectionAlgorithms.transform(schemaSnippets, new Transformer<ByteArrayOutputStream, String>() {
297             @Override
298             public String transform(final ByteArrayOutputStream input) {
299                 return input.toString().trim();
300             }
301         });
302     }

304     @Test
305     public void validateJaxbQnameAndNamespacePrefixMapperOperation() throws Exception {
306 
307         // Assemble
308         final JAXBContext ctx = JAXBContext.newInstance(ThreePartCerealWithNillableAnnotationElement.class);
309         final ThreePartCerealWithNillableAnnotationElement cereal =
310                 new ThreePartCerealWithNillableAnnotationElement("wholegrain", "raisins", null, 2, 1);
311         final QName qName = new QName("http://www.jguru.se/foobar", "rootElementName", "xmlPrefix");
312         final JAXBElement<ThreePartCerealWithNillableAnnotationElement> element =
313                 new JAXBElement<ThreePartCerealWithNillableAnnotationElement>(qName,
314                         ThreePartCerealWithNillableAnnotationElement.class, cereal);
315         final SortedMap<String, Tuple<String, Boolean>> namespaceToPrefixTupleMap =
316                 new TreeMap<String, Tuple<String, Boolean>>();
317 
318         final String expected = "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>\n" +
319                 "<xmlPrefix:rootElementName flavourPercentage=\"2\" colourPercentage=\"1\" xmlns:colour=\"http://cereal/ingredients/colour\" xmlns:base=\"http://cereal/ingredients/base\" xmlns:xmlPrefix=\"http://www.jguru.se/foobar\" xmlns:flavour=\"http://cereal/ingredients/flavour\">\n" +
320                 "    <base:baseIngredient>wholegrain</base:baseIngredient>\n" +
321                 "    <flavour:flavourIngredient>raisins</flavour:flavourIngredient>\n" +
322                 "    <colour:colourIngredient xsi:nil=\"true\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\"/>\n" +
323                 "</xmlPrefix:rootElementName>";
324 
325         final Marshaller marshaller = ctx.createMarshaller();
326         marshaller.setProperty(Marshaller.JAXB_ENCODING, "UTF-8");
327         marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
328         marshaller.setProperty("com.sun.xml.bind.namespacePrefixMapper", new NamespacePrefixMapper() {
329             @Override
330             public String getPreferredPrefix(final String namespaceUri,
331                                              final String suggestion,
332                                              final boolean requirePrefix) {
333 
334                 // Log the usage.
335                 namespaceToPrefixTupleMap.put(namespaceUri, new Tuple<String, Boolean>(suggestion, requirePrefix));
336 
337                 if (suggestion == null && namespaceUri != null) {
338                     return namespaceUri.equals("")
339                             ? "fooBarz"
340                             : namespaceUri.substring(namespaceUri.lastIndexOf("/") + 1);
341                 }
342 
343                 // We have a suggested namespace prefix. Simply return it.
344                 return suggestion;
345             }
346         });
347 
348         // Act
349         final StringWriter out = new StringWriter();
350         marshaller.marshal(element, new StreamResult(out));
351 
352         // Assert
353         final Diff diff = XmlTestUtils.compareXmlIgnoringWhitespace(expected, out.toString());
354         Assert.assertTrue(diff.identical());
355 
356         for (String current : namespaceToPrefixTupleMap.keySet()) {
357 
358             if (current.equals("http://www.w3.org/2001/XMLSchema-instance")) {
359                 Assert.assertEquals("xsi", namespaceToPrefixTupleMap.get(current).getKey());
360                 Assert.assertEquals(true, namespaceToPrefixTupleMap.get(current).getValue());
361             } else {
362 
363                 // Only the http://www.jguru.se/foobar namespace has a defined prefix via the QName.
364                 final String suppliedPrefix = "http://www.jguru.se/foobar".equals(current) ? "xmlPrefix" : null;
365 
366                 Assert.assertEquals(suppliedPrefix, namespaceToPrefixTupleMap.get(current).getKey());
367                 Assert.assertEquals(false, namespaceToPrefixTupleMap.get(current).getValue());
368             }
369         }
