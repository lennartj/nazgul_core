23  package se.jguru.nazgul.core.algorithms.event.api;
24  
25  import java.util.List;
26  
27  /**
28   * Specification for how to add, remove and list event consumers of a certain type.
29   *
30   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
31   */
32  public interface EventConsumerWrapper<T> {
33  
34      /**
35       * Adds the provided listener to this EventProducer provided that the listener was not already added.
36       * Silently ignores adding an already added EventConsumer instance.
37       *
38       * @param consumer The EventConsumer instance to add.
39       * @return The consumerID of the registered consumer.
40       */
41      String addConsumer(T consumer);
42  
43      /**
44       * @return a non-null List containing unique identifiers for all known/added EventConsumers.
45       * @throws UnsupportedOperationException if this EventProducer cannot not provide the
46       *                                       IDs of all known EventConsumers.
47       */
48      List<String> getConsumerIDs() throws UnsupportedOperationException;
49  
50      /**
51       * Removes the EventConsumer with the given consumerID from this EventProducer.
52       *
53       * @param consumerID The unique identifier of the EventConsumer to remove.
54       * @return {@code true} if the EventConsumer with the given ID was properly removed,
55       * and {@code false} otherwise.
56       */
57      boolean removeConsumer(String consumerID);
58  
59      /**
60       * Acquires the registered EventConsumer with the provided consumerID.
61       *
62       * @param consumerID The unique identifier for the EventConsumer to retrieve
63       * @return the EventConsumer with the given consumerID or {@code null} if none exists.
64       */
65      T getConsumer(String consumerID);
66  }
