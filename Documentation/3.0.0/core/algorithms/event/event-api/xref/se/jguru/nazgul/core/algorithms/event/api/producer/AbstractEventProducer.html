31  import se.jguru.nazgul.core.clustering.api.IdGenerator;

33  import javax.xml.bind.annotation.XmlAccessType;
34  import javax.xml.bind.annotation.XmlAccessorType;
35  import javax.xml.bind.annotation.XmlAttribute;
36  import javax.xml.bind.annotation.XmlType;
37  import java.util.ArrayList;
38  import java.util.Collections;
39  import java.util.List;
40  import java.util.concurrent.ConcurrentHashMap;
41  import java.util.concurrent.ConcurrentMap;

43  /**
44   * Abstract implementation of the EventProducer interface, sporting Clusterable behaviour.
45   * That is - this AbstractEventProducer implementation is intended for use within a cluster.
46   *
47   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
48   */
49  @SuppressWarnings("ValidExternallyBoundObject")
50  @XmlType(namespace = "http://www.jguru.se/nazgul/core", propOrder = {"tClass", "consumers"})
51  @XmlAccessorType(XmlAccessType.FIELD)
52  public abstract class AbstractEventProducer<T extends EventConsumer>
53          extends AbstractClusterable implements EventProducer<T> {
54  
55      // Our log
56      private static final Logger log = LoggerFactory.getLogger(AbstractEventProducer.class);
57  
58      // Internal state
59      @XmlAttribute(required = true)
60      private Class<T> tClass;
61      private ConcurrentMap<String, T> consumers;
62  
63      /**
64       * Creates a new AbstractEventProducer with the provided IdGenerator and EventConsumer type.
65       *
66       * @param idGenerator        The ID generator used to acquire a cluster-unique identifier for
67       *                           this AbstractEventProducer instance.
68       * @param eventConsumerClass The type of EventConsumer handled by this AbstractEventProducer.
69       */
70      public AbstractEventProducer(final IdGenerator idGenerator, final Class<T> eventConsumerClass) {
71  
72          // Delegate
73          super(idGenerator, false);
74  
75          // Check sanity
76          Validate.notNull(eventConsumerClass, "eventConsumerClass");
77  
78          // Assign internal state
79          this.tClass = eventConsumerClass;
80          this.consumers = new ConcurrentHashMap<>();
81      }
82  
83      /**
84       * Creates a new AbstractIdentifiable and assigns the provided
85       * cluster-unique ID to this AbstractClusterable instance.
86       *
87       * @param clusterUniqueID    A cluster-unique Identifier.
88       * @param eventConsumerClass The type of EventConsumer handled by this AbstractEventProducer.
89       */
90      public AbstractEventProducer(final String clusterUniqueID, final Class<T> eventConsumerClass) {
91          this(new ConstantIdGenerator(clusterUniqueID), eventConsumerClass);
92      }
93  
94      /**
95       * {@inheritDoc}
96       */
97      @Override
98      public final Class<T> getConsumerType() {
99          return tClass;
100     }
101 
102     /**
103      * {@inheritDoc}
104      */
105     @Override
106     public final String addConsumer(final T consumer) {
107 
108         String toReturn = null;
109 
110         // Check sanity
111         if (consumer != null) {
112 
113             // Already registered?
114             final String consumerID = consumer.getClusterId();
115             if (consumers.containsKey(consumerID)) {
116                 log.warn("Consumer with id [" + consumerID + "] already registered.");
117             } else if (registerConsumerToEventProducers(consumer)) {
118 
119                 // Register the consumer
120                 consumers.put(consumerID, consumer);
121 
122                 // Notify any listeners about the newly added consumer
123                 try {
124                     onConsumerRegistered(getClusterId(), consumer);
125                 } catch (final Exception e) {
126 
127                     log.error("Could not properly notify other AbstractEventProducer about adding an EventConsumer.",
128                             e);
129                 }
130             }
131 
132             toReturn = consumerID;
133         }
134 
135         // All done.
136         return toReturn;
137     }
138 
139     /**
140      * @return an unmodifiable List holding the IDs of all known EventConsumers.
141      * @throws UnsupportedOperationException if this EventProducer cannot not provide the
142      *                                       IDs of all known EventConsumers.
143      */
144     @Override
145     public final List<String> getConsumerIDs() throws UnsupportedOperationException {
146         return Collections.unmodifiableList(new ArrayList<>(consumers.keySet()));
147     }
148 
149     /**
150      * Removes the EventConsumer with the given consumerID from this EventProducer.
151      *
152      * @param consumerID The unique identifier of the EventConsumer to remove.
153      */
154     @Override
155     public final boolean removeConsumer(final String consumerID) {
156 
157         // Check sanity
158         if (consumerID == null || consumerID.equals("")) {
159             log.warn("Ignoring removing EventConsumer with null or empty ID.");
160             return false;
161         }
162 
163         // Does the listener exist locally?
164         T removed = consumers.remove(consumerID);
165         if (removed != null) {
166 
167             // Delegate to de-register the EventConsumer from its EventProducers.
168             removeConsumerFromEventProducers(removed);
169             return true;
170         }
171 
172         // Send a removeListener event message to the other cluster EventProducers,
173         // to remove the EventConsumer from any of them.
174         return onRemoveNonRegisteredConsumer(getClusterId(), consumerID);
175     }
176 
177     /**
178      * Acquires the registered EventConsumer with the provided consumerID.
179      *
180      * @param consumerID The unique identifier for the EventConsumer to retrieve
181      * @return the EventConsumer with the given consumerID or {@code null} if none exists.
182      */
183     @Override
184     public final T getConsumer(final String consumerID) {
185         return consumerID == null ? null : consumers.get(consumerID);
186     }
187 
188     /**
189      * Adds the EventConsumer to its originating EventProducers, by connecting it to
190      * the underlying EventGenerator.
191      * For example, this is the method where you would call
192      * <code>aButton.addActionListener(listener)</code> or equivalent.
193      *
194      * @param validEventConsumer A non-null EventConsumer which has been validated by this
195      *                           EventProducer WRT its ID.
196      * @return true if the registration process was successful, and false otherwise.
197      */
198     protected boolean registerConsumerToEventProducers(final T validEventConsumer) {
199         return true;
200     }
201 
202     /**
203      * Removes the ListenerType from its proper EventProducers, by removing from the
204      * EventGenerator appropriate for the provided ListenerType. For example, this
205      * is the method where you would call <code>aButton.removeActionListener(listener)</code>
206      * to remove your EventListener from its EventGenerator(s).
207      *
208      * @param validEventConsumer A non-null EventListener which has been validated by this
209      *                           EventProducer WRT its ID.
210      * @return true if the removal process was successful, and false otherwise.
211      */
212     protected boolean removeConsumerFromEventProducers(final T validEventConsumer) {
213         return true;
214     }
215 
216     /**
217      * Override this method to provide implementation notifying other AbstractEventProducer working in
218      * clustered mode with this one about the newly registered EventConsumer.
219      * The default implementation does nothing.
220      *
221      * @param senderID The ID of this AbstractEventProducer.
222      * @param consumer The newly registered EventConsumer.
223      */
224     protected void onConsumerRegistered(final String senderID, final T consumer) {
225         // Do nothing.
226     }
227 
228     /**
229      * <p>Override this method to provide implementation notifying other AbstractEventProducers
230      * working in clustered mode with this one about a remove request for the consumer with
231      * the provided consumerID. The intended usage caters for the following situation:</p>
232      * <pre><code>
233      * // Create 2 clustered EventProducers
234      * AbstractEventProducer&lt;SomeEventConsumer&gt; clusteredEventProducer1 = ...
235      * AbstractEventProducer&lt;SomeEventConsumer&gt; clusteredEventProducer2 = ...
236      *
237      * // Register an EventConsumer
238      * SomeEventConsumer consumer = ...
239      * clusteredEventProducer1.addConsumer(anEventConsumer);
240      *
241      * // Remove the EventConsumer from another node in the cluster
242      * clusteredEventProducer2.removeConsumer(anEventConsumer.getClusterId());
243      * </code>
244      * </pre>
245      * <p>Here, clusteredEventProducer2 must notify clusteredEventProducer1 about the removeConsumer
246      * call and its corresponding ID. ClusteredEventProducer1 removes anEventConsumer from its internal
247      * storage. The default implementation of this method does nothing.</p>
248      *
249      * @param senderID        The ID of this AbstractEventProducer.
250      * @param eventConsumerID The ID of the EventConsumer to remove.
251      * @return {@code true} if the EventConsumer with the given ID was removed, and
252      * {@code false} otherwise.
253      */
254     protected boolean onRemoveNonRegisteredConsumer(final String senderID, final String eventConsumerID) {
255         // Do nothing
256         return false;
257     }
258 
259     /**
260      * If an exception occurs during notification to a listener, an event will be sent to this method.
261      *
262      * @param eventConsumer the listener that the exception occurred in.
263      * @param exception     the exception that occurred.
264      */
265     protected void onExceptionDuringConsumerNotification(final T eventConsumer, final Exception exception) {
266         log.error("Unable to notify EventConsumer '" + eventConsumer.getClusterId() + "'", exception);
267     }
268 
269     /**
270      * {@inheritDoc}
271      */
272     @Override
273     public final void notifyConsumers(final EventConsumerCallback<T> consumerCallback) {
274 
275         // Check sanity
276         Validate.notNull(consumerCallback, "Cannot handle null callback argument.");
277 
278         // Perform notification
279         for (final String current : getConsumerIDs()) {
280             final T listener = getConsumer(current);
281 
282             try {
283                 consumerCallback.onEvent(listener);
284             } catch (final Exception exception) {
285                 onExceptionDuringConsumerNotification(listener, exception);
286             }
287         }
288     }
