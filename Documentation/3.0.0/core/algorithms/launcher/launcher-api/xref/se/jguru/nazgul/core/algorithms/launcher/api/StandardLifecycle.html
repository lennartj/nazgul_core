24  package se.jguru.nazgul.core.algorithms.launcher.api;
25  
26  import org.apache.commons.cli.CommandLine;
27  
28  /**
29   * ApplicationLauncher augmentation specification, defining a known
30   * set of lifecycle methods for implementation by the concrete application.
31   * The StandardLifecycle uses apache's commons CLI to define, parse and
32   * print help for command-line arguments.
33   *
34   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
35   */
36  public interface StandardLifecycle extends ApplicationLauncher {
37  
38      /**
39       * Implement this method to perform any application-specific validation of the supplied CommandLine arguments.
40       * Such validation should focus on argument semantics, since the actual CLI argument parsing is already
41       * performed before invoking this method.
42       *
43       * @param commandLine The application command line.
44       * @throws IllegalArgumentException if any argument was invalid. Supply a human-readable error message
45       *                                  within the IllegalArgumentException, as that argument will be read
46       *                                  by the application launcher.
47       */
48      void validateArguments(final CommandLine commandLine) throws IllegalArgumentException;
49  
50      /**
51       * Main entrypoint to the application, which should execute the standard lifecycle
52       * methods in the following order:
53       * <pre>
54       *   try {
55       *      // First, validate the given CLI arguments.
56       *      validateArguments(commandLine);
57       *
58       *      // Second, fire the actual application.
59       *      runApplication(commandLine);
60       *   } catch (RuntimeException e) {
61       *
62       *      // Whoops.
63       *      printUsage(e.getMessage());
64       *   }
65       * </pre>
66       */
67      void execute();
68  
69      /**
70       * Implement this method to execute the actual application.
71       *
72       * Throw a RuntimeException (or any subclass thereof) to indicate
73       * that the application was terminated improperly. A thrown
74       * RuntimeException is caught and handled by means of running
75       * {@code printUsage} with the exception message passed as message.
76       *
77       * @see #execute()
78       * @see #printUsage(String)
79       */
80      void runApplication();
81  }
