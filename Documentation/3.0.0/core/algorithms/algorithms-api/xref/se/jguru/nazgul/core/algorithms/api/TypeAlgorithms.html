76      /**
77       * Compares two classes by comparing their fully qualified ClassNames.
78       * Any null argument is replaced with an empty string before comparing.
79       */
80      public static final Comparator<Class<?>> CLASSNAME_COMPARATOR =
81              Comparator.comparing(aClass -> (aClass == null ? "" : aClass.getName()));
82  
83      /**
84       * Compares two Annotations by comparing the fully qualified ClassNames of their annotation types.
85       * Any null argument is replaced with an empty string before comparing.
86       *
87       * @see Annotation#annotationType()
88       */
89      public static final Comparator<Annotation> ANNOTATION_COMPARATOR =
90              Comparator.comparing(anAnnotation -> (anAnnotation == null ? "" : anAnnotation.annotationType().getName()));
91  
92      /**
93       * Compares two Members by comparing their respective Declaring Class + toString value.
94       * Any null argument is replaced with an empty string before comparing.
95       */
96      public static final Comparator<Member> MEMBER_COMPARATOR = (l, r) -> {
97  
98          final String leftSortKey = l == null ? "" : l.getDeclaringClass().getName() + l.toString();
99          final String rightSortKey = r == null ? "" : r.getDeclaringClass().getName() + r.toString();
100 
101         // All Done.
102         return leftSortKey.compareTo(rightSortKey);
103     };

177     /**
178      * Function returning all public methods (including those inherited from superclasses) within a given class.
179      * The sort order of the returned SortedSet is given by {@link #MEMBER_COMPARATOR}.
180      */
181     public static final Function<Class<?>, SortedSet<Method>> FIND_PUBLIC_METHODS = c -> {
182 
183         final SortedSet<Method> toReturn = new TreeSet<>(MEMBER_COMPARATOR);
184         toReturn.addAll(Arrays.asList(c.getMethods()));
185 
186         // All Done.
187         return toReturn;
188     };

190     /**
191      * Function retrieving a SortedSet relating readable JavaBean property (names) to
192      * their corresponding getter Methods. The JavaBean getter methods within the Object class
193      * (i.e. "getClass()" will be ignored).
194      */
195     public static final Function<Class<?>, SortedMap<String, Method>> FIND_JAVABEAN_GETTERS = c -> {
196 
197         final SortedMap<String, Method> name2GetterMap = new TreeMap<>();
198 
199         try {
200 
201             // Find the BeanInfo of the current class.
202             final BeanInfo beanInfo = Introspector.getBeanInfo(c, Object.class);
203 
204             // Find the method corresponding to each JavaBean property
205             Arrays.stream(beanInfo.getPropertyDescriptors())
206                     .forEach(propertyDescriptor -> {
207 
208                         final String name = propertyDescriptor.getName();
209                         final Method getter = propertyDescriptor.getReadMethod();
210 
211                         if (getter != null && name != null) {
212 
213                             // Should we overwrite the current name2GetterMap value?
214                             boolean writeProperty;
215 
216                             final Method existingGetterMethod = name2GetterMap.get(name);
217                             if (existingGetterMethod != null) {
218 
219                                 final Class<?> currentDeclaringClass = existingGetterMethod.getDeclaringClass();
220                                 final Class<?> candidateDeclaringClass = getter.getDeclaringClass();
221 
222                                 writeProperty = currentDeclaringClass.isAssignableFrom(candidateDeclaringClass);
223                             } else {
224                                 writeProperty = true;
225                             }
226 
227                             // (Over)write the property?
228                             if (writeProperty) {
229                                 name2GetterMap.put(name, getter);
230                             }
231                         }
232                     });
233         } catch (Exception e) {
234 
235             // Complain
236             throw new IllegalArgumentException("Could not extract BeanInfo/JavaBeanGetters for class "
237                     + c.getName(), e);
238         }
239 
240         // All Done.
241         return name2GetterMap;
242     };

244     /**
245      * Function retrieving a SortedSet relating writable JavaBean property (names) to
246      * their corresponding setter Methods.
247      */
248     public static final Function<Class<?>, SortedMap<String, Method>> FIND_JAVABEAN_SETTERS = c -> {
249 
250         final SortedMap<String, Method> name2SetterMap = new TreeMap<>();
251 
252         try {
253 
254             // Find the BeanInfo of the current class.
255             final BeanInfo beanInfo = Introspector.getBeanInfo(c);
256 
257             // Find the method corresponding to each JavaBean property
258             Arrays.stream(beanInfo.getPropertyDescriptors())
259                     .forEach(propertyDescriptor -> {
260 
261                         final String name = propertyDescriptor.getName();
262                         final Method setter = propertyDescriptor.getWriteMethod();
263 
264                         if (setter != null && name != null && !name2SetterMap.containsKey(name)) {
265                             name2SetterMap.put(name, setter);
266                         }
267                     });
268         } catch (Exception e) {
269 
270             // Complain
271             throw new IllegalArgumentException("Could not extract BeanInfo/JavaBeanSetters for class "
272                     + c.getName(), e);
273         }
274 
275         // All Done.
276         return name2SetterMap;
277     };

302     /**
303      * <p>Retrieves a property from the supplied pointOfOrigin Object using the supplied propertyExpression as the
304      * definition of a property path. The property path should be on the form {@code foo.bar.baz} providing the
305      * JavaBean property names ("foo", "bar" and "baz"). Such JavaBean property names do - in turn - imply methods
306      * with the signature {@code getFoo(), getBar()} and {@code getBaz()} respectively.</p>
307      * <p>The code above should yield an implementation chain similar to the following:</p>
308      * <pre>
309      *     <code>
310      *         // The call:
311      *         final SomeType result = TypeAlgorithms.getProperty(pointOfOrigin, "foo.bar.baz");
312      *
313      *         // ... is equal to calling the JavaBean getter methods in the
314      *         // order given within the propertyExpression:
315      *         final SomeType result = pointOfOrigin.getFoo().getBar().getBaz();
316      *     </code>
317      * </pre>
318      *
319      * @param pointOfOrigin      The object in which we should start the JavaBean invocation chain.
320      * @param propertyExpression The property expression defining a chained
321      * @return The result of the
322      * @throws IllegalArgumentException if a JavaBean getter method is not present within the current/intermediary
323      *                                  type in which it was requested to be invoked.
324      */
325     public static Object getProperty(@NotNull final Object pointOfOrigin,
326                                      @NotNull final String propertyExpression) throws IllegalArgumentException {
327 
328         // #1) Check sanity
329         Validate.notNull(pointOfOrigin, "pointOfOrigin");
330         Validate.notEmpty(propertyExpression, "propertyExpression");
331 
332         // #2) Splice the propertyExpression into tokens, and define the intermediary state
333         final List<String> tokens = splicePropertyExpression(propertyExpression);
334         if (log.isDebugEnabled()) {
335             log.debug("Split propertyExpression [" + propertyExpression + "] into " + tokens.size() + " tokens: "
336                     + tokens);
337         }
338 
339         // #3) Execute the JavaBean invocation chain.
340         Object currentResult = null;
341         final AtomicInteger index = new AtomicInteger();
342         for (Object current = pointOfOrigin; index.get() < tokens.size(); current = currentResult,
343                 index.incrementAndGet()) {
344 
345             // #1) Get the current class in which to invoke a JavaBean getter.
346             final Class<?> currentClass = current.getClass();
347 
348             // #2) Find the current JavaBean property
349             final String currentPropertyName = tokens.get(index.get());
350 
351             // #3) Find the JavaBean getters for this Class, and - more specifically - the Method
352             //     corresponding to the currentPropertyName.
353             final TypeInformation currentTypeInformation = new TypeInformation(currentClass);
354             final SortedMap<String, Method> currentGetters = currentTypeInformation.getJavaBeanGetterMethods();
355             final Method getter = currentGetters.get(currentPropertyName);
356 
357             if (getter == null) {
358                 throw new IllegalArgumentException("Nonexistent expected JavaBean getter for property [" +
359                         currentPropertyName + "] within class [" + currentClass.getName()
360                         + "]. Found JavaBean properties: "
361                         + currentGetters.keySet().stream().reduce((l, r) -> l + ", " + r).orElse("<none>"));
362             }
363 
364             // #4) Invoke the getter, update the result.
365             try {
366                 currentResult = getter.invoke(current);
367             } catch (Exception e) {
368                 throw new IllegalArgumentException("JavaBean property [" + currentPropertyName
369                         + "] getter method call failed in class [" + currentClass.getName() + "]", e);
370             }
371         }
372 
373         // All done.
374         return currentResult;
375     }

377     //
378     // Private helpers
379     //
380     private static List<String> splicePropertyExpression(@NotNull final String propertyExpression) {
381 
382         final List<String> toReturn = new ArrayList<>();
383 
384         final StringTokenizer tok = new StringTokenizer(propertyExpression, PROPERTY_DELIMITER, false);
385         while (tok.hasMoreTokens()) {
386 
387             final String candidate = tok.nextToken().trim();
388 
389             // Check sanity, at least trivially.
390             if (candidate.contains(" ")) {
391                 throw new IllegalArgumentException("PropertyExpressions cannot contain internal whitespace. " +
392                         "(Found token '" + candidate + "' within propertyExpression [" + propertyExpression + "]");
393             }
394 
395             // Add the candidate
396             toReturn.add(candidate);
397         }
398 
399         // All Done.
400         return toReturn;
401     }
