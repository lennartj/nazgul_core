111     @Test
112     public void validateLocalizedLookup() {
113 
114         // Assemble
115         final String fooKey = "foo";
116         final String bazKey = "baz";
117         final Locale defaultLocale = Locale.ENGLISH;
118         final Locale norwegian = WellKnownLocale.NORWEGIAN.getLocale();
119         final Locale italian = Locale.ITALIAN;
120         final ResourceBundleLocalResources unitUnderTest
121                 = new ResourceBundleLocalResources("test/resources/keyvalues", defaultLocale);
122 
123         // Act
124         final String result1 = unitUnderTest.getLocalized(fooKey, "fooFallback");
125         final String result2 = unitUnderTest.getLocalized(fooKey, "fooFallback", WellKnownLocale.SWEDISH.getLocale());
126         final String result3 = unitUnderTest.getLocalized(fooKey, "fooFallback", norwegian);
127         final String result4 = unitUnderTest.getLocalized(fooKey, "fooFallback", italian);
128 
129         final String result5 = unitUnderTest.getLocalized(bazKey, "bazFallback");
130         final String result6 = unitUnderTest.getLocalized(bazKey, "bazFallback", WellKnownLocale.SWEDISH.getLocale());
131         final String result7 = unitUnderTest.getLocalized(bazKey, "bazFallback", norwegian);
132 
133         final String result8 = unitUnderTest.getLocalized("noKey", "noKeyFallback", norwegian);
134         final String result9 = unitUnderTest.getLocalized("noKey", "noKeyFallback", italian);
135 
136         // Assert
137         Assert.assertEquals("bar_en", result1);
138         Assert.assertEquals("bar_sv", result2);
139         Assert.assertEquals("Morrn Da!", result3);
140         Assert.assertEquals("bar_en", result4);
141 
142         Assert.assertEquals("gnat_en", result5);
143         Assert.assertEquals("gnat_sv", result6);
144         Assert.assertEquals("Hei Gutter!", result7);
145 
146         Assert.assertEquals("noKeyFallback", result8);
147         Assert.assertEquals("noKeyFallback", result9);
148     }

150     @Test
151     public void validateSwitchingDefaultLocales() {
152 
153         // Assemble
154         final String fooKey = "foo";
155         final Locale english = Locale.ENGLISH;
156         final Locale norwegian = new Locale("no", "no");
157 
158         final ResourceBundleLocalResources unitUnderTest = new ResourceBundleLocalResources("test/resources/keyvalues", english);
159 
160         // Act
161         final String englishFoo = unitUnderTest.getLocalized(fooKey, "fooDefault");
162         unitUnderTest.setDefaultLocale(WellKnownLocale.SWEDISH.getLocale());
163         final String swedishFoo = unitUnderTest.getLocalized(fooKey, "fooDefault");
164         unitUnderTest.setDefaultLocale(norwegian);
165         final String norwegianFoo = unitUnderTest.getLocalized(fooKey, "fooDefault");
166         unitUnderTest.setDefaultLocale(english);
167         final String anotherEnglishFoo = unitUnderTest.getLocalized(fooKey, "fooDefault");
168 
169         // Assert
170         Assert.assertEquals("bar_en", englishFoo);
171         Assert.assertEquals("bar_sv", swedishFoo);
172         Assert.assertEquals("Morrn Da!", norwegianFoo);
173         Assert.assertEquals("bar_en", anotherEnglishFoo);
174     }

176     @Test
177     public void validateTokenization() {
178 
179         // Assemble
180         final String barKey = "bar";
181         final String defaultValue = "barDefault";
182         final Locale fallback = Locale.ITALIAN;
183         final Locale swedish = WellKnownLocale.SWEDISH.getLocale();
184 
185         final ResourceBundleLocalResources unitUnderTest = new ResourceBundleLocalResources(
186                 "test/tokenizedresources/keyvalues", fallback);
187 
188         // Act & Assert
189         Assert.assertEquals("a ${nice} bar", unitUnderTest.getLocalized(barKey, defaultValue));
190         Assert.assertEquals("a grand bar", unitUnderTest.getLocalized(barKey, defaultValue, "nice=grand"));
191         Assert.assertEquals("en ${nice} bar", unitUnderTest.getLocalized(barKey, defaultValue, swedish));
192         Assert.assertEquals("en kul bar", unitUnderTest.getLocalized(barKey, defaultValue, swedish, "nice=kul"));
193 
194         unitUnderTest.setDefaultLocale(WellKnownLocale.SWEDISH.getLocale());
195         Assert.assertEquals("en ${nice} bar", unitUnderTest.getLocalized(barKey, defaultValue));
196     }

200 
201         // Assemble

203         final String stringValue = "åäöÅÄÖ";
204         final Locale swedish = WellKnownLocale.SWEDISH.getLocale();
205 
206         final ResourceBundleLocalResources unitUnderTest = new ResourceBundleLocalResources("test/resources/utf8chars", Locale.US);

210         Assert.assertEquals(stringValue, unitUnderTest.getLocalized("nonAsciiLetters", null, swedish));
211         Assert.assertEquals("", unitUnderTest.getLocalized("emptyValue", null, swedish));
212         Assert.assertNull(unitUnderTest.getLocalized("nonexistentKey", null, swedish));
213     }
