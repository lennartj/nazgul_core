24  package se.jguru.nazgul.core.resource.api.extractor;

26  import org.apache.commons.lang3.Validate;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import java.io.File;
31  import java.io.FileOutputStream;
32  import java.io.IOException;
33  import java.io.InputStream;
34  import java.net.MalformedURLException;
35  import java.net.URL;
36  import java.nio.channels.Channels;
37  import java.nio.channels.FileChannel;
38  import java.nio.channels.ReadableByteChannel;
39  import java.util.Enumeration;
40  import java.util.jar.JarEntry;
41  import java.util.jar.JarFile;
42  import java.util.regex.Pattern;

44  /**
45   * Trivial utility which extracts resources from a (self-contained) JAR.
46   * Also provides utility methods to simplify finding JarFiles originating from URLs to resources placed within JARs.
47   *
48   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
49   */
50  public abstract class JarExtractor {
51  
52      // Our Log
53      private static final Logger log = LoggerFactory.getLogger(JarExtractor.class);
54  
55      /**
56       * Pattern matching all resource names.
57       */
58      public static final Pattern ALL_RESOURCES = Pattern.compile(".*");
59  
60      // Internal state
61      private static final String JAR_PATH_SEPARATOR = "!";
62  
63      /**
64       * Acquires the JarFile for the supplied packageResourceURL, which should be pointing to a resource packaged
65       * within a JAR. The first/outermost protocol for the supplied packagedResourceURL must be "jar",
66       * as a normal JAR resource URL has the form {@code jar:file:/some/path/aJarFile.jar!/path/in/jar/aFile.txt},
67       * where the path part is operating system dependent. (For instance, the windows counterpart could start with
68       * something like {@code jar:file:/C:/some/path/aJarFile.jar!/path/in/jar/aFile.txt}).
69       *
70       * @param packagedResourceURL a non-null URL pointing to a resource packaged within a JAR.
71       * @return The JarFile for the JAR within which the packagedResourceURL is found.
72       * @throws java.lang.IllegalArgumentException if the supplied URL did not correspond to a resource packaged
73       *                                            within a JAR.
74       */
75      public static JarFile getJarFileFor(final URL packagedResourceURL) throws IllegalArgumentException {
76  
77          // Check sanity
78          Validate.notNull(packagedResourceURL, "Cannot handle null packagedResourceURL argument.");
79  
80          final String jarProtocol = packagedResourceURL.getProtocol();
81          Validate.isTrue("jar".equalsIgnoreCase(jarProtocol),
82                  "packagedResourceURL must have a 'jar' protocol. (Found: " + jarProtocol + ")");
83  
84          // Peel off all protocols, to find the path to the JarFile's File.
85          URL innermostURL = packagedResourceURL;
86          while (true) {
87              try {
88                  innermostURL = new URL(innermostURL.getPath());
89              } catch (MalformedURLException e) {
90                  // Expected
91                  break;
92              }
93          }
94  
95          final String path = innermostURL.getPath();
96          final int exclamationIndex = path.indexOf(JAR_PATH_SEPARATOR);
97          Validate.isTrue(exclamationIndex > 0, "Required JAR path separator [" + JAR_PATH_SEPARATOR
98                  + "] not found in URLs path [" + path + "], distilled from packagedResourceURL ["
99                  + packagedResourceURL.toString() + "].");
100 
101         final File jarFileFile = new File(path.substring(0, exclamationIndex));
102         Validate.isTrue(jarFileFile.exists() && jarFileFile.isFile(), "Inconsistent JarFile ["
103                 + jarFileFile.getAbsolutePath() + "]: nonexistent or not a File, distilled from packagedResourceURL ["
104                 + packagedResourceURL.toString() + "].");
105 
106         // All seems well.
107         try {
108             return new JarFile(jarFileFile);
109         } catch (IOException e) {
110             throw new IllegalArgumentException("Could not create a JarFile from File ["
111                     + jarFileFile.getAbsolutePath() + "]", e);
112         }
113     }
114 
115     /**
116      * Retrieves the JarEntry name for the supplied packagedResourceURL. Typically,
117      * this value can be used to acquire a JarEntry for the
118      * <pre>
119      *     <code>
120      *         // Get the JarFile for a JAR-based URL
121      *         final JarFile foundJarFile = JarExtractor.getJarFileFor(resourceInJarURL);
122      *
123      *         // Now, find the JarEntry name for the given resourceInJarURL
124      *         final String name = JarExtractor.getEntryNameFor(resourceInJarURL);
125      *
126      *         // Get the JarEntry for the resourceInJarURL
127      *         final JarEntry entry = foundJarFile.getJarEntry(name);
128      *     </code>
129      * </pre>
130      *
131      * @param packagedResourceURL A JAR URL, which must contain an '!' char. Typically on a form similar to
132      *                            {@code jar:file:/some/path/aJarFile.jar!/path/in/jar/aFile.txt}
133      * @return The entry name of the URL, which is the part following the '!/' char. Note that the first '/' must be
134      * peeled off to retrieve a valid JarEntry name.
135      */
136     public static String getEntryNameFor(final URL packagedResourceURL) {
137 
138         // Check sanity
139         Validate.notNull(packagedResourceURL, "Cannot handle null packagedResourceURL argument.");
140 
141         // Find the path for the supplied packagedResourceURL
142         final String urlString = packagedResourceURL.toString();
143         final String tmp = urlString.substring(urlString.indexOf(JAR_PATH_SEPARATOR) + 1);
144 
145         // Peel off the initial '/' to make a valid name.
146         Validate.isTrue(tmp.startsWith("/"), "The absolute path within the JAR should start with a '/' char. Got ["
147                 + tmp + "]");
148         return tmp.substring(1);
149     }
150 
151     /**
152      * Extracts all resources whose names matches the supplied resourceIdentifier from jarFile
153      * to the targetDirectory. If targetDirectory does not exist and the createTargetDirectoryIfNonexistent
154      * parameter is {@code true}, the directory (and any parent directories) will be created.
155      *
156      * @param jarFile                            The JAR file from which some resources should be extracted.
157      * @param resourceIdentifier                 A Pattern matching the names of all resources which should be extracted.
158      * @param targetDirectory                    The directory to which all resources should be extracted.
159      * @param createTargetDirectoryIfNonexistent if {@code true}, the targetDirectory will be
160      *                                           created if it does not already exist.
161      */
162     public static void extractResourcesFrom(final JarFile jarFile,
163                                             final Pattern resourceIdentifier,
164                                             final File targetDirectory,
165                                             final boolean createTargetDirectoryIfNonexistent) {
166 
167         // Check sanity
168         Validate.notNull(jarFile, "Cannot handle null jarFile argument.");
169         Validate.notNull(resourceIdentifier, "Cannot handle null resourceIdentifier argument.");
170         Validate.notNull(targetDirectory, "Cannot handle null targetDirectory argument.");
171 
172         if (targetDirectory.exists() && !targetDirectory.isDirectory()) {
173             throw new IllegalArgumentException("Target [" + targetDirectory.getAbsolutePath()
174                     + "] exists and is not a directory.");
175         }
176 
177         if (!createTargetDirectoryIfNonexistent && !targetDirectory.exists()) {
178             throw new IllegalArgumentException("Target directory [" + targetDirectory.getAbsolutePath()
179                     + "] does not exist - and instructed not to create it.");
180         }
181 
182         // Do we need to create targetDir?
183         if (!targetDirectory.exists()) {
184             if (!targetDirectory.mkdirs()) {
185                 throw new IllegalStateException("Could not create directory ["
186                         + targetDirectory.getAbsolutePath() + "]");
187             } else {
188                 targetDirectory.mkdirs();
189             }
190         }
191 
192         for (Enumeration<JarEntry> en = jarFile.entries(); en.hasMoreElements(); ) {
193 
194             // Dig out the entry and its internal path.
195             final JarEntry current = en.nextElement();
196             final String entryPath = current.getName();
197 
198             if (!current.isDirectory() && resourceIdentifier.matcher(entryPath).matches()) {
199 
200                 // Extract the file at its relative path location.
201                 final File toWrite = new File(targetDirectory, entryPath);
202                 final File parentFile = toWrite.getParentFile();
203 
204                 if (!parentFile.exists()) {
205                     parentFile.mkdirs();
206                 }
207 
208                 log.debug("Extracting [" + current.getName() + "] to [" + toWrite.getAbsolutePath() + "]");
209 
210                 InputStream inStream = null;
211                 ReadableByteChannel inChannel = null;
212                 FileOutputStream outStream = null;
213                 FileChannel outChannel = null;
214 
215                 try {
216                     try {
217 
218                         // Copy using a NIO channel to improve performance.
219                         inStream = jarFile.getInputStream(current);
220                         inChannel = Channels.newChannel(inStream);
221                         outStream = new FileOutputStream(toWrite);
222                         outChannel = outStream.getChannel();
223                         outChannel.transferFrom(inChannel, 0, current.getSize());
224 
225                     } finally {
226 
227                         // Close all opened NIO objects
228                         if (inStream != null) {
229                             inStream.close();
230                         }
231                         if (inChannel != null) {
232                             inChannel.close();
233                         }
234                         if (outStream != null) {
235                             outStream.close();
236                         }
237                         if (outChannel != null) {
238                             outChannel.close();
239                         }
240                     }
241                 } catch (IOException e) {
242                     throw new IllegalStateException("Could not create copy [" + entryPath + "] to ["
243                             + targetDirectory.getAbsolutePath() + "]", e);
244                 }
245             }
246         }
247     }
