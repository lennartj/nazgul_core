23  package se.jguru.nazgul.core.resource.api;
24  
25  import java.util.Locale;
26  
27  /**
28   * Enumeration of well-known Locale definitions.
29   * Convenience enumeration to have said data available immediately.
30   *
31   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
32   */
33  public enum WellKnownLocale {
34  
35      /**
36       * Swedish locale.
37       */
38      SWEDISH("sv", "SE"),
39  
40      /**
41       * Norwegian locale.
42       */
43      NORWEGIAN("no", "NO"),
44  
45      /**
46       * Plain/international english locale.
47       */
48      ENGLISH("en", "EN"),
49  
50      /**
51       * American english locale.
52       */
53      US("en", "US"),
54  
55      /**
56       * British english locale.
57       */
58      BRITISH("en", "GB");
59  
60      // Internal state
61      private Locale activeLocale;
62  
63      /**
64       * Construct a locale from language, country, variant.
65       * NOTE: ISO 639 is not a stable standard; some of the language codes it defines
66       * (specifically iw, ji, and in) have changed.  This constructor accepts both the
67       * old codes (iw, ji, and in) and the new codes (he, yi, and id), but all other
68       * API on Locale will return only the OLD codes.
69       *
70       * @param language lowercase two-letter ISO-639 code.
71       * @param country  uppercase two-letter ISO-3166 code.
72       * @param variant  vendor and browser specific code. See class description.
73       * @throws NullPointerException thrown if any argument is null.
74       */
75      private WellKnownLocale(final String language,
76                              final String country,
77                              final String variant) {
78  
79          // Create internal state.
80          this.activeLocale = new Locale(language, country, variant);
81      }
82  
83      /**
84       * Construct a locale from language, country.
85       * NOTE: ISO 639 is not a stable standard; some of the language codes it defines
86       * (specifically iw, ji, and in) have changed.  This constructor accepts both the
87       * old codes (iw, ji, and in) and the new codes (he, yi, and id), but all other
88       * API on Locale will return only the OLD codes.
89       *
90       * @param language lowercase two-letter ISO-639 code.
91       * @param country  uppercase two-letter ISO-3166 code.
92       * @throws NullPointerException thrown if any argument is null.
93       */
94      private WellKnownLocale(final String language,
95                              final String country) {
96          this(language, country, "");
97      }
98  
99      /**
100      * @return The java.util.Locale for this WellKnownLocales instance.
101      */
102     public final Locale getLocale() {
103         return activeLocale;
104     }
105 }
