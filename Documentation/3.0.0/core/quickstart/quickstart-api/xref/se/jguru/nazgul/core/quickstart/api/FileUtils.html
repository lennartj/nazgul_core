69      /**
70       * A FileFilter which accepts directories only.
71       */
72      public static final FileFilter DIRECTORY_FILTER = new FileFilter() {
73          @Override
74          public boolean accept(final File pathname) {
75  
76              System.getProperties();
77              return pathname.exists() && pathname.isDirectory();
78          }
79      };

81      /**
82       * A FileFilter which accepts files only.
83       */
84      public static final FileFilter FILE_FILTER = new FileFilter() {
85          @Override
86          public boolean accept(final File pathname) {
87              return pathname.exists() && pathname.isFile();
88          }
89      };
90  
91      /**
92       * A FileFilter which accepts Maven module directories only (i.e. directories containing a 'pom.xml' file).
93       */
94      public static final FileFilter MODULE_NAME_FILTER = new FileFilter() {
95          @Override
96          public boolean accept(final File moduleCandidate) {
97              if (moduleCandidate.exists() && moduleCandidate.isDirectory()) {
98                  return FILE_FILTER.accept(new File(moduleCandidate, "pom.xml"));
99              }
100 
101             // Not a module directory/name.
102             return false;
103         }
104     };
105 
106     /**
107      * A FileFilter which accepts nonexistent or empty directories.
108      * For this FileFilter, an "empty" directory implies that no directories or files are present within a successful
109      * candidate. All files except those whose names starts with "." are considered files.
110      */
111     public static final FileFilter NONEXISTENT_OR_EMPTY_DIRECTORY_FILTER = new FileFilter() {
112         @Override
113         public boolean accept(final File candidate) {
114 
115             boolean okCandidate = !candidate.exists();
116             if (!okCandidate && DIRECTORY_FILTER.accept(candidate)) {
117 
118                 final File[] childFiles = candidate.listFiles();
119                 if (childFiles != null && childFiles.length != 0) {
120                     for (File current : childFiles) {
121                         if (!current.getName().startsWith(".")) {
122                             return false;
123                         }
124                     }
125                 }
126 
127                 // All seems well.
128                 okCandidate = true;
129             }
130 
131             // All done.
132             return okCandidate;
133         }
134     };

136     /**
137      * A list containing file suffixes of files normally containing character data, implying that the
138      * content of such files can normally be manipulated using TokenParsers.
139      */
140     public static final List<String> CHARDATA_FILE_SUFFIXES = Arrays.asList("txt", "text", "xml", "xsd", "properties",
141             "apt", "log", "md", "readme", "csv", "tab", "odt", "1st", "java", "jsp", "js", "jsf", "c", "cpp", "html",
142             "css", "js", "less", "scss");
143 
144     /**
145      * A FileFilter which identifies file types normally holding character data content.
146      */
147     public static final FileFilter CHARACTER_DATAFILE_FILTER = new FileFilter() {
148 
149         /**
150          * Accepts aFile if it is a file whose suffix is found in the CHARDATA_FILE_SUFFIXES list.
151          *
152          * @param aFile The non-null File to check.
153          * @return {@code true} if the supplied aFile is a file whose suffix is found in
154          * the {@code CHARDATA_FILE_SUFFIXES} list.
155          */
156         @Override
157         public boolean accept(final File aFile) {
158 
159             if (FILE_FILTER.accept(aFile)) {
160 
161                 // Find the file suffix to compare with.
162                 final String fileName = aFile.getName();
163                 final int suffixDelimiterIndex = fileName.lastIndexOf(".");
164                 final String fileSuffix = "" + (suffixDelimiterIndex == -1
165                         ? fileName
166                         : fileName.substring(suffixDelimiterIndex + 1)).trim().toLowerCase();
167 
168                 // All done.
169                 return CHARDATA_FILE_SUFFIXES.contains(fileSuffix);
170             }
171 
172             // not recognized as a CharacterData-based file.
173             return false;
174         }
175     };

187     /**
188      * Gets the canonical path of the supplied (non-null) File.
189      *
190      * @param fileOrDirectory A non-null File object.
191      * @return The canonical path to the supplied fileOrDirectory.
192      */
193     public static String getCanonicalPath(final File fileOrDirectory) {
194 
195         // Check sanity
196         Validate.notNull(fileOrDirectory, "Cannot handle null or empty fileOrDirectory argument.");
197 
198         try {
199             return fileOrDirectory.getCanonicalPath();
200         } catch (IOException e) {
201             throw new IllegalArgumentException("Could not acquire canonical path for [" + fileOrDirectory + "]", e);
202         }
203     }

205     /**
206      * Creates all intermediary directories from the rootDirectory to the leafDirectory
207      * given by the relativePath supplied.
208      *
209      * @param rootDirectory The existent or nonexistent rootDirectory.
210      * @param relativePath  The relative path of directories to make.
211      * @return The directory just made.
212      * @throws java.lang.IllegalArgumentException if the leafDirectory (or any of its intermediary directories)
213      *                                            could not be created.
214      */
215     public static File makeDirectory(final File rootDirectory, final String relativePath)
216             throws IllegalArgumentException {
217 
218         // Check sanity
219         Validate.notNull(rootDirectory, "Cannot handle null rootDirectory argument.");
220         Validate.notNull(relativePath, "Cannot handle null relativePath argument.");
221 
222         final File toReturn = relativePath.isEmpty() ? rootDirectory : new File(rootDirectory, relativePath);
223         Validate.isTrue(!toReturn.exists() || (toReturn.exists() && toReturn.isDirectory()),
224                 "Insane state. [" + getCanonicalPath(toReturn) + "] exists but is not a Directory.");
225 
226         // Delegate and return
227         if (toReturn.mkdirs()) {
228             return toReturn;
229         }
230 
231         // Should we fail?
232         if (!exists(toReturn, true)) {
233             throw new IllegalArgumentException("Could not create path to [" + getCanonicalPath(toReturn)
234                     + "] fully. Check filesystem; state may be broken.");
235         }
236 
237         // Seems we are OK anyways.
238         return toReturn;
239     }

241     /**
242      * Acquires a Maven Model from the supplied POM file.
243      *
244      * @param aPomFile A non-null pom.xml File.
245      * @return The Maven model converted from the supplied aPomFile.
246      */
247     public static Model getPomModel(final File aPomFile) {
248 
249         // Check sanity
250         Validate.notNull(aPomFile, "Cannot handle null aPomFile argument.");
251         Validate.isTrue(FILE_FILTER.accept(aPomFile), "File [" + getCanonicalPath(aPomFile)
252                 + "] must exist and be a File.");
253 
254         try {
255             return pomReader.read(new FileReader(aPomFile));
256         } catch (Exception e) {
257             throw new IllegalArgumentException("Could not read POM file [" + getCanonicalPath(aPomFile) + "]", e);
258         }
259     }

261     /**
262      * Given a directory within a Maven project reactor, list the names of all subdirectories
263      * containing a 'pom.xml' file. These directories typically correspond to module names within a
264      * reactor POM found in the supplied reactorDirectory.
265      *
266      * @param reactorDirectory An existing directory.
267      * @return The names of all subdirectories to the provided reactorDirectory where a file called "pom.xml" exists.
268      * The names of these directories should typically be used as module names within a reactor pom located within
269      * the supplied reactorDirectory. No validation of any found pom.xml files (consistency, well-formed-ness etc.)
270      * are done within this method.
271      */
272     public static List<String> getModuleNames(final File reactorDirectory) {
273 
274         // Check sanity
275         Validate.notNull(reactorDirectory, "Cannot handle null or empty reactorDirectory argument.");
276         Validate.isTrue(DIRECTORY_FILTER.accept(reactorDirectory), "reactorDirectory argument ["
277                 + getCanonicalPath(reactorDirectory) + "] must refer to an existing directory.");
278 
279         final List<String> toReturn = new ArrayList<>();
280         for (File current : reactorDirectory.listFiles(MODULE_NAME_FILTER)) {
281             toReturn.add(current.getName());
282         }
283 
284         // All done.
285         return toReturn;
286     }

288     /**
289      * Extracts a SimpleArtifact from a Maven model.
290      *
291      * @param aModel The Maven Model from which to extract the SimpleArtifact data.
292      * @return A SimpleArtifact wrapping the data found in the supplied aModel.
293      */
294     public static SimpleArtifact getSimpleArtifact(final Model aModel) {
295 
296         // Check sanity
297         Validate.notNull(aModel, "Cannot handle null aModel argument.");
298 
299         String version = aModel.getVersion();
300         if (version == null) {
301             Validate.notNull(aModel.getParent(), "A Model requires either a version or a non-null Parent definition.");
302             version = aModel.getParent().getVersion();
303         }
304 
305         // All done.
306         return new SimpleArtifact(aModel.getGroupId(), aModel.getArtifactId(), version);
307     }

309     /**
310      * Identifies if the supplied fileOrDir exists.
311      *
312      * @param fileOrDir   A non-null File.
313      * @param isDirectory if {@code true}, the fileOrDir is assumed to be a directory - and otherwise a file.
314      * @return {@code true} if the supplied fileOrDir exists and is of the type indicated by {@code isDirectory}.
315      */
316     public static boolean exists(final File fileOrDir, final boolean isDirectory) {
317 
318         Validate.notNull(fileOrDir, "Cannot handle null fileOrDir argument.");
319         return isDirectory ? DIRECTORY_FILTER.accept(fileOrDir) : FILE_FILTER.accept(fileOrDir);
320     }

322     /**
323      * Writes the supplied data to the given File.
324      *
325      * @param aFile The non-null File to write data to.
326      * @param data  The non-null data string to write.
327      */
328     public static void writeFile(final File aFile, final String data) {
329 
330         // Check sanity
331         Validate.notNull(aFile, "Cannot handle null aFile argument.");
332         Validate.notNull(data, "Cannot handle null data argument.");
333 
334         final File dirForFile = aFile.getParentFile();
335         if (dirForFile == null || !(dirForFile.exists() && dirForFile.isDirectory())) {
336             throw new IllegalArgumentException("Cannot write file [" + FileUtils.getCanonicalPath(aFile)
337                     + "], since its parent is not an existing directory.");
338         }
339 
340         // All seems sane. Write the file.
341         try (BufferedWriter writer = new BufferedWriter(new FileWriter(aFile))) {
342             writer.write(data);
343             writer.flush();
344         } catch (IOException e) {
345             log.warn("Could not write data to [" + getCanonicalPath(aFile) + "]", e);
346         }
347     }

349     /**
350      * Reads all (text) data from the supplied File, returning it as a String.
351      * All line feeds are converted to {@code System.getProperty("line.separator")}.
352      *
353      * @param aFile The non-null File to read data from.
354      * @return The content of the supplied File.
355      */
356     public static String readFile(final File aFile) {
357 
358         // Check sanity
359         Validate.notNull(aFile, "Cannot handle null aFile argument.");
360         Validate.isTrue(FILE_FILTER.accept(aFile), "File [" + getCanonicalPath(aFile)
361                 + "] must exist and be a (text) file.");
362 
363         try {
364             return readFully(new FileInputStream(aFile), getCanonicalPath(aFile));
365         } catch (FileNotFoundException e) {
366 
367             // This should never happen
368             throw new IllegalArgumentException("Could not read file", e);
369         }
370     }

372     /**
373      * Reads all (text) data from the supplied resource URL, returning it as a String.
374      * All line feeds are converted to {@code System.getProperty("line.separator")}.
375      *
376      * @param resourceURL The non-empty resource URL to read data from.
377      * @return The content of the supplied resource URL.
378      */
379     public static String readFile(final String resourceURL) {
380 
381         // Check sanity
382         Validate.notEmpty(resourceURL, "Cannot handle null resourceURL argument.");
383 
384         // Peel off any initial '/' chars
385         final String effectiveURL = resourceURL.startsWith("/") ? resourceURL.substring(1) : resourceURL;
386         final InputStream in = Thread.currentThread().getContextClassLoader().getResourceAsStream(effectiveURL);
387         if (in == null) {
388             throw new IllegalArgumentException("No file (or stream) found for resource [" + effectiveURL + "].");
389         }
390 
391         // All done.
392         return readFully(in, resourceURL);
393     }

395     /**
396      * Maps the relative path of all files found under the supplied aDirectory to the files themselves.
397      * Note that this operation consumes considerable amounts of resources (memory buffers) when mapping large file
398      * trees. Therefore, it is wise to confine the aDirectory to smaller file trees.
399      *
400      * @param aDirectory A directory under which all Files are mapped.
401      * @return A non-null SortedMap relating the relative paths of all Files found under the supplied aDirectory
402      * to the Files themselves.
403      */
404     public static SortedMap<String, File> listFilesRecursively(final File aDirectory) {
405 
406         // Check sanity
407         Validate.notNull(aDirectory, "Cannot handle null aDirectory argument.");
408         Validate.isTrue(DIRECTORY_FILTER.accept(aDirectory),
409                 "Argument aDirectory must point to an existing directory. Got [" + getCanonicalPath(aDirectory) + "]");
410 
411         final SortedMap<String, File> toReturn = new TreeMap<>();
412         populate(toReturn, aDirectory, aDirectory);
413 
414         // All done.
415         return toReturn;
416     }

422     /**
423      * Reads all data from the supplied InputStream, assumed to point to a character-based stream.
424      * All line feeds are converted to {@code System.getProperty("line.separator")}.
425      *
426      * @param stream The non-null stream to read fully. The stream is closed before returning from this method,
427      *               irrespective of the results of reading the stream.
428      * @param desc   A description of the stream. Used within an Exception message should an IOException
429      *               occur while reading the stream data.
430      * @return The fully read text data.
431      */
432     @SuppressWarnings("all")
433     private static String readFully(final InputStream stream, final String desc) {
434 
435         final StringBuilder result = new StringBuilder();
436 
437         try (BufferedReader reader = new BufferedReader(new InputStreamReader(stream))) {
438             for (String aLine = reader.readLine(); aLine != null; aLine = reader.readLine()) {
439                 result.append(aLine).append(LINE_ENDING);
440             }
441         } catch (IOException e) {
442             throw new IllegalArgumentException("Could not read data from [" + desc + "]", e);
443         } finally {
444 
445             try {
446                 // Close the original stream.
447                 stream.close();
448             } catch (IOException e) {
449                 throw new IllegalArgumentException("Could not close original stream for [" + desc + "]", e);
450             }
451         }
452 
453         // All done.
454         return result.toString();
455     }

457     /**
458      * Populates the provided SortedMap with the relative path of each File found under the rootDirectory,
459      *
460      * @param toPopulate       The SortedMap to populate.
461      * @param currentDirectory The current directory.
462      * @param rootDirectory    The root directory of the structure to populate, used to calculate the relative
463      *                         path of the files found within the currentDirectory (i.e. the keys within the
464      *                         toPopulate SortedMap).
465      */
466     private static void populate(final SortedMap<String, File> toPopulate,
467                                  final File currentDirectory,
468                                  final File rootDirectory) {
469 
470         // Calculate the relative path
471         final String rootDirPath = getCanonicalPath(rootDirectory);
472         final String currentDirPath = getCanonicalPath(currentDirectory);
473         final String tmp = currentDirPath.substring(currentDirPath.indexOf(rootDirPath)
474                 + rootDirPath.length()
475                 + (rootDirPath.equals(currentDirPath) ? 0 : 1));
476         final String prefix = tmp.length() == 0 ? "" : tmp + "/";
477 
478         // Map all files in the current directory
479         for (File current : currentDirectory.listFiles(FILE_FILTER)) {
480             final File existingFile = toPopulate.put(prefix + current.getName(), current);
481             Validate.isTrue(existingFile == null, "Already mapped file at path [" + getCanonicalPath(current) + "]");
482         }
483 
484         // Recurse
485         for (File current : currentDirectory.listFiles(DIRECTORY_FILTER)) {
486             populate(toPopulate, current, rootDirectory);
487         }
488     }
