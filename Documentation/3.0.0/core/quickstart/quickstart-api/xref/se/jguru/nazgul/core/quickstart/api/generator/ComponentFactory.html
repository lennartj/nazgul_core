23  package se.jguru.nazgul.core.quickstart.api.generator;
24  
25  import se.jguru.nazgul.core.quickstart.api.InvalidStructureException;
26  
27  import java.io.File;
28  import java.util.SortedMap;
29  
30  /**
31   * Specification for how to create a new Software Component, which consists of several collaborating Maven
32   * projects within the same reactor.
33   *
34   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
35   */
36  public interface ComponentFactory {
37  
38      /**
39       * Creates a new SoftwareComponent containing the supplied parts in the given componentDirectory.
40       *
41       * @param componentDirectory An non-existent or empty directory where a SoftwareComponent should be
42       *                           created. The directory name will be identical to the SoftwareComponent name,
43       *                           and must therefore be compliant with the NamingStrategy in use.
44       * @param parts2SuffixMap    A non-empty Map relating SoftwareComponentParts to be created within the
45       *                           SoftwareComponent to their respective suffix. A non-empty value is only required
46       *                           within the Map if {@code SoftwareComponentPart.isSuffixRequired()} yields {@code true}.
47       * @throws InvalidStructureException If {@code componentDirectory} already contained a POM,
48       *                                   or was not compliant with the NamingStrategy chosen. Also thrown if the parts
49       *                                   list is empty or holds an Implementation SoftwareComponentPart but neither
50       *                                   API nor SPI SoftwareComponentParts.
51       */
52      void createSoftwareComponent(File componentDirectory,
53                                   SortedMap<SoftwareComponentPart, String> parts2SuffixMap)
54              throws InvalidStructureException;
55  
56      /**
57       * Adds a SoftwareComponentPart to the SoftwareComponent found within the supplied componentDirectory.
58       *
59       * @param componentDirectory A directory containing an existing SoftwareComponent
60       *                           (i.e. containing the reactor POM of the SoftwareComponent).
61       * @param toAdd              The SoftwareComponentPart to add to the supplied SoftwareComponent.
62       * @param suffix             The suffix of the SoftwareComponent's type. Ignored unless
63       *                           {@code toAdd.isSuffixRequired()} yields {@code true}.
64       * @throws InvalidStructureException if the supplied componentDirectory did not contain a SoftwareComponent,
65       *                                   or if the toAdd SoftwareComponentPart was already present within
66       *                                   the {@code componentDirectory} directory.
67       * @see SoftwareComponentPart#isSuffixRequired()
68       */
69      void addSoftwareComponentPart(File componentDirectory,
70                                    SoftwareComponentPart toAdd,
71                                    String suffix)
72              throws InvalidStructureException;
73  }
