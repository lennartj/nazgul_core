42  /**
43   * Abstract implementation of the ProjectFactory sporting
44   *
45   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
46   */
47  public abstract class AbstractProjectFactory extends AbstractFactory implements ProjectFactory {
48  
49      // Our log
50      private static final Logger log = LoggerFactory.getLogger(AbstractProjectFactory.class.getName());
51  
52      /**
53       * Creates a new AbstractProjectFactory wrapping the supplied data.
54       *
55       * @param namingStrategy The active NamingStrategy, used to validate given Project data.
56       */
57      protected AbstractProjectFactory(final NamingStrategy namingStrategy) {
58          super(namingStrategy);
59      }
60  
61      /**
62       * {@inheritDoc}
63       */
64      @Override
65      public boolean createProject(final File projectParentDir,
66                                   final Project project,
67                                   final String packagePrefix,
68                                   final String reactorParentMavenVersion,
69                                   final String topmostParentMavenVersion)
70              throws IllegalArgumentException, IllegalStateException {
71  
72          // Check sanity
73          Validate.notNull(project, "Cannot handle null projectDefinition argument.");
74          Validate.notNull(projectParentDir, "Cannot handle null projectParentDir argument.");
75          Validate.isTrue(projectParentDir.exists() && projectParentDir.isDirectory(),
76                  "Project parent directory [" + FileUtils.getCanonicalPath(projectParentDir)
77                          + "] must exist and be a directory.");
78          Validate.notEmpty(reactorParentMavenVersion, "Cannot handle null or empty reactorParentMavenVersion argument.");
79          Validate.notEmpty(topmostParentMavenVersion, "Cannot handle null or empty topmostParentMavenVersion argument.");
80  
81          final String reverseDNS = packagePrefix == null ? "" : packagePrefix.replace("/", ".");
82  
83          // Synthesize the Name from the supplied project, and validate it.
84          final Name projectName = new Name(
85                  project.getPrefix(),
86                  project.getName(),
87                  AbstractNamingStrategy.REACTOR_SUFFIX);
88          getNamingStrategy().validate(projectName, PomType.ROOT_REACTOR);
89  
90          // Find the directory of the new project, which should be on the form [parent]/[name].
91          final String projectRootDirName = getNamingStrategy().isPrefixRequiredOnAllFolders()
92                  ? project.getPrefix() + Name.DEFAULT_SEPARATOR + project.getName()
93                  : project.getName();
94          final File projectRootDir = new File(projectParentDir, projectRootDirName);
95          if (log.isInfoEnabled()) {
96              log.info("Using project root directory [" + projectRootDir.getAbsolutePath() + "].");
97          }
98          if (!FileUtils.exists(projectRootDir, true) && !projectRootDir.mkdirs()) {
99              throw new IllegalStateException("Project root directory [" + FileUtils.getCanonicalPath(projectRootDir)
100                     + "] was nonexistent and could not be created.");
101         }
102 
103         // Create the project structure.
104         createParentPom(projectRootDir, "", PomType.ROOT_REACTOR, project,
105                 reactorParentMavenVersion, topmostParentMavenVersion, reverseDNS);
106         createParentPom(projectRootDir, "poms", PomType.PARENT, project,
107                 reactorParentMavenVersion, topmostParentMavenVersion, reverseDNS);
108         createParentPom(projectRootDir, "poms", PomType.API_PARENT, project,
109                 reactorParentMavenVersion, topmostParentMavenVersion, reverseDNS);
110         createParentPom(projectRootDir, "poms", PomType.MODEL_PARENT, project,
111                 reactorParentMavenVersion, topmostParentMavenVersion, reverseDNS);
112         createParentPom(projectRootDir, "poms", PomType.WAR_PARENT, project,
113                 reactorParentMavenVersion, topmostParentMavenVersion, reverseDNS);
114         createParentPom(projectRootDir, "poms", PomType.REACTOR, project,
115                 reactorParentMavenVersion, topmostParentMavenVersion, reverseDNS);
116 
117         // All done.
118         return true;
119     }

121     /**
122      * {@inheritDoc}
123      */
124     @Override
125     public Project createProjectDefinition(final String prefix,
126                                            final String name,
127                                            final SimpleArtifact reactorParent,
128                                            final SimpleArtifact parentParent) {
129         // Check sanity
130         Validate.notEmpty(name, "Cannot handle null or empty name argument.");
131         Validate.notNull(reactorParent, "Cannot handle null reactorParent argument.");
132         Validate.notNull(parentParent, "Cannot handle null parentParent argument.");
133 
134         // All done.
135         return new Project(prefix, name, reactorParent, parentParent);
136     }
137 
138     /**
139      * Synthesizes the directory name from the supplied Project and PomType.
140      * The resulting name should adhere to the NamingStrategy's
141      * {@code NamingStrategy.isPrefixRequiredOnAllFolders()} property.
142      *
143      * @param projectName   The name of the active project.
144      * @param projectPrefix The optional prefix of the active project. Must be supplied (and non-empty) if
145      *                      the NamingStrategy requires prefixes to be present on all folders' names.
146      * @param aPomType      The PomType for which a directory name should be synthesized.
147      * @return The name of the directory.
148      */
149     protected String getSkeletonPomDirectoryName(final String projectName,
150                                                  final PomType aPomType,
151                                                  final String projectPrefix) {
152 
153         // Check sanity
154         Validate.notNull(aPomType, "Cannot handle null aPomType argument.");
155         Validate.notEmpty(projectName, "Cannot handle null or empty projectName argument.");
156         if (getNamingStrategy().isPrefixRequiredOnAllFolders() && (projectPrefix == null || projectPrefix.isEmpty())) {
157             throw new IllegalArgumentException("NamingStrategy [" + getNamingStrategy().getClass().getSimpleName()
158                     + "] requires that project prefix is required on all folders. Therefore, "
159                     + "a nonempty projectPrefix argument must be supplied.");
160         }
161 
162         // Reactor directories should only be called their respective component name.
163         if (aPomType == PomType.ROOT_REACTOR || aPomType == PomType.REACTOR) {
164             return getNamingStrategy().isPrefixRequiredOnAllFolders()
165                     ? projectPrefix : "";
166         }
167 
168         final String dirPrefix = (getNamingStrategy().isPrefixRequiredOnAllFolders()
169                 ? projectPrefix + Name.DEFAULT_SEPARATOR + projectName
170                 : projectName)
171                 + Name.DEFAULT_SEPARATOR;
172 
173         String dirSuffix = aPomType.name().toLowerCase().replace("_", "-");
174         if (aPomType == PomType.OTHER_PARENT) {
175             dirSuffix = AbstractNamingStrategy.PARENT_SUFFIX;
176         }
177 
178         final String toReturn = dirPrefix + dirSuffix;
179         if (log.isDebugEnabled()) {
180             log.debug("PomType [" + aPomType + "] yields directory name [" + toReturn + "]");
181         }
182 
183         // All done.
184         return toReturn;
185     }

191     @SuppressWarnings("all")
192     private void createParentPom(final File rootDirectory,
193                                  final String relativePomDirectory,
194                                  final PomType pomType,
195                                  final Project project,
196                                  final String reactorPomMavenVersion,
197                                  final String parentPomMavenVersion,
198                                  final String packagePrefix) throws InvalidStructureException {
199 
200         // Sane state?
201         final String directoryName = getSkeletonPomDirectoryName(project.getName(), pomType, project.getPrefix());
202         final String relativePath = relativePomDirectory.isEmpty() ? "" : relativePomDirectory + "/";
203         final File pomDirectory = FileUtils.makeDirectory(rootDirectory, relativePath + directoryName);
204         final File pomFile = new File(pomDirectory, "pom.xml");
205 
206         if (FileUtils.exists(pomFile, false)) {
207             throw new InvalidStructureException("POM file [" + FileUtils.getCanonicalPath(pomFile)
208                     + "] could not be created as it already exists.");
209         }
210 
211         if (log.isDebugEnabled()) {
212             log.debug("Creating POM file [" + FileUtils.getCanonicalPath(pomFile) + "]");
213         }
214 
215         // Read the POM resource, and write the POM file.
216         final SingleBracketPomTokenParserFactory.Builder builder = SingleBracketPomTokenParserFactory
217                 .create(pomType, project)
218                 .withoutProjectNameAsDirectoryPrefix()
219                 .inSoftwareComponentWithRelativePath(relativePath)
220                 .withProjectGroupIdPrefix(packagePrefix)
221                 .withoutProjectSuffix()
222                 .withMavenVersions(reactorPomMavenVersion, parentPomMavenVersion);
223 
224         if (pomType == PomType.REACTOR) {
225 
226             // Find all module names for the active reactor directory.
227             final StringBuilder moduleBuilder = new StringBuilder();
228             for (String current : FileUtils.getModuleNames(pomDirectory)) {
229                 moduleBuilder.append("<module>").append(current).append("</module>\n");
230             }
231             builder.addToken(PomToken.MODULES.getToken(), moduleBuilder.toString());
232         }
233 
234         final TokenParser tokenParser = builder.build();
235 
236         final String pomData = tokenParser.substituteTokens(getTemplateResource(pomType + "/pom.xml"));
237         FileUtils.writeFile(pomFile, pomData);
238     }
