23  package se.jguru.nazgul.core.quickstart.api.analyzer;
24  
25  import org.apache.maven.model.Model;
26  import se.jguru.nazgul.core.quickstart.api.InvalidStructureException;
27  import se.jguru.nazgul.core.quickstart.api.PomType;
28  
29  /**
30   * <p>Specification for how to validate the content of POMs. In the Nazgul Framework structure, POMs have
31   * distinct roles and - therefore - consists of separate patterns. This PomAnalyzer provides a top-level
32   * specification for how a client can validate the content of a Maven POM. Such validation is required to
33   * quickly detect incompatibilities in a desired multi-module Maven reactor structure - which, in turn, is
34   * required to reduce entropy increase in big Maven projects.</p>
35   * <p>Refer to <a href="http://maven.apache.org/pom.html">the Maven POM specification</a> which hosts a reference to
36   * all parts of the POM.</p>
37   *
38   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
39   */
40  public interface PomAnalyzer {
41  
42      /**
43       * Validates the supplied Maven Root reactor POM Model or throws an exception indicating why the
44       * supplied POM was invalid. The "Root reactor POM" is the topmost POM within the local reactor,
45       * not counting a Parent POM which is not built within the reactor.
46       * This method is a convenience specification, and should yield the equivalent of calling
47       * {@code validate(aRootReactorPom, PomAnalyzer.PomType.ROOT_REACTOR, null)}
48       *
49       * @param aRootReactorPom A non-null maven Model from the topmost POM within the reactor (a.k.a. root reactor POM).
50       * @throws InvalidStructureException if the supplied aRootReactorPom had invalid dependencies according
51       *                                   to this PomAnalyzer.
52       */
53      void validateRootReactorPom(Model aRootReactorPom) throws InvalidStructureException;
54  
55      /**
56       * Validates the supplied Maven topmost parent POM Model or throws an exception indicating why the
57       * supplied POM was invalid. The "topmost parent POM" is the topmost POM within the local reactor,
58       * used as Parent for other POMs. This differs from the root reactor pom in that parent poms are used
59       * to define dependencies and plugins but not modules.
60       * This method is a convenience specification, and should yield the equivalent of calling
61       * {@code validate(topmostParentPom, PomAnalyzer.PomType.PARENT, null)}
62       *
63       * @param topmostParentPom A non-null maven Model from the topmost parent POM within the reactor.
64       * @throws InvalidStructureException if the supplied aRootReactorPom had invalid dependencies according
65       *                                   to this PomAnalyzer.
66       */
67      void validateTopmostParentPom(Model topmostParentPom) throws InvalidStructureException;
68  
69      /**
70       * Validates the supplied Maven POM Model or throws an exception indicating why the supplied POM was invalid.
71       *
72       * @param aPOM         The POM model to validate.
73       * @param parentOrNull The expected parent of aPOM, or {@code null} if it is irrelevant.
74       * @param expectedType The expected type of POM to validate the supplied Model against.
75       * @throws InvalidStructureException if the supplied aPOM was invalid according to this PomDependencyStrategy.
76       */
77      void validate(Model aPOM, PomType expectedType, Model parentOrNull) throws InvalidStructureException;
78  }
