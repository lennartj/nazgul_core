37  /**
38   * Abstract PomAnalyzer implementation, providing convenience methods for validating various
39   * properties of Maven POMs. Override this for a concrete PomAnalyzer class which can be applied
40   * along with a concrete NamingStrategy to analyze and validate the content of POMs for
41   * compliance with a given strategy.
42   *
43   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
44   */
45  public abstract class AbstractPomAnalyzer implements PomAnalyzer {
46  
47      // Our log
48      private static final Logger log = LoggerFactory.getLogger(AbstractPomAnalyzer.class.getName());
49  
50      // Internal state
51      private NamingStrategy namingStrategy;
52  
53      /**
54       * Creates an AbstractPomAnalyzer using the supplied NamingStrategy to create Names from Models and
55       * to validate the well-formedness of POMs.
56       *
57       * @param namingStrategy A non-null NamingStrategy used to validate Models.
58       */
59      protected AbstractPomAnalyzer(final NamingStrategy namingStrategy) {
60  
61          Validate.notNull(namingStrategy, "Cannot handle null namingStrategy argument.");
62          this.namingStrategy = namingStrategy;
63      }
64  
65      /**
66       * {@inheritDoc}
67       */
68      @Override
69      @SuppressWarnings("all")
70      public final void validateRootReactorPom(final Model aRootReactorPom) throws InvalidStructureException {
71  
72          // Validate that the supplied Model complies with the naming strategy.
73          final Name name = namingStrategy.createName(aRootReactorPom);
74          try {
75              namingStrategy.validate(name, PomType.ROOT_REACTOR);
76          } catch (IllegalArgumentException e) {
77              final StringBuilder builder = getErrorMessageBuilder("Incompatible POM name for strategy ["
78                      + namingStrategy.getClass().getSimpleName() + "]: " + e.getMessage(), aRootReactorPom);
79              throw new InvalidStructureException(builder.toString());
80          }
81  
82          // Delegate
83          performCustomRootReactorPomValidation(aRootReactorPom);
84      }
85  
86      /**
87       * {@inheritDoc}
88       */
89      @Override
90      @SuppressWarnings("all")
91      public final void validateTopmostParentPom(final Model topmostParentPom) throws InvalidStructureException {
92  
93          // Validate that the supplied Model complies with the naming strategy.
94          final Name name = namingStrategy.createName(topmostParentPom);
95          try {
96              namingStrategy.validate(name, PomType.PARENT);
97          } catch (IllegalArgumentException e) {
98              final StringBuilder builder = getErrorMessageBuilder("Incompatible POM name for strategy ["
99                      + namingStrategy.getClass().getSimpleName() + "]: " + e.getMessage(), topmostParentPom);
100             throw new InvalidStructureException(builder.toString());
101         }
102 
103         // Parent POMs should not define Modules
104         verifyNoModules(topmostParentPom);
105 
106         // Delegate
107         performCustomTopmostParentPomValidation(topmostParentPom);
108     }
109 
110     /**
111      * {@inheritDoc}
112      */
113     @Override
114     @SuppressWarnings("all")
115     public final void validate(final Model aPOM, final PomType expectedType, final Model parentOrNull) throws
116             InvalidStructureException {
117 
118         // Check sanity
119         Validate.notNull(aPOM, "Cannot handle null aPOM argument.");
120         Validate.notNull(expectedType, "Cannot handle null expectedType argument.");
121 
122         // Validate that the supplied Model complies with the naming strategy.
123         final Name name = namingStrategy.createName(aPOM);
124         try {
125             namingStrategy.validate(name, expectedType);
126         } catch (IllegalArgumentException e) {
127             final StringBuilder builder = getErrorMessageBuilder("Incompatible POM name for strategy ["
128                     + namingStrategy.getClass().getSimpleName() + "]: " + e.getMessage(), aPOM);
129             throw new InvalidStructureException(builder.toString());
130         }
131 
132         switch (expectedType) {
133             case ROOT_REACTOR:
134                 validateRootReactorPom(aPOM);
135                 break;
136 
137             case PARENT:
138                 // Delegate
139                 validateTopmostParentPom(aPOM);
140                 break;
141 
142             case API_PARENT:
143             case MODEL_PARENT:
144             case WAR_PARENT:
145             case OTHER_PARENT:
146 
147                 // Check sanity
148                 Validate.notNull(parentOrNull, "Cannot handle null parentOrNull argument for expected type ["
149                         + expectedType + "].");
150 
151                 // Delegate
152                 validateParentPom(aPOM, expectedType, parentOrNull);
153                 break;
154 
155             case REACTOR:
156 
157                 // Check sanity
158                 Validate.notNull(parentOrNull, "Cannot handle null parentOrNull argument for expected type ["
159                         + expectedType + "].");
160 
161                 validateReactorPom(aPOM, parentOrNull);
162                 break;
163 
164             default:
165                 throw new UnsupportedOperationException("Cannot handle expectedType [" + expectedType + "]");
166         }
167     }
168 
169     /**
170      * Verifies that the supplied toValidate Model does not contain any defined Modules, and
171      * that the supplied nonNullParent has the same GAV coordinates as the supplied toValidate POM.
172      *
173      * @param toValidate   The POM model to validate.
174      * @param expectedType The expected type of POM to validate the supplied Model against.
175      * @param parentOrNull A Model which holds the expected Maven GAV coordinates for the aPOM's Parent POM.
176      * @throws InvalidStructureException if the Model does not conform to structure specifications.
177      */
178     protected void validateParentPom(final Model toValidate, final PomType expectedType, final Model parentOrNull)
179             throws InvalidStructureException {
180 
181         // Parent POMs should not define Modules
182         verifyNoModules(toValidate);
183 
184         // Is the supplied parentOrNull correct?
185         if (parentOrNull != null) {
186             verifyParentPom(toValidate, parentOrNull);
187         }
188     }
189 
190     /**
191      * Verifies that the supplied toValidate Model does not contain any defined Modules, and
192      * that the supplied nonNullParent has the same GAV coordinates as the supplied toValidate POM.
193      *
194      * @param toValidate    The POM model to validate.
195      * @param nonNullParent A Model which holds the expected Maven GAV coordinates for the aPOM's Parent POM.
196      * @throws InvalidStructureException if the Model does not conform to structure specifications, i.e. if the
197      *                                   reactor POM was invalid.
198      */
199     protected void validateReactorPom(final Model toValidate, final Model nonNullParent)
200             throws InvalidStructureException {
201 
202         // Is the supplied nonNullParent correct?
203         verifyParentPom(toValidate, nonNullParent);
204     }
205 
206     /**
207      * Override this method to perform custom validation of the Maven Model from the root reactor POM.
208      * The NamingStrategy has already been used to validate maven GAV coordinates for the supplied root reactor
209      * when this method is invoked.
210      *
211      * @param rootReactorModel the root rootReactorModel which should be validated in a custom way.
212      * @throws InvalidStructureException if the supplied aRootReactorModel was invalid.
213      */
214     protected void performCustomRootReactorPomValidation(final Model rootReactorModel)
215             throws InvalidStructureException {
216         // Do nothing.
217     }
218 
219     /**
220      * Override this method to perform custom validation of the Maven Model from the topmost parent POM.
221      * The NamingStrategy has already been used to validate maven GAV coordinates for the supplied topmost
222      * parent POM when this method is invoked.
223      *
224      * @param topmostParentModel the root rootReactorModel which should be validated in a custom way.
225      * @throws InvalidStructureException if the supplied topmostParentModel was invalid.
226      */
227     protected void performCustomTopmostParentPomValidation(final Model topmostParentModel)
228             throws InvalidStructureException {
229         // Do nothing.
230     }

236     /**
237      * Verifies that the supplied toValidate Model has identical groupId, artifactId and version
238      * of the supplied requiredParent Model.
239      *
240      * @param toVerify       The Model to verify.
241      * @param requiredParent The Maven Model containing data of the required Parent for the supplied toVerify Model.
242      * @throws InvalidStructureException if the GAV coordinates of the supplied toVerify Model did not match those of
243      *                                   the supplied requiredParent Module.
244      */
245     protected final void verifyParentPom(final Model toVerify, final Model requiredParent)
246             throws InvalidStructureException {
247 
248         final List<String> errors = new ArrayList<>();
249 
250         //
251         // Note that the version of the toValidate project might be null
252         // (in case the version is inherited from the parent, and therefore matching).
253         //
254         final Parent parentToValidate = toVerify.getParent();
255         final boolean versionMatches = toVerify.getVersion() == null
256                 || requiredParent.getVersion().equals(parentToValidate.getVersion());
257         if (!versionMatches) {
258             errors.add("Required version mismatch (Required: " + requiredParent.getVersion() + ", Actual: "
259                     + parentToValidate.getVersion() + ")");
260         }
261         if (!requiredParent.getGroupId().equals(parentToValidate.getGroupId())) {
262             errors.add("Required groupId mismatch (Required: " + requiredParent.getGroupId() + ", Actual: "
263                     + parentToValidate.getGroupId() + ")");
264         }
265         if (!requiredParent.getArtifactId().equals(parentToValidate.getArtifactId())) {
266             errors.add("Required artifactId mismatch (Required: " + requiredParent.getArtifactId() + ", Actual: "
267                     + parentToValidate.getArtifactId() + ")");
268         }
269 
270         if (errors.size() > 0) {
271 
272             final StringBuilder builder = getErrorMessageBuilder("Incorrect parent relationship", toVerify);
273             for (int i = 0; i < errors.size(); i++) {
274                 builder.append("\n " + (i + 1) + ") ").append(errors.get(i));
275             }
276 
277             // All done.
278             throw new InvalidStructureException(builder.toString());
279         }
280     }
281 
282     /**
283      * Verifies that the supplied Model does not have any defined Modules.
284      * This is a requirement for all Parent POMs, since they should be the direct parents of
285      * artifact/leaf projects within a multi-module reactor.
286      *
287      * @param toVerify The Model to verify.
288      * @throws InvalidStructureException if the supplied Model contained Module definitions.
289      */
290     protected void verifyNoModules(final Model toVerify) throws InvalidStructureException {
291         final List<String> modules = toVerify.getModules();
292         if (modules != null && modules.size() > 0) {
293 
294             final StringBuilder builder = getErrorMessageBuilder("POM should not contain Modules", toVerify);
295             throw new InvalidStructureException(builder.toString());
296         }
297     }

299     /**
300      * Retrieves the expected suffix for an artifactID of a project Model with the supplied type.
301      *
302      * @param type The PomType of a project.
303      * @return the expected suffix for an artifactID of a project Model with the supplied type, or {@code null}
304      * if none could be provided. (This is the case for a OTHER_PARENT only).
305      */
306     protected String getExpectedArtifactIdSuffix(final PomType type) {
307 
308         // Check sanity
309         Validate.notNull(type, "Cannot handle null type argument.");
310 
311         String toReturn = null;
312 
313         switch (type) {
314             case ROOT_REACTOR:
315             case REACTOR:
316                 toReturn = AbstractNamingStrategy.REACTOR_SUFFIX;
317                 break;
318 
319             case PARENT:
320             case API_PARENT:
321             case MODEL_PARENT:
322             case WAR_PARENT:
323                 toReturn = type.name().toLowerCase().replace('_', '-');
324                 break;
325 
326             case OTHER_PARENT:
327                 toReturn = AbstractNamingStrategy.PARENT_SUFFIX;
328                 break;
329 
330             default:
331                 log.warn("PomType [" + type + "] not registered.");
332                 break;
333         }
334 
335         // All done.
336         return toReturn;
337     }
338 
339     private StringBuilder getErrorMessageBuilder(final String mainError, final Model model) {
340 
341         // Check sanity
342         Validate.notEmpty(mainError, "Cannot handle null or empty mainError argument.");
343         Validate.notNull(model, "Cannot handle null model argument.");
344 
345         final String effectiveVersion = model.getVersion() == null
346                 ? model.getParent().getVersion()
347                 : model.getVersion();
348 
349         return new StringBuilder(mainError + " for POM (" + model.getGroupId() + "/" + model.getArtifactId()
350                 + "/" + effectiveVersion + "): ");
351     }
