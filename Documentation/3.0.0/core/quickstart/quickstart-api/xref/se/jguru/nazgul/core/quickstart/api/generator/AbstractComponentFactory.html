91      /**
92       * {@inheritDoc}
93       */
94      @Override
95      public final void createSoftwareComponent(final File componentDirectory,
96                                                final SortedMap<SoftwareComponentPart, String> parts2SuffixMap)
97              throws InvalidStructureException {
98  
99          // Check sanity
100         Validate.notNull(componentDirectory, "Cannot handle null componentDirectory argument.");
101         Validate.notEmpty(parts2SuffixMap, "Cannot handle null or empty parts2SuffixMap argument.");
102         Validate.isTrue(!FileUtils.FILE_FILTER.accept(componentDirectory),
103                 "ComponentDirectory [" + FileUtils.getCanonicalPath(componentDirectory)
104                         + "] must not refer to an existing file.");
105         Validate.isTrue(FileUtils.NONEXISTENT_OR_EMPTY_DIRECTORY_FILTER.accept(componentDirectory),
106                 "ComponentDirectory [" + FileUtils.getCanonicalPath(componentDirectory)
107                         + "] must refer to a nonexistent or empty directory.");
108 
109         // Check structure of the supplied arguments
110         final Set<SoftwareComponentPart> parts = parts2SuffixMap.keySet();
111         if (parts.contains(SoftwareComponentPart.IMPLEMENTATION)
112                 && !(parts.contains(SoftwareComponentPart.API) || parts.contains(SoftwareComponentPart.SPI))) {
113             throw new InvalidStructureException("Software components containing an Implementation project should"
114                     + "contain either an API or SPI project as well.");
115         }
116         for (Map.Entry<SoftwareComponentPart, String> current : parts2SuffixMap.entrySet()) {
117             final String suffix = current.getValue();
118             if (current.getKey().isSuffixRequired()) {
119                 Validate.notEmpty(suffix, "A suffix is required for SoftwareComponentPart [" + current.getKey().name()
120                         + "], but none was given.");
121             }
122         }
123 
124         // Validate that we should be able to create a component here.
125         final File componentPomFile = new File(componentDirectory, "pom.xml");
126         if (FileUtils.FILE_FILTER.accept(componentPomFile)) {
127 
128             // The componentPomFile must be either of type REACTOR or ROOT_REACTOR.
129             final Model pomModel = FileUtils.getPomModel(componentPomFile);
130             final PomType pomType = getNamingStrategy().getPomType(pomModel);
131 
132             Validate.isTrue(pomType == PomType.REACTOR || pomType == PomType.ROOT_REACTOR,
133                     "ComponentPart projects can only be added to components. This implies that ["
134                             + FileUtils.getCanonicalPath(componentPomFile)
135                             + "] must either be of type REACTOR or ROOT_REACTOR. (Detected type [" + pomType + "])");
136         }
137 
138         // Ensure that existing parent directories contain REACTOR or ROOT_REACTOR poms.
139         final File originatingDirectory = componentPomFile.exists()
140                 ? componentDirectory
141                 : componentDirectory.getParentFile();
142         final File rootDir = navigator.getProjectRootDirectory(originatingDirectory);
143         final String relativePath = navigator.getRelativePath(componentDirectory, false);
144 
145         final List<String> pathSegments = new ArrayList<>();
146         if (relativePath != null) {
147             final StringTokenizer tok = new StringTokenizer(relativePath, File.separator, false);
148             while (tok.hasMoreTokens()) {
149                 pathSegments.add(tok.nextToken());
150             }
151         }
152 
153         File tmp = new File(rootDir, "pom.xml");
154         for (String current : pathSegments) {
155             if (FileUtils.FILE_FILTER.accept(tmp)) {
156 
157                 if (log.isDebugEnabled()) {
158                     log.debug("Validating that [" + FileUtils.getCanonicalPath(tmp)
159                             + "] is a REACTOR or ROOT_REACTOR pom to enable creating a SoftwareComponent.");
160                 }
161 
162                 // The current POM should be of the correct type.
163                 final Model pomModel = FileUtils.getPomModel(tmp);
164                 final PomType pomType = getNamingStrategy().getPomType(pomModel);
165 
166                 Validate.isTrue(pomType == PomType.REACTOR || pomType == PomType.ROOT_REACTOR,
167                         "ComponentPart projects can only be added to components. This implies that ["
168                                 + FileUtils.getCanonicalPath(tmp) + "] must either be of type REACTOR "
169                                 + "or ROOT_REACTOR. (Detected type [" + pomType + "])");
170 
171                 // Descend
172                 tmp = new File(tmp, current + "/pom.xml");
173             } else {
174 
175                 // No point in going deeper.
176                 break;
177             }
178         }
179 
180         // OK to create the SoftwareComponentPart.
181         File componentDir = componentDirectory;
182         if (!FileUtils.exists(componentDir, true)) {
183             componentDir = FileUtils.makeDirectory(rootDir, relativePath);
184         }
185 
186         // Create each software component part project.
187         for (Map.Entry<SoftwareComponentPart, String> current : parts2SuffixMap.entrySet()) {
188             addSoftwareComponentPart(componentDir, current.getKey(), current.getValue());
189         }
190 
191         // Create or update the reactor POM for the software component.
192         generateReactorPom(componentDir);
193 
194         // TODO: Update the parent reactor POM to include the currently added component?
195     }

197     /**

200     @Override
201     public void addSoftwareComponentPart(final File componentReactorDirectory,
202                                          final SoftwareComponentPart toAdd,
203                                          final String suffix) throws InvalidStructureException {
204 
205         // Check sanity
206         Validate.notNull(toAdd, "Cannot handle null toAdd argument.");
207         Validate.notNull(componentReactorDirectory, "Cannot handle null componentDirectory argument.");
208         Validate.isTrue(FileUtils.DIRECTORY_FILTER.accept(componentReactorDirectory),
209                 "Software Component Directory [" + FileUtils.getCanonicalPath(componentReactorDirectory)
210                         + "] was not an existing directory.");
211         if (toAdd.isSuffixRequired()) {
212             Validate.notEmpty(suffix,
213                     "Cannot handle null or empty suffix argument for SoftwareComponentPart [" + toAdd
214                             + "], as it requires a project suffix.");
215         }
216 
217         // Get Project data.
218         final Project projectData = getProjectFor(componentReactorDirectory);
219 
220         // Proceed to create the structures as instructed.
221         final File projectRootDirectory = navigator.getProjectRootDirectory(componentReactorDirectory);
222         final String relativePath = navigator.getRelativePath(componentReactorDirectory, false);
223         final Model rootReactorPomModel = FileUtils.getPomModel(new File(projectRootDirectory, "pom.xml"));
224         final Name rootReactorName = getNamingStrategy().createName(rootReactorPomModel);
225 
226         final File parentPomDir = navigator.getParentPomDirectory(componentReactorDirectory);
227         final Model parentPomModel = FileUtils.getPomModel(new File(parentPomDir, "pom.xml"));
228 
229         File componentDir = componentReactorDirectory;
230         if (!FileUtils.exists(componentDir, true)) {
231             componentDir = FileUtils.makeDirectory(projectRootDirectory, relativePath);
232         }
233 
234         //
235         // Find the name of the part directory, and then create it.
236         // The directory name here is something like "messaging-model", or
237         // "foobar-messaging-model" depending on the desired prefix structure.
238         //
239         final String dirPrefix = rootReactorName.getPrefix() != null && !"".equals(rootReactorName.getPrefix())
240                 ? rootReactorName.getPrefix() + Name.DEFAULT_SEPARATOR
241                 : "";
242         final String dirSuffix = toAdd.isSuffixRequired() ? Name.DEFAULT_SEPARATOR + suffix : "";
243         final String dirName = dirPrefix + rootReactorName.getName()
244                 + Name.DEFAULT_SEPARATOR + componentDir.getName()
245                 + Name.DEFAULT_SEPARATOR + toAdd.getType()
246                 + dirSuffix;
247 
248         final File partDir = FileUtils.makeDirectory(componentDir, dirName);
249         final String topReactorPomVersion = rootReactorPomModel.getVersion() == null
250                 ? projectData.getReactorParent().getMavenVersion()
251                 : rootReactorPomModel.getVersion();
252         final String parentPomVersion = parentPomModel.getVersion() == null
253                 ? projectData.getParentParent().getMavenVersion()
254                 : parentPomModel.getVersion();
255 
256         // Create a Maven project for the given PomType within PartDir
257         createMavenProjectForSoftwareComponentPart(toAdd,
258                 partDir,
259                 topReactorPomVersion,
260                 parentPomVersion,
261                 projectData,
262                 getProjectGroupIdPrefix(rootReactorPomModel.getGroupId(), rootReactorPomModel.getArtifactId()),
263                 suffix);
264     }

266     /**
267      * Creates a maven project within {@code mavenProjectDir} for a SoftwareComponentPart of the supplied type.
268      * This is a convenience method to create Maven project structures for SoftwareComponentParts. (I.e. not a
269      * generic Maven project creation method).
270      *
271      * @param part                   The non-null SoftwareComponentPart indicating which type of project should be created.
272      * @param mavenProjectDir        The directory where the maven project should be generated.
273      *                               Must exist and be a directory.
274      * @param reactorPomMavenVersion The non-empty Maven project version of reactor POMs within the project.
275      *                               Should be identical to the version of the reactor parent POM.
276      * @param parentPomMavenVersion  The non-empty Maven project version of parent POMs within the project (and,
277      *                               normally, the version of all leaf projects within the Maven project reactor).
278      *                               Should be identical to the version of the parent parent POM.
279      * @param project                The non-null Project data, primarily used to initialize the active TokenParser.
280      * @param optionalGroupIdPrefix  The groupId prefix (commonly the reverse DNS, similar to {@code se.jguru} or
281      *                               {@code org.apache}) of the organisation that owns the project created.
282      *                               The groupId prefix is prepended to the groupId of the created Maven project.
283      * @param partSuffix             The suffix of the project SoftwareComponentPart project,
284      *                               used only if {@code SoftwareComponentPart.isSuffixRequired()} yields {@code true}.
285      */
286     @SuppressWarnings("All")
287     protected void createMavenProjectForSoftwareComponentPart(final SoftwareComponentPart part,
288                                                               final File mavenProjectDir,
289                                                               final String reactorPomMavenVersion,
290                                                               final String parentPomMavenVersion,
291                                                               final Project project,
292                                                               final String optionalGroupIdPrefix,
293                                                               final String partSuffix) {
294 
295         // Check sanity
296         Validate.notNull(part, "Cannot handle null part argument.");
297         Validate.notNull(mavenProjectDir, "Cannot handle null mavenProjectDir argument.");
298         Validate.isTrue(mavenProjectDir.exists() && mavenProjectDir.isDirectory(), "Maven project directory ["
299                 + FileUtils.getCanonicalPath(mavenProjectDir) + "] must exist and be a directory.");
300         Validate.notEmpty(reactorPomMavenVersion, "Cannot handle null or empty reactorPomMavenVersion argument.");
301         Validate.notEmpty(parentPomMavenVersion, "Cannot handle null or empty parentPomMavenVersion argument.");
302 
303         // 1) Extract all template files corresponding to the SoftwareComponentPart to a temporary directory
304         final String dirName = mavenProjectDir.getName();
305         final File tmpExtractedFilesRoot = extractTemplateFiles(dirName, part.getComponentPomType().name());
306         if (log.isDebugEnabled()) {
307             log.debug("Using temporary directory [" + FileUtils.getCanonicalPath(tmpExtractedFilesRoot)
308                     + "] to extract template files.");
309         }
310 
311         // 2) If mavenProjectDir starts with the project prefix, we should use that
312         //    setting when creating the tokens as well, and vice versa.
313         final String projectNameSep = project.getName() + Name.DEFAULT_SEPARATOR;
314         final boolean useProjectNameAsDirectoryPrefix = dirName.startsWith(projectNameSep)
315                 || dirName.startsWith(project.getPrefix() + Name.DEFAULT_SEPARATOR + projectNameSep);
316 
317         // 3) Create a TokenParser to handle template tokens
318         final File softwareComponentDir = part.getComponentPomType().name().startsWith("COMPONENT_")
319                 ? mavenProjectDir.getParentFile()
320                 : mavenProjectDir;
321         final TokenParser tokenParser = SingleBracketPomTokenParserFactory
322                 .create(part.getComponentPomType(), project)
323                 .prependProjectNameAsDirectoryPrefix(useProjectNameAsDirectoryPrefix)
324                 .inSoftwareComponentWithRelativePath(navigator.getRelativePath(softwareComponentDir, false))
325                 .withProjectGroupIdPrefix(optionalGroupIdPrefix)
326                 .withProjectSuffix(partSuffix)
327                 .withMavenVersions(reactorPomMavenVersion, parentPomMavenVersion)
328                 .build();
329 
330         // 3) Tokenize (as appropriate) and move each file to its destination.
331         tokenizeAndDeploy(tmpExtractedFilesRoot, tokenParser, mavenProjectDir);
332     }

334     /**
335      * Generates a reactor POM in the supplied softwareComponentDir directory.
336      * Also - before calling this method, the REACTOR_ROOT pom must exist.
337      *
338      * @param softwareComponentDir A directory which must exist and be located within a Maven reactor.
339      */
340     protected void generateReactorPom(final File softwareComponentDir) {
341 
342         // Check sanity
343         Validate.notNull(softwareComponentDir, "Cannot handle null softwareComponentDir argument.");
344         Validate.isTrue(FileUtils.DIRECTORY_FILTER.accept(softwareComponentDir),
345                 "SoftwareComponent directory [" + FileUtils.getCanonicalPath(softwareComponentDir)
346                         + "] must exist and be a directory.");
347 
348         // Read the template data
349         final Project projectData = getProjectFor(softwareComponentDir);
350         final String dirName = softwareComponentDir.getName();
351         final String reactorParentVersion = projectData.getReactorParent().getMavenVersion();
352         final String parentParentVersion = projectData.getParentParent().getMavenVersion();
353         final File projectRootDirectory = navigator.getProjectRootDirectory(softwareComponentDir);
354         final Model rootReactorPomModel = FileUtils.getPomModel(new File(projectRootDirectory, "pom.xml"));
355         final PomType pomType = FileUtils.getCanonicalPath(softwareComponentDir)
356                 .equals(FileUtils.getCanonicalPath(projectRootDirectory))
357                 ? PomType.ROOT_REACTOR
358                 : PomType.REACTOR;
359         final String relativePath = navigator.getRelativePath(softwareComponentDir, false);
360 
361         if (log.isDebugEnabled()) {
362             log.debug("Generating reactor pom within directory [" + FileUtils.getCanonicalPath(softwareComponentDir)
363                     + "], which implies relative path [" + relativePath + "]");
364         }
365 
366         // 1) Extract all template files corresponding to the SoftwareComponentPart to a temporary directory
367         final File tmpExtractedFilesRoot = extractTemplateFiles(softwareComponentDir.getName(), pomType.name());
368 
369         // 2) Calculate the groupId prefix
370         final String projectGroupIdPrefix = getProjectGroupIdPrefix(
371                 rootReactorPomModel.getGroupId(),
372                 rootReactorPomModel.getArtifactId());
373 
374         // 3) Generate the modules XML text.
375         final StringBuilder modulesElementBuilder = new StringBuilder("");
376         for (File current : softwareComponentDir.listFiles(FileUtils.MODULE_NAME_FILTER)) {
377             modulesElementBuilder.append("<module>").append(current.getName()).append("</module>\n");
378         }
379         if (modulesElementBuilder.toString().isEmpty() && log.isWarnEnabled()) {
380             log.warn("Directory [" + FileUtils.getCanonicalPath(softwareComponentDir) + "] held no module "
381                     + "subdirectories. This implies a non-functional SoftwareComponent of type ["
382                     + pomType.name() + "] without any parts.");
383             modulesElementBuilder.append("<module/>");
384         }
385 
386         // 4) If mavenProjectDir starts with the project prefix, we should use that
387         //    setting when creating the tokens as well, and vice versa.
388         final String projectNameSep = projectData.getName() + Name.DEFAULT_SEPARATOR;
389         final boolean useProjectNameAsDirectoryPrefix = dirName.startsWith(projectNameSep)
390                 || dirName.startsWith(projectData.getPrefix() + Name.DEFAULT_SEPARATOR + projectNameSep);
391 
392         // 5) Create a TokenParser to handle template tokens
393         final TokenParser tokenParser = SingleBracketPomTokenParserFactory
394                 .create(pomType, projectData)
395                 .prependProjectNameAsDirectoryPrefix(useProjectNameAsDirectoryPrefix)
396                 .inSoftwareComponentWithRelativePath(relativePath)
397                 .withProjectGroupIdPrefix(projectGroupIdPrefix)
398                 .withoutProjectSuffix()
399                 .withMavenVersions(reactorParentVersion, parentParentVersion)
400                 .addToken(PomToken.MODULES.getToken(), modulesElementBuilder.toString())
401                 .build();
402 
403         // 5) Tokenize and write the resulting file(s).
404         tokenizeAndDeploy(tmpExtractedFilesRoot, tokenParser, softwareComponentDir);
405     }

407     /**
408      * Performs three tasks:
409      * <ol>
410      * <li>Converts the supplied templateResourcePath to a URL</li>
411      * <li>Creates a temporary directory with the local path {@code "template/" + tmpDirName + "_" + i} under the
412      * standard temporary directory path</li>
413      * <li>Extracts all resources found in the Classpath under the templateResourcePath to the created temporary
414      * directory</li>
415      * </ol>
416      *
417      * @param tmpDirName           The name of the temporary directory to which the templates should be extracted.
418      * @param templateResourcePath The resource path (not starting with '/') under which all templates are found.
419      * @return The root directory under which all templates were extracted.
420      */
421     protected File extractTemplateFiles(final String tmpDirName,
422                                         final String templateResourcePath) {
423 
424         // Check sanity
425         Validate.notNull(tmpDirName, "Cannot handle null tmpDirName argument.");
426 
427         File toReturn;
428         for (int i = 0; true; i++) {
429             toReturn = new File(TMP_DIR, STANDARD_TEMPLATE_DIR + tmpDirName + "_" + i);
430             if (!toReturn.exists()) {
431                 toReturn.mkdirs();
432                 break;
433             }
434         }
435 
436         // Find the template retrieval specification.
437         final URL templateRootURL = getTemplateResourceURL(templateResourcePath);
438         final String protocol = templateRootURL.getProtocol();
439 
440         // Update the error message if other resource URL protocols are supported.
441         final boolean usesJarPackaging = "jar".equalsIgnoreCase(protocol);
442         final boolean usesFilePackaging = "file".equalsIgnoreCase(protocol);
443         if (usesJarPackaging) {
444 
445             // Find the template JarFile, and extract all relevant templates to a temporary directory.
446             final JarFile templateJarFile = JarExtractor.getJarFileFor(templateRootURL);
447             JarExtractor.extractResourcesFrom(
448                     templateJarFile,
449                     Pattern.compile(templateResourcePath + "/.*"),
450                     toReturn,
451                     false);
452 
453         } else if (usesFilePackaging) {
454 
455             // Check sanity
456             final File templateDirectory = new File(templateRootURL.getPath());
457             Validate.isTrue(FileUtils.DIRECTORY_FILTER.accept(templateDirectory),
458                     "File template root directory must exist and be a directory. (Got: ["
459                             + templateDirectory.getAbsolutePath() + "]");
460 
461             // Copy all resources to the tmpExtractedFilesRoot directory.
462             final SortedMap<String, File> path2File = FileUtils.listFilesRecursively(templateDirectory);
463 
464             for (Map.Entry<String, File> current : path2File.entrySet()) {
465 
466                 // All these files are considered temporary; delete on exit
467                 final File toWrite = new File(toReturn, current.getKey());
468                 toWrite.deleteOnExit();
469 
470                 final File parentDir = toWrite.getParentFile();
471                 if (!parentDir.exists()) {
472                     if(parentDir.mkdirs()) {
473                         log.debug("Successfully created parent directory [" + FileUtils.getCanonicalPath(parentDir)
474                                 + "]");
475                     } else {
476                         log.warn("Could not create parent directory ["
477                                 + FileUtils.getCanonicalPath(parentDir) + "]");
478                     }
479                 }
480 
481                 // Copy the template file.
482                 FileUtils.writeFile(toWrite, FileUtils.readFile(current.getValue()));
483             }
484 
485         } else {
486             throw new IllegalStateException("Cannot handle resource URL [" + templateRootURL.toString()
487                     + "]. Supported protocols are ('jar', 'file').");
488         }
489 
490         // Since these templates are temporary in nature, ensure that they are deleted on program exit
491         final SortedMap<String, File> path2ExtractedTemplateFile = FileUtils.listFilesRecursively(toReturn);
492         for (Map.Entry<String, File> current : path2ExtractedTemplateFile.entrySet()) {
493             current.getValue().deleteOnExit();
494         }
495 
496         // All done.
497         return toReturn;
498     }

500     /**
501      * Retrieves the prefix to be prepended to groupIds within the project. Normally, the project's groupIdPrefix is
502      * identical to the reverse DNS of the organisation hosting the project, plus the name of the project itself.
503      * Override this method if you choose to derive the project's groupId prefix in another way.
504      * The default implementation strips the type suffix from rootReactorArtifactId and converts the remainder to a
505      * path snippet, used as a delimiter to strip of the previous part from the rootReactorGroupId. This is
506      * illustrated by an example.
507      * <ol>
508      * <li>Given <strong>rootReactorGroupId</strong>: {@code se.jguru.nazgul.core}, and</li>
509      * <li>Given <strong>rootReactorArtifactId</strong>: {@code nazgul-core-reactor}</li>
510      * <li>First, the prefix and name of the rootReactorArtifactId is joined with the
511      * {@code Name.DEFAULT_SEPARATOR} yielding {@code nazgul.core}</li>
512      * <li>The return value is found as a substring of the rootReactorGroupId up until the prefix snippet is
513      * found, stripping off the trailing ".": {@code se.jguru}</li>
514      * </ol>
515      * Override this method if your concrete ComponentFactory uses another algorithm to acquire the
516      * projectGroupIdPrefix.
517      *
518      * @param rootReactorGroupId    The groupId of the root reactor POM within the project.
519      * @param rootReactorArtifactId The artifactId of the root reactor POM within the project.
520      * @return the projectGroupIdPrefix, prepended to the relative path of a software component to retrieve the full
521      * groupID within its POM.
522      */
523     protected String getProjectGroupIdPrefix(final String rootReactorGroupId,
524                                              final String rootReactorArtifactId) {
525 
526         // Check sanity
527         Validate.notEmpty(rootReactorGroupId, "Cannot handle null or empty rootReactorGroupId argument.");
528         Validate.notEmpty(rootReactorArtifactId, "Cannot handle null or empty rootReactorArtifactId argument.");
529 
530         final Name artifactIdName = Name.parse(rootReactorArtifactId);
531         final String packageSnippet = (artifactIdName.getPrefix() != null ? artifactIdName.getPrefix() + "." : "")
532                 + artifactIdName.getName();
533 
534         if (log.isDebugEnabled()) {
535             log.debug("rootReactorArtifactId [" + rootReactorArtifactId + "] yielded packageSnippet ["
536                     + packageSnippet + "]");
537         }
538 
539         // Normal calculation.
540         int endIndex = rootReactorGroupId.indexOf(packageSnippet);
541         Validate.isTrue(endIndex != -1, "Could not calculate projectGroupIdPrefix from groupID ["
542                 + rootReactorGroupId + "] and artifactID [" + rootReactorArtifactId
543                 + "]. (Calculated package snippet [" + packageSnippet + "] not found within groupID).");
544 
545         // Polish the rootReactorGroupId and return.
546         final String candidate = rootReactorGroupId.substring(0, endIndex);
547         int cutoffIndex = candidate.endsWith(".") ? candidate.length() - 1 : candidate.length();
548         return candidate.substring(0, cutoffIndex);
549     }

555     /**
556      * Creates the Project originating from the supplied dirInReactor directory (which must exist).
557      *
558      * @param dirInReactor A non-null directory within the reactor structure.
559      * @return The Project data for the supplied dirInReactor.
560      */
561     private Project getProjectFor(final File dirInReactor) {
562 
563         // Check sanity
564         Validate.notNull(dirInReactor, "Cannot handle null dirInReactor argument.");
565         Validate.isTrue(FileUtils.DIRECTORY_FILTER.accept(dirInReactor),
566                 "Software Component Directory [" + FileUtils.getCanonicalPath(dirInReactor)
567                         + "] was not an existing directory.");
568 
569         // Proceed to create the structures as instructed.
570         final File projectRootDirectory = navigator.getProjectRootDirectory(dirInReactor);
571         final Model rootReactorPomModel = FileUtils.getPomModel(new File(projectRootDirectory, "pom.xml"));
572         final Name rootReactorName = getNamingStrategy().createName(rootReactorPomModel);
573 
574         final File parentPomDir = navigator.getParentPomDirectory(dirInReactor);
575         final Model parentPomModel = FileUtils.getPomModel(new File(parentPomDir, "pom.xml"));
576 
577         // All done.
578         return new Project(
579                 rootReactorName.getPrefix(),
580                 rootReactorName.getName(),
581                 FileUtils.getSimpleArtifact(rootReactorPomModel),
582                 FileUtils.getSimpleArtifact(parentPomModel));
583     }

585     /**
586      * Tokenizes all files found under extractedTemplatesRoot using the supplied tokenParser.
587      * Following the tokenization, the files are copied to their respective relative paths under
588      * the mavenProjectDir.
589      *
590      * @param extractedTemplatesRoot A directory holding the extracted template files.
591      * @param tokenParser            The TokenParser used to tokenize the templates.
592      * @param targetDirectory        The directory where the project should be written.
593      */
594     private void tokenizeAndDeploy(final File extractedTemplatesRoot,
595                                    final TokenParser tokenParser,
596                                    final File targetDirectory) {
597 
598         final FileFilter shouldTokenizeFilter = getShouldTokenizeFilter();
599         final SortedMap<String, File> path2FileMap = FileUtils.listFilesRecursively(extractedTemplatesRoot);
600         for (Map.Entry<String, File> current : path2FileMap.entrySet()) {
601 
602             // Tokenize the relative path, and the file data if applicable.
603             final String relativePath = tokenParser.substituteTokens(current.getKey());
604             final String data = shouldTokenizeFilter.accept(current.getValue())
605                     ? tokenParser.substituteTokens(FileUtils.readFile(current.getValue()))
606                     : FileUtils.readFile(current.getValue());
607 
608             final File toWrite = new File(targetDirectory, relativePath);
609             final File parentFile = toWrite.getParentFile();
610             if (!parentFile.exists()) {
611                 parentFile.mkdirs();
612             }
613 
614             // Finally, write the file.
615             FileUtils.writeFile(toWrite, data);
616         }
617     }
