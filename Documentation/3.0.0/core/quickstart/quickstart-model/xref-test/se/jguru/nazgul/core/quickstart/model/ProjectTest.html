23  package se.jguru.nazgul.core.quickstart.model;
24  
25  import org.junit.Assert;
26  import org.junit.Test;
27  import se.jguru.nazgul.test.xmlbinding.XmlTestUtils;
28  
29  import java.io.StringReader;
30  
31  /**
32   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
33   */
34  public class ProjectTest extends AbstractJaxbBinderTest {
35  
36      @Test
37      public void validateMarshalling() throws Exception {
38  
39          // Assemble
40          final SimpleArtifact reactorParent = new SimpleArtifact(
41                  "reactorGroupId", "reactorArtifactId", "reactorMavenVersion");
42          final SimpleArtifact parentParent = new SimpleArtifact("groupId", "artifactId", "mavenVersion");
43          final Project unitUnderTest = new Project("prefix", "name", reactorParent, parentParent);
44  
45          final String expected = XmlTestUtils.readFully("testdata/project.xml");
46  
47          // Act
48          final String result = binder.marshal(unitUnderTest);
49          // System.out.println("Got: " + result);
50  
51          // Assert
52          Assert.assertTrue(XmlTestUtils.compareXmlIgnoringWhitespace(expected, result).identical());
53      }
54  
55      @Test
56      public void validateUnmarshalling() throws Exception {
57  
58          // Assemble
59          final String data = XmlTestUtils.readFully("testdata/project.xml");
60  
61          final SimpleArtifact reactorParent = new SimpleArtifact(
62                  "reactorGroupId", "reactorArtifactId", "reactorMavenVersion");
63          final SimpleArtifact parentParent = new SimpleArtifact("groupId", "artifactId", "mavenVersion");
64          final Project expected = new Project("prefix", "name", reactorParent, parentParent);
65  
66          // Act
67          final Project result = binder.unmarshalInstance(new StringReader(data));
68  
69          // Assert
70          Assert.assertEquals(expected, result);
71          Assert.assertNotSame(expected, result);
72      }
73  
74      @Test
75      public void validateComparisonAndEquality() {
76  
77          // Assemble
78          final SimpleArtifact reactorParent = new SimpleArtifact(
79                  "reactorGroupId", "reactorArtifactId", "reactorMavenVersion");
80          final SimpleArtifact parentParent = new SimpleArtifact("groupId", "artifactId", "mavenVersion");
81  
82          final Project project1 = new Project("prefix", "name", reactorParent, parentParent);
83          final Project project2 = new Project("anotherPrefix", "anotherName", reactorParent, parentParent);
84          final Project project3 = new Project("prefix", "name", reactorParent, parentParent);
85  
86          // Act & Assert
87          Assert.assertEquals(project1, project3);
88          Assert.assertEquals(project1.hashCode(), project3.hashCode());
89          Assert.assertNotSame(project1, project3);
90          Assert.assertEquals(0, project1.compareTo(project3));
91          Assert.assertEquals(project1.getName().compareTo(project2.getName()), project1.compareTo(project2));
92          Assert.assertTrue(project1.equals(project3));
93          Assert.assertFalse(project1.equals(project2));
94      }
95  }
