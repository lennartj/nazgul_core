37  /**
38   * Utility methods to simplify JPA Query usage and increase its usability.
39   *
40   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
41   */
42  public final class QueryOperations {
43  
44      /**
45       * Hide constructor for utility classes.
46       */
47      private QueryOperations() {
48          // Do nothing
49      }
50  
51      /**
52       * Acquires a T-type Entity using the provided primary key.
53       *
54       * @param entityType    The type of Entity expected.
55       * @param entityManager The active EntityManager.
56       * @param primaryKey    The primary key used to acquire the Entity.
57       * @param <T>           The type of Entity expected.
58       * @return The Entity of type T with the provided primaryKey.
59       * @throws PersistenceOperationException if a JPA-related exception occurred while performing the operation.
60       */
61      public static <T extends NazgulEntity> T findByPrimaryKey(final Class<T> entityType,
62                                                                final EntityManager entityManager,
63                                                                final Object primaryKey)
64              throws PersistenceOperationException {
65  
66          try {
67              return JpaOperations.validateInternalState(entityManager.find(entityType, primaryKey));
68          } catch (InternalStateValidationException e) {
69              throw new PersistenceOperationException("Found invalid NazgulEntity", e);
70          }
71      }
72  
73      /**
74       * Fires a JPA NamedQuery, returning the results as a List.
75       *
76       * @param query         The name of the JPA NamedQuery to fire.
77       * @param entityManager The active EntityManager.
78       * @param parameters    The parameters for the NamedQuery.
79       * @param <T>           The type of Entity expected.
80       * @return The List of resulting Entities.
81       * @throws PersistenceOperationException if a JPA-related exception occurred while performing the operation.
82       */
83      public static <T extends NazgulEntity> List<T> fireNamedQuery(final String query,
84                                                                    final EntityManager entityManager,
85                                                                    final Object... parameters)
86              throws PersistenceOperationException {
87  
88          // Delegate and return
89          return fireNamedQueryWithResultLimit(query, entityManager, Integer.MIN_VALUE, parameters);
90      }
91  
92      /**
93       * Fires a JPA NamedQuery, returning the results as a List.
94       *
95       * @param query         The name of the JPA NamedQuery to fire.
96       * @param entityManager The active EntityManager. Cannot be {@code null}.
97       * @param parameters    The parameters for the NamedQuery, assuming that the named query contains
98       *                      named parameters matching the key values within the supplied parameters Map.
99       * @param <T>           The type of Entity expected.
100      * @return The List of resulting Entities.
101      * @throws PersistenceOperationException if a JPA-related exception occurred while performing the operation.
102      */
103     public static <T extends NazgulEntity> List<T> fireNamedQuery(final String query,
104                                                                   final EntityManager entityManager,
105                                                                   final Map<String, Object> parameters)
106             throws PersistenceOperationException {
107 
108         // Delegate
109         return fireNamedQuery(query, entityManager, Integer.MIN_VALUE, parameters);
110     }
111 
112     /**
113      * Fires a JPA NamedQuery, returning the maximum given results as a List.
114      *
115      * @param query         The name of the JPA NamedQuery to fire.
116      * @param entityManager The active EntityManager. Cannot be {@code null}.
117      * @param parameters    The parameters for the NamedQuery.
118      * @param maxResults    The maximum number of results returned. Must not be a negative number.
119      * @param <T>           The type of Entity expected.
120      * @return The List of resulting Entities.
121      * @throws PersistenceOperationException if a JPA-related exception occurred while performing the operation.
122      */
123     @SuppressWarnings("unchecked")
124     public static <T extends NazgulEntity> List<T> fireNamedQueryWithResultLimit(final String query,
125                                                                                  final EntityManager entityManager,
126                                                                                  final int maxResults,
127                                                                                  final Object... parameters)
128             throws PersistenceOperationException {
129 
130         // Check sanity
131         final Query namedQuery = getQuery(query, entityManager, maxResults);
132         setParameters(namedQuery, parameters);
133 
134         // All done.
135         return fireQuery(namedQuery);
136     }
137 
138     /**
139      * Fires a JPA NamedQuery, returning the maximum given results as a List.
140      *
141      * @param query         The name of the JPA NamedQuery to fire.
142      * @param entityManager The active EntityManager. Cannot be {@code null}.
143      * @param parameters    The parameters for the NamedQuery, assuming that the named query contains
144      *                      named parameters matching the key values within the supplied parameters Map.
145      * @param maxResults    The maximum number of results returned. Must not be a negative number.
146      * @param <T>           The type of Entity expected.
147      * @return The List of resulting Entities.
148      * @throws PersistenceOperationException if a JPA-related exception occurred while performing the operation.
149      */
150     public static <T extends NazgulEntity> List<T> fireNamedQuery(final String query,
151                                                                   final EntityManager entityManager,
152                                                                   final int maxResults,
153                                                                   final Map<String, Object> parameters)
154             throws PersistenceOperationException {
155 
156         final Query namedQuery = getQuery(query, entityManager, maxResults);
157 
158         // Assign the supplied parameters.
159         for (Map.Entry<String, Object> entry : parameters.entrySet()) {
160             namedQuery.setParameter(entry.getKey(), entry.getValue());
161         }
162 
163         // Fire the query; no null Lists are returned.
164         return fireQuery(namedQuery);
165     }
166 
167     /**
168      * Assigns the supplied indexed JPQL parameters to the supplied query.
169      *
170      * @param query  The Query whose indexed parameters should be set.
171      * @param params The parameters to set.
172      */
173     public static void setParameters(final Query query,
174                                      final Object... params) {
175 
176         // Assign all parameters, if any.
177         if (params != null) {
178             for (int i = 0; i < params.length; i++) {
179                 // Bump the parameter index value with
180                 // 1, to comply with JDBC...
181                 query.setParameter((i + 1), params[i]);
182             }
183         }
184     }

187     // Private helpers
188     //
189 
190     /**
191      * Retrieves a JPA Query from the supplied EntityManager, sporting the supplied maxResults.
192      *
193      * @param query         The JPA NamedQuery to fire.
194      * @param entityManager The active EntityManager. Cannot be {@code null}.
195      * @param maxResults    Either a positive integer, or {@code Integer.MIN_VALUE} to indicate that the max result
196      *                      value should not be set.
197      * @return The JPA NamedQuery with the supplied name.
198      */
199     private static Query getQuery(@NotNull @Size(min = 1) final String query,
200                                   @NotNull final EntityManager entityManager,
201                                   final int maxResults) {
202 
203         // Check sanity
204         Validate.notEmpty(query, "query");
205         Validate.notNull(entityManager, "entityManager");
206 
207         // Create the query, and handle the amount of results returned.
208         Query namedQuery = entityManager.createNamedQuery(query);
209         if (maxResults != Integer.MIN_VALUE) {
210             namedQuery.setMaxResults(maxResults);
211         }
212 
213         // All done.
214         return namedQuery;
215     }
216 
217     @SuppressWarnings("unchecked")
218     private static <T extends NazgulEntity> List<T> fireQuery(final Query namedQuery) {
219 
220         // Fire the query; no null Lists are returned.
221         List<T> toReturn = (List<T>) namedQuery.getResultList();
222         for (T current : toReturn) {
223 
224             // Validate each NazgulEntity.
225             try {
226                 JpaOperations.validateInternalState(current);
227             } catch (InternalStateValidationException e) {
228                 throw new PersistenceOperationException("NazgulEntity state validation failed", e);
229             }
230         }
231 
232         // All done.
233         return toReturn;
234     }
