118     @Test
119     public void validateBlueprint() throws Exception {
120 
121         // Assemble
122         final long SERVICE_LOOKUP_WAIT = 2000l;
123         final long CALLTRACE_WAIT = 15000l;
124         Assert.assertNotNull("Injected BundleContext is null", context);
125 
126         // Act
127         final Bundle[] bundles = context.getBundles();
128 
129         for (int i = 0; i < bundles.length; i++) {
130             final Bundle current = bundles[i];
131 
132             // Simply printout some debug data.
133             System.out.println(" Bundle [" + i + "]: " + current.getSymbolicName() + " (" + current.getVersion()
134                     + "): " + bundleState2NameMap.get(current.getState()));
135         }
136 
137         final CallTrace callTraceService = ServiceLookup.getService(context, CallTrace.class, SERVICE_LOOKUP_WAIT);
138         System.out.println("\n\n   ==> Waiting for " + (CALLTRACE_WAIT / 1000) + " seconds for completion <==\n\n");
139         Thread.sleep(CALLTRACE_WAIT);
140 
141         final Hello service = ServiceLookup.getService(context, Hello.class, SERVICE_LOOKUP_WAIT);
142         final String result = service.sayHello();
143         final List<String> callTrace = callTraceService.getCallTrace();
144 
145         // Assert
146         Assert.assertNotNull("Got null service result", result);
147         Assert.assertNotNull("Got null callTrace result", callTrace);
148 
149         /*
150         Typical call trace:
151 
152         [Call 1/5]: No Hello services injected.
153         Added helloservice [Service ID:10, objectClasses: [se.jguru.nazgul.core.test.bundles.hello.api.Hello]]
154         Added helloservice [Service ID:9, objectClasses: [se.jguru.nazgul.core.test.bundles.hello.api.Hello]]
155         Added helloservice [Service ID:11, objectClasses: [se.jguru.nazgul.core.test.bundles.hello.api.Hello]]
156         [Call 2/5, Service 1/3]: Yo, Malin
157         [Call 2/5, Service 2/3]: Nice to see you!
158         [Call 2/5, Service 3/3]: Yo, Lennart
159         [Call 3/5, Service 1/3]: Yo, Malin
160         [Call 3/5, Service 2/3]: Nice to see you!
161         [Call 3/5, Service 3/3]: Yo, Lennart
162         [Call 4/5, Service 1/3]: Yo, Malin
163         [Call 4/5, Service 2/3]: Nice to see you!
164         [Call 4/5, Service 3/3]: Yo, Lennart
165         [Call 5/5, Service 1/3]: Yo, Malin
166         [Call 5/5, Service 2/3]: Nice to see you!
167         [Call 5/5, Service 3/3]: Yo, Lennart
168          */
169         final List<String> serviceAddedLines = CollectionAlgorithms.filter(callTrace, new Filter<String>() {
170             @Override
171             public boolean accept(final String candidate) {
172 
173                 // Typical line:
174                 //
175                 // Added helloservice [Service ID:14,
176                 // objectClasses: [se.jguru.nazgul.core.test.bundles.hello.api.Hello]]
177                 return candidate.startsWith("Added helloservice [Service ID:")
178                         && candidate.endsWith("objectClasses: [se.jguru.nazgul.core.test.bundles.hello.api.Hello]]");
179             }
180         });
181 
182         Assert.assertEquals(3, serviceAddedLines.size());
183     }
