22  package se.jguru.nazgul.core.resource.api.binary;
23  
24  import org.apache.commons.lang3.Validate;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import java.awt.Image;
29  import java.awt.Toolkit;
30  import java.net.URL;
31  
32  /**
33   * Convenience class used to load pre-bundled images from the local
34   * classpath - frequently JAR(s).
35   *
36   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
37   */
38  public abstract class BundledImageLoader {
39  
40      // Our Log
41      private static final Logger log = LoggerFactory.getLogger(BundledImageLoader.class);
42  
43      // Internal state
44      private static Toolkit toolkit;
45  
46      /**
47       * Loads/creates an Image as a resource using the ClassLoader of BundledImageLoader.
48       *
49       * @param classpathRelativeResource The classpath-relative resource to create an Image from.
50       * @return The loaded bundled Image.
51       */
52      public static Image getBundledImage(final String classpathRelativeResource) {
53          return getBundledImage(classpathRelativeResource, BundledImageLoader.class.getClassLoader());
54      }
55  
56      /**
57       * Loads/creates an Image as a resource using the provided ClassLoader.
58       *
59       * @param classpathRelativeResource The classpath-relative resource to create an Image from.
60       * @param loader                    The ClassLoader to use in loading the image.
61       * @return The loaded bundled Image.
62       * @throws IllegalArgumentException if any argument was null or empty or the actual call to getImage
63       */
64      public static Image getBundledImage(final String classpathRelativeResource, final ClassLoader loader)
65              throws IllegalArgumentException {
66  
67          // Check sanity
68          Validate.notEmpty(classpathRelativeResource, "Cannot handle null or empty classpathRelativeResource.");
69          Validate.notNull(loader, "Cannot handle null loader argument.");
70  
71          // Peel off any beginning "/".
72          final String resourcePath = (classpathRelativeResource.charAt(0) == '/'
73                  ? classpathRelativeResource.substring(1)
74                  : classpathRelativeResource);
75  
76          final URL imgURL = loader.getResource(resourcePath);
77          log.debug("Loading image (ResourcePath: " + resourcePath + ") from URL [" + imgURL + "] ");
78  
79          // No point in trying to load a nonexistent resource.
80          if (imgURL == null) {
81              return null;
82          }
83  
84          try {
85              return getToolkit().getImage(imgURL);
86          } catch (Exception e) {
87              throw new IllegalArgumentException("Could not load image from classpathRelativeResource ["
88                      + classpathRelativeResource + "]", e);
89          }
90      }
91  
92      //
93      // Private helpers
94      //
95  
96      /**
97       * Acquires the default Toolkit.
98       *
99       * @return the default Toolkit.
100      */
101     @SuppressWarnings("PMD")
102     private static Toolkit getToolkit() {
103         if (toolkit == null) {
104             toolkit = Toolkit.getDefaultToolkit();
105         }
106 
107         return toolkit;
108     }
109 }
