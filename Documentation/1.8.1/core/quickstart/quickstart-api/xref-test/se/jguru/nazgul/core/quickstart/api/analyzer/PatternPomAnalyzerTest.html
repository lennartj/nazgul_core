26  import org.junit.Before;
27  import org.junit.Test;
28  import se.jguru.nazgul.core.quickstart.api.InvalidStructureException;
29  import se.jguru.nazgul.core.quickstart.api.PomType;

32  import java.io.File;

34  /**
35   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
36   */
37  public class PatternPomAnalyzerTest extends AbstractMavenModelTest {
38  
39      // Shared state
40      private NamingStrategy namingStrategy;
41      private PatternPomAnalyzer unitUnderTest;
42  
43      @Before
44      public void setupSharedState() {
45          namingStrategy = new TestNamingStrategy();
46          unitUnderTest = new PatternPomAnalyzer(namingStrategy);
47      }
48  
49      @Test(expected = NullPointerException.class)
50      public void validateExceptionOnNullNamingStrategy() {
51  
52          // Act & Assert
53          new PatternPomAnalyzer(null);
54      }
55  
56      @Test
57      public void validateCorrectReactorPoms() {
58  
59          // Assemble
60          final File fooRootReactorPomFile = getTestDataFile("foo/pom.xml");
61          final File fooPomsReactorPomFile = getTestDataFile("foo/poms/pom.xml");
62  
63          // Act & Assert
64          final Model rootReactorPomModel = getPomModel(fooRootReactorPomFile);
65          unitUnderTest.validateRootReactorPom(rootReactorPomModel);
66          unitUnderTest.validate(getPomModel(fooPomsReactorPomFile), PomType.REACTOR, rootReactorPomModel);
67      }
68  
69      @Test
70      public void validateCorrectRootParentPoms() {
71  
72          // Assemble
73          final File fooParentPomFile = getTestDataFile("foo/poms/foo-parent/pom.xml");
74          final File fooApiParentPomFile = getTestDataFile("foo/poms/foo-api-parent/pom.xml");
75          final File fooModelParentPomFile = getTestDataFile("foo/poms/foo-model-parent/pom.xml");
76          final File fooWarParentPomFile = getTestDataFile("foo/poms/foo-war-parent/pom.xml");
77  
78          // Act & Assert
79          final Model fooParentPomModel = getPomModel(fooParentPomFile);
80          final Model fooApiParentPomModel = getPomModel(fooApiParentPomFile);
81          final Model fooModelParentPomModel = getPomModel(fooModelParentPomFile);
82          final Model fooWarParentPomModel = getPomModel(fooWarParentPomFile);
83  
84          unitUnderTest.validateTopmostParentPom(fooParentPomModel);
85          unitUnderTest.validate(fooApiParentPomModel, PomType.API_PARENT, fooParentPomModel);
86          unitUnderTest.validate(fooModelParentPomModel, PomType.MODEL_PARENT, fooApiParentPomModel);
87          unitUnderTest.validate(fooWarParentPomModel, PomType.WAR_PARENT, fooApiParentPomModel);
88      }
89  
90      @Test
91      public void validateExceptionOnIncorrectRootReactorGroupId() {
92  
93          // Assemble
94          final File fooRootReactorPomFile = getTestDataFile("invalid_rootreactor_parentgroupid/pom.xml");
95  
96          // Act
97          final Model rootReactorPomModel = getPomModel(fooRootReactorPomFile);
98  
99          // Assert
100         try {
101             unitUnderTest.validateRootReactorPom(rootReactorPomModel);
102             Assert.fail("An invalid root reactor POM should yield an exception.");
103         } catch (InvalidStructureException e) {
104             Assert.assertTrue(e.getMessage().contains(" 1: Incorrect parent groupId [some.incorrect.parent.group.id]"));
105         }
106     }
107 
108     @Test(expected = InvalidStructureException.class)
109     public void validateExceptionOnIncompatibleRootReactorPom() {
110 
111         // Assemble
112         final File fooPomsReactorPomFile = getTestDataFile("foo/poms/pom.xml");
113 
114         // Act & Assert
115         unitUnderTest.validateRootReactorPom(getPomModel(fooPomsReactorPomFile));
116     }
117 
118     @Test(expected = InvalidStructureException.class)
119     public void validateExceptionOnIncompatibleTopmostParentPom() {
120 
121         // Assemble
122         final File fooApiParentPomFile = getTestDataFile("foo/poms/foo-api-parent/pom.xml");
123 
124         // Act & Assert
125         unitUnderTest.validateTopmostParentPom(getPomModel(fooApiParentPomFile));
126     }
127 }
