23  package se.jguru.nazgul.core.algorithms.api.trees.node;
24  
25  import se.jguru.nazgul.core.algorithms.api.collections.predicate.Filter;
26  
27  /**
28   * Specification for a Node whose internal state can be manipulated, in terms
29   * of moving it within trees, and adding/removing children.
30   *
31   * @param <V> The value type of this MutableNode.
32   * @param <K> The key type of this MutableNode.
33   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
34   */
35  public interface MutableNode<K extends Comparable<K>, V> extends Node<K, V> {
36  
37      /**
38       * Adds the provided node as a child to this MutableNode.
39       *
40       * @param node The node to add.
41       * @throws IllegalArgumentException if the given node could not be added properly.
42       */
43      void addChild(MutableNode<K, V> node) throws IllegalArgumentException;
44  
45      /**
46       * Removes all immediate children of this MutableNode matching the provided Filter.
47       *
48       * @param nodeFilter a Filter defining which children should be removed.
49       */
50      void removeChildren(Filter<Node<K, V>> nodeFilter);
51  
52      /**
53       * Removes the provided child node from this MutableNode, if it exists.
54       * If the provided node is not a child node of this MutableNode, this method
55       * should return silently.
56       *
57       * @param node The node to remove.
58       */
59      void removeChild(MutableNode<K, V> node);
60  
61      /**
62       * Assigns the parent of this MutableNode.
63       *
64       * @param parent The parent of this node.
65       */
66      void setParent(MutableNode<K, V> parent);
67  
68      /**
69       * Removes this MutableNode from its parent, by removing both the
70       * reference to the parent from this MutableNode and the reference
71       * from the parent MutableNode to this MutableNode (in its list of children).
72       */
73      void clearParent();
74  }
