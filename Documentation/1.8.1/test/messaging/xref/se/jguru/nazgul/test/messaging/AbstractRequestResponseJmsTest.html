22  package se.jguru.nazgul.test.messaging;
23  
24  import javax.jms.Connection;
25  import javax.jms.DeliveryMode;
26  import javax.jms.JMSException;
27  import javax.jms.Message;
28  import javax.jms.MessageConsumer;
29  import javax.jms.MessageProducer;
30  import javax.jms.Queue;
31  import javax.jms.Session;
32  import java.util.ArrayList;
33  import java.util.List;
34  
35  /**
36   * Abstract implementation of an AbstractJmsTest which sports a simplified setup
37   * for service-side transacted MessageListeners.
38   *
39   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid, jGuru Europe AB</a>
40   */
41  public abstract class AbstractRequestResponseJmsTest extends AbstractJmsTest {
42  
43      public static final String SERVER_SIDE_INBOUND_REQUEST = "service.inbound.request";
44      public static final String SERVER_SIDE_OUTBOUND_RESPONSE = "service.outbound.response";
45      public static final String CLIENT_SIDE_OUTBOUND_REQUEST = SERVER_SIDE_INBOUND_REQUEST;
46      public static final String CLIENT_SIDE_INBOUND_RESPONSE = SERVER_SIDE_OUTBOUND_RESPONSE;
47  
48      // Shared state
49      protected List<Message> serverSideReceivedMessages;
50      protected Connection serverSideConnection;
51  
52      /**
53       * Creates a new AbstractRequestResponseJmsTest instance, returning transacted JMS objects
54       * by default as indicated by the supplied parameter.
55       *
56       * @param transactedOperation if {@code true}, all retrieved sessions will be transacted by default.
57       */
58      public AbstractRequestResponseJmsTest(final boolean transactedOperation, final MessageBroker broker) {
59          super(transactedOperation, broker);
60      }
61  
62      /**
63       * {@inheritDoc}
64       */
65      @Override
66      public void tearDownServices() throws JMSException {
67          // Do nothing.
68      }
69  
70      /**
71       * Override this method to perform any normal tear-down
72       * before the Broker is stopped. You might cleanup any
73       * instances which should be de-registered from the broker.
74       */
75      @Override
76      protected void beforeStopJmsBroker() {
77  
78          // Close the serverSideConnection.
79          try {
80              serverSideConnection.close();
81          } catch (JMSException e) {
82              throw new IllegalStateException("Could not close the serverSideConnection.", e);
83          }
84      }
85  
86      /**
87       * {@inheritDoc}
88       */
89      @Override
90      public void setupServices() throws JMSException {
91  
92          serverSideReceivedMessages = new ArrayList<Message>();
93  
94          // This is where we set up JMS objects on the server side.
95          // These objects are created before any test cases are launched.
96  
97          // 1) Get a connection to the JMS broker.
98          this.serverSideConnection = createConnection();
99  
100         // 2) Create a server-side Session, Queue and MessageConsumer reading messages from the broker.
101         final Session serverSideRequestSession = createSession(serverSideConnection);
102         final Queue serviceSideInboundQueue = serverSideRequestSession.createQueue(SERVER_SIDE_INBOUND_REQUEST);
103         final MessageConsumer requestMessageConsumer = serverSideRequestSession.createConsumer(serviceSideInboundQueue);
104 
105         // 3) Create a server-side Session, Queue and MessageProducer sending messages to the broker.
106         final Session serverSideResponseSession = createSession(serverSideConnection);
107         final Queue serviceSideOutboundQueue = serverSideResponseSession.createQueue(SERVER_SIDE_OUTBOUND_RESPONSE);
108         final MessageProducer responseMessageProducer = serverSideResponseSession
109                 .createProducer(serviceSideOutboundQueue);
110         responseMessageProducer.setDeliveryMode(DeliveryMode.NON_PERSISTENT);
111 
112         // 4) Register a MessageListener to read messages from the requestMessageConsumer
113         //    and write messages to the responseMessageProducer.
114         //    This completes the server-side setup.
115         requestMessageConsumer.setMessageListener(
116                 getServiceSideListener(
117                         serverSideReceivedMessages,
118                         serverSideResponseSession,
119                         responseMessageProducer));
120     }
121 
122     /**
123      * Creates a new AbstractTransactionalMessageListener instance for service-side use
124      * in this AbstractRequestResponseJmsTest.
125      *
126      * @param serverSideReceivedMessages The non-null List to which this AbstractTransactionalMessageListener will
127      *                                   copy all inbound messages for test tracking purposes.
128      * @param serverSideResponseSession  The non-null session used to create outbound (i.e. response) messages from
129      *                                   this MessageListener. Also used to commit JMS transactions.
130      * @param responseMessageProducer    The non-null MessageProducer, created from the supplied
131      *                                   serverSideResponseSession, used to send response messages from the
132      * @return The service-side listener used to handle inbound messages and send out responses.
133      */
134     protected abstract <T extends AbstractTransactionalMessageListener> T getServiceSideListener(
135             final List<Message> serverSideReceivedMessages,
136             final Session serverSideResponseSession,
137             final MessageProducer responseMessageProducer);
138 }
