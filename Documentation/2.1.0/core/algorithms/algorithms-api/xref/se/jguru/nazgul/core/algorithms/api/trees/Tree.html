23  package se.jguru.nazgul.core.algorithms.api.trees;
24  
25  import se.jguru.nazgul.core.algorithms.api.trees.node.MutableNode;
26  import se.jguru.nazgul.core.algorithms.api.trees.node.Node;
27  import se.jguru.nazgul.core.algorithms.api.trees.path.Path;
28  
29  /**
30   * A generic typed Tree structure holding nodes.
31   *
32   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
33   */
34  public interface Tree<K extends Comparable<K>, V> {
35  
36      /**
37       * @return The root node of this tree.
38       */
39      MutableNode<K, V> getRoot();
40  
41      /**
42       * Retrieves the Node at the provided path.
43       *
44       * @param <X>  The Node (or subtype thereof, such as {@code MutableNode}) returned.
45       * @param path The path to the node which should be acquired.
46       * @return The Node at the provided path, or {@code null} if no node was found.
47       */
48      <X extends Node<K, V>> X get(Path<K> path);
49  }
