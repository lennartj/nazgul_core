23  package se.jguru.nazgul.core.parser.api;
24  
25  import se.jguru.nazgul.core.parser.api.agent.ParserAgent;
26  
27  /**
28   * Specification for a Parser that recognizes String tokens, and offers
29   * substituting values for the token placeholders.
30   * The substitution is delegated to an extensible set of ParserAgents,
31   * to simplify complex substitution policies.
32   *
33   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
34   */
35  public interface TokenParser {
36  
37      /**
38       * Initializes this TokenParser, indicating that the supplied TokenDefinitions should be used to acquire
39       * charSequences for start and end of tokens, as well as a pattern to recognize a Token.
40       * This initialize method can only be called once, before any call to the {@code substituteTokens} method
41       * has been done.
42       *
43       * @param tokenDefinitions a non-null TokenDefinitions object.
44       * @throws java.lang.IllegalStateException if this TokenParser has already been initialized.
45       */
46      void initialize(TokenDefinitions tokenDefinitions) throws IllegalStateException;
47  
48      /**
49       * Adds a parserAgent to the list of known AbstractParserAgents.
50       *
51       * @param parserAgent the parserAgent to add.
52       * @throws IllegalArgumentException if the parserAgent argument was <code>null</code>.
53       */
54      void addAgent(ParserAgent parserAgent) throws IllegalArgumentException;
55  
56      /**
57       * Replaces the tokens found within the data.
58       *
59       * @param data The data in which to replace existing tokens.
60       * @return The token-substituted data.
61       */
62      String substituteTokens(String data);
63  }
