24  import org.apache.commons.lang3.Validate;
25  import org.junit.Assert;
26  import org.junit.rules.TestWatcher;
27  import org.junit.runner.Description;
28  
29  import javax.management.JMX;
30  import javax.management.MBeanServer;
31  import javax.management.MalformedObjectNameException;
32  import javax.management.ObjectInstance;
33  import javax.management.ObjectName;
34  import java.lang.management.ManagementFactory;
35  import java.util.ArrayList;
36  import java.util.Collections;
37  import java.util.List;

42  /**
43   * jUnit rule implementation to manage a local (i.e. in-process) MBeanServer during a Test Case.
44   * This has significance for all tests requiring a local and controllable MBean server.
45   *
46   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
47   */
48  public class LocalMBeanServerRule extends TestWatcher {
49  
50      // Shared state
51      private final Object[] lock = new Object[0];
52      private SortedSet<String> domains;
53      private MBeanServer mBeanServer;
54      private boolean unregisterAllBeansInTestDomain = true;
55      private String jmxBeanTestDomain;
56  
57      /**
58       * Default constructor creating a new LocalMBeanServerRule which unregisters all beans in its TestDomain
59       * after the test completes (i.e. in the finished() method of this class).
60       */
61      public LocalMBeanServerRule() {
62          this(true);
63      }
64  
65      /**
66       * Compound constructor creating a new LocalMBeanServerRule which performs automatic de-registration
67       * of all beans in the test JMX domain.
68       *
69       * @param unregisterAllBeansInTestDomain If {@code true}, all beans in the test JMX domain are unregistered
70       *                                       when the test finishes.
71       * @see #finished(Description)
72       */
73      public LocalMBeanServerRule(final boolean unregisterAllBeansInTestDomain) {
74  
75          // Assign internal state
76          this.unregisterAllBeansInTestDomain = unregisterAllBeansInTestDomain;
77      }
78  
79      /**
80       * <p>Starts a local/in-process MBeanServer, and validates that the JMX test domain is empty (i.e. not
81       * registered as a domain in the JMX server when the test starts). The test domain is identical to the
82       * class name of the test class:</p>
83       * <pre>
84       * <code>
85       *     // Stash the name of the test JMX domain.
86       *     setJmxBeanTestDomain(description.getClassName());
87       * </code>
88       * </pre>
89       *
90       * @see #getJmxBeanTestDomain()
91       * @see #setJmxBeanTestDomain(String)
92       */
93      @Override
94      protected void starting(final Description description) {
95  
96          // Acquire the local MBeanServer.
97          mBeanServer = ManagementFactory.getPlatformMBeanServer();
98          Assert.assertNotNull("Could not retrieve PlatformMBeanServer.", mBeanServer);
99  
100         // Stash the name of the test JMX domain.
101         setJmxBeanTestDomain(description.getClassName());
102 
103         // Retrieve the existing domains from the mBeanServer.
104         domains = new TreeSet<>();
105         getDomains();
106 
107         // The JMX bean test domain should be empty at the start of the unit test.
108         Assert.assertFalse("Test domain [" + getJmxBeanTestDomain() + "] already present within local mBeanServer. "
109                         + "Your unit test state should be cleaned out before launching a JMX test.",
110                 this.domains.contains(getJmxBeanTestDomain()));
111     }
112 
113     /**
114      * Calls {@code mBeanServer.unregisterMBean(objectName);} for each MBean whose objectName starts with the
115      * jmxBeanTestDomain string - which normally is the same as the fully qualified class name of the test
116      * class being executed.
117      *
118      * @throws MBeanServerUnregisterException if the LocalMBeanServerRule was instructed to unregister all beans in
119      *                                        the test domain and was unable to do so.
120      */
121     @Override
122     protected void finished(final Description description) {
123 
124         final List<Exception> exceptions = new ArrayList<>();
125 
126         // Remove the test domain and all its sub-elements, if asked to.
127         if (unregisterAllBeansInTestDomain) {
128             for (ObjectInstance current : mBeanServer.queryMBeans(null, null)) {
129                 final ObjectName objectName = current.getObjectName();
130                 if (objectName.getDomain().startsWith(getJmxBeanTestDomain())) {
131                     try {
132                         mBeanServer.unregisterMBean(objectName);
133                     } catch (Exception e) {
134                         exceptions.add(e);
135                     }
136                 }
137             }
138         }
139 
140         if (exceptions.size() > 0) {
141             final MBeanServerUnregisterException ex = new MBeanServerUnregisterException("unregistering MBeans");
142             ex.getExceptions().addAll(exceptions);
143         }
144     }
145 
146     /**
147      * Override to provide a JMX domain other than the default test domain.
148      *
149      * @return {@code getClass().getPackage().getName()}, unless overridden in subclasses.
150      */
151     public String getJmxBeanTestDomain() {
152         return jmxBeanTestDomain;
153     }
154 
155     /**
156      * Assigns the JMX bean test domain, which is a domain where test code is applied.
157      * The JMX bean test domain is normally empty at the start of the unit.
158      *
159      * @param jmxBeanTestDomain The JMX bean test domain to assign to the local MBeanServer.
160      */
161     public void setJmxBeanTestDomain(final String jmxBeanTestDomain) {
162 
163         // Check sanity
164         Validate.notNull(jmxBeanTestDomain, "Cannot handle null 'jmxBeanTestDomain' argument.");
165 
166         // Assign internal state
167         this.jmxBeanTestDomain = jmxBeanTestDomain;
168     }
169 
170     /**
171      * @return An unmodifiable SortedSet containing the domains currently registered within the MBean server.
172      */
173     public SortedSet<String> getDomains() {
174 
175         final String[] currentDomains = mBeanServer.getDomains();
176         if (currentDomains.length != domains.size()) {
177             synchronized (lock) {
178 
179                 // Retrieve the existing domains from the mBeanServer.
180                 domains.clear();
181                 Collections.addAll(domains, currentDomains);
182             }
183         }
184 
185         // All done.
186         return Collections.unmodifiableSortedSet(domains);
187     }
188 
189     /**
190      * Retrieves the value of the attribute with the supplied name from the given ObjectName.
191      *
192      * @param objectName    The objectName of the MBean whose readable attribute should be retrieved.
193      * @param attributeName The name of the attribute. <strong>Note!</strong> The attribute should be capitalized,
194      *                      implying that a JavaBean getter {@code String getFoo()} implies an attributeName
195      *                      "Foo" (not "foo"), according to the JMX specification.
196      * @param <T>           The type of the JMX attribute requested.
197      * @return The value of the attribute in the supplied objectName.
198      */
199     public final <T> T getAttribute(final ObjectName objectName, final String attributeName) {
200 
201         try {
202             return (T) mBeanServer.getAttribute(objectName, attributeName);
203         } catch (Exception e) {
204             throw new IllegalArgumentException("Attribute [" + attributeName + "] was not found in objectName ["
205                     + objectName + "]", e);
206         }
207     }
208 
209     /**
210      * Retrieves an MBean proxy with the supplied interface type from the local MBeanServer.
211      *
212      * @param objectName    The objectName of the MBean which must implement the supplied interfaceType.
213      * @param interfaceType The interface type which should be retrieved.
214      * @param <T>           The interface type to retrieve.
215      * @return An MBean proxy implementing the supplied interfaceType.
216      */
217     public final <T> T getMXBeanProxy(final ObjectName objectName, final Class<T> interfaceType) {
218         return JMX.newMXBeanProxy(mBeanServer, objectName, interfaceType, true);
219     }
220 
221     /**
222      * Retrieves all ObjectInstances within the JMX domain supplied.
223      *
224      * @param jmxDomain a name of a JMX domain. Cannot be null.
225      * @return a Set containing all ObjectInstances (i.e. MBeans) within the supplied JMX domain.
226      */
227     public Set<ObjectInstance> getMBeansInDomain(final String jmxDomain) {
228         return mBeanServer.queryMBeans(getSearchObjectNameFor(jmxDomain), null);
229     }
230 
231     /**
232      * Retrieves all ObjectNames within the JMX domain supplied.
233      *
234      * @param jmxDomain a name of a JMX domain. Cannot be null.
235      * @return a Set containing all ObjectNames within the supplied JMX domain.
236      */
237     public Set<ObjectName> getNamesInDomain(final String jmxDomain) {
238         return mBeanServer.queryNames(getSearchObjectNameFor(jmxDomain), null);
239     }
240 
241     /**
242      * Retrieves the active MBeanServer from this Rule.
243      * Normally, this
244      *
245      * @return the active MBeanServer from this Rule.
246      */
247     public MBeanServer getMBeanServer() {
248         return mBeanServer;
249     }
250 
251     //
252     // Private helpers
253     //
254 
255     private ObjectName getSearchObjectNameFor(final String jmxDomain) {
256 
257         // Check sanity
258         Validate.notNull(jmxDomain, "Cannot handle null 'jmxDomain' argument.");
259 
260         try {
261             return new ObjectName(jmxDomain + ":*");
262         } catch (MalformedObjectNameException e) {
263             throw new IllegalArgumentException("Could not create ObjectName for jmxDomain '" + jmxDomain + "'", e);
264         }
265     }
