98      /**
99       * Acquires a properly configured JAXB marshaller from the provided JAXBContext.
100      * <pre><code>
101      *  // Acquire the JAXBContext for the types found within an EntityTransporter.
102      *  final NamespacePrefixMapper mapper = ...
103      *  final JAXBContext ctx = JaxbUtils.getJaxbContext(anEntityTransporter);
104      *
105      *  // Now acquire the marshaller
106      *  final Marshaller marshaller = JaxbUtils.getHumanReadableStandardMarshaller(ctx, resolver);
107      *
108      *  // ... and use it ...
109      *  marshaller.marshal(someInstance, someOutputTarget);
110      * </code></pre>
111      *
112      * @param ctx                   The properly set up JAXBContext, holding all required class definitions.
113      * @param namespacePrefixMapper A JAXB NamespacePrefixMapper to be used by the provided marshaller.
114      *                              Cannot be {@code null}.
115      * @param validate              if {@code true}, performs validation - implying that the resourceResolver
116      *                              must be non-null.
117      * @return A standard JAXB Marshaller for human-readable (as opposed to extreme compactness) XML marshalling.
118      * @throws NullPointerException if the {@code namespacePrefixMapper} was {@code null}.
119      */
120     public static Marshaller getHumanReadableStandardMarshaller(final JAXBContext ctx,
121                                                                 final NamespacePrefixMapper namespacePrefixMapper,
122                                                                 final boolean validate)
123             throws NullPointerException {
124 
125         Validate.notNull(namespacePrefixMapper, "Cannot handle null namespacePrefixMapper argument.");
126 
127         try {
128 
129             // Acquire the properly configured Marshaller.
130             final Marshaller toReturn = ctx.createMarshaller();
131             toReturn.setProperty(Marshaller.JAXB_ENCODING, "UTF-8");
132             toReturn.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
133             toReturn.setProperty(EXTERNAL_JAXB_NAMESPACEPREFIXMAPPER_KEY, namespacePrefixMapper);
134 
135             // Should we validate what we write?
136             if (validate) {
137                 toReturn.setSchema(generateTransientXSD(ctx).getKey());
138             }
139 
140             return toReturn;
141         } catch (final JAXBException e) {
142             throw new IllegalStateException("Could not create marshaller", e);
143         }
144     }

146     /**
147      * Retrieves a JAXBContext instance, geared to converting all class types found within the
148      * provided transporter.
149      *
150      * @param transporter   The EntityTransporter which should be marshalled to XML.
151      * @param isMarshalling {@code true} if the supplied JAXBContext is going to be used for marshalling,
152      *                      as opposed to unmarshalling.
153      * @return A JAXBContext able to marshal the provided transporter.
154      * @throws NullPointerException if the transporter argument was {@code null}.
155      */
156     @SuppressWarnings({"rawtypes", "unchecked"})
157     public static JAXBContext getJaxbContext(final EntityTransporter transporter,
158                                              final boolean isMarshalling)
159             throws NullPointerException {
160 
161         // Check sanity
162         Validate.notNull(transporter, "Cannot handle null transporter argument.");
163 
164         // Does a cached JAXBContext exist?
165         final Tuple<SortedClassNameSetKey, JAXBContext> cachedContext
166                 = getCachedJaxbContext(transporter.getClassInformation());
167         if (cachedContext.getValue() != null) {
168             return cachedContext.getValue();
169         }
170 
171         // Load all relevant classes
172         Set<Class<?>> loadedClasses = new HashSet<Class<?>>();
173         for (Object current : transporter.getClassInformation()) {
174 
175             final Class<?> classType = THREADLOCAL_TRANSFORMER.transform((String) current);
176             addNonTransientInternalClass(loadedClasses, classType);
177         }
178 
179         // If we are marshalling, acquire the non-transient internal
180         // field types for the
181         if (isMarshalling) {
182 
183             // The items List should be fully populated here.
184             for (Object currentItem : transporter.getItems()) {
185                 addNonTransientInternalFieldTypes(loadedClasses, currentItem);
186             }
187         }
188 
189         try {
190 
191             // Create the JAXBContext and cache it.
192             JAXBContext toReturn = JAXBContext.newInstance(loadedClasses.toArray(new Class[loadedClasses.size()]));
193             jaxbContextCache.put(cachedContext.getKey(), toReturn);
194 
195             return toReturn;
196         } catch (JAXBException e) {
197             throw new IllegalStateException("Could not create JAXBContext", e);
198         }
199     }

201     /**
202      * Acquires a JAXB Schema from the provided JAXBContext.
203      *
204      * @param ctx The context for which am XSD should be constructed.
205      * @return A tuple holding the constructed XSD from the provided JAXBContext, and
206      * the LSResourceResolver synthesized during the way.
207      * @throws NullPointerException     if ctx was {@code null}.
208      * @throws IllegalArgumentException if a JAXB-related exception occurred while extracting the schema.
209      */
210     public static Tuple<Schema, LSResourceResolver> generateTransientXSD(final JAXBContext ctx)
211             throws NullPointerException, IllegalArgumentException {
212 
213         // Check sanity
214         Validate.notNull(ctx, "Cannot handle null ctx argument.");
215 
216         final SortedMap<String, ByteArrayOutputStream> namespace2SchemaMap
217                 = new TreeMap<String, ByteArrayOutputStream>();
218 
219         try {
220             ctx.generateSchema(new SchemaOutputResolver() {
221                 @Override
222                 public Result createOutput(final String namespaceUri, final String suggestedFileName)
223                         throws IOException {
224 
225                     // The types should really be annotated with @XmlType(namespace = "... something ...")
226                     // to avoid using the default ("") namespace.
227                     if (namespaceUri.isEmpty()) {
228                         log.warn("Received empty namespaceUri while resolving a generated schema. "
229                                 + "Did you forget to add a @XmlType(namespace = \"... something ...\") annotation "
230                                 + "to your class?");
231                     }
232 
233                     // Create the result ByteArrayOutputStream
234                     final ByteArrayOutputStream out = new ByteArrayOutputStream();
235                     final StreamResult toReturn = new StreamResult(out);
236                     toReturn.setSystemId("");
237 
238                     // Map the namespaceUri to the schemaResult.
239                     namespace2SchemaMap.put(namespaceUri, out);
240 
241                     // All done.
242                     return toReturn;
243                 }
244             });
245         } catch (IOException e) {
246             throw new IllegalArgumentException("Could not acquire Schema snippets.", e);
247         }
248 
249         // Convert to an array of StreamSource.
250         final MappedSchemaResourceResolver resourceResolver = new MappedSchemaResourceResolver();
251         final StreamSource[] schemaSources = new StreamSource[namespace2SchemaMap.size()];
252         int counter = 0;
253         for (Map.Entry<String, ByteArrayOutputStream> current : namespace2SchemaMap.entrySet()) {
254 
255             final byte[] schemaSnippetAsBytes = current.getValue().toByteArray();
256             resourceResolver.addNamespace2SchemaEntry(current.getKey(), new String(schemaSnippetAsBytes));
257 
258             if (log.isDebugEnabled()) {
259                 log.info("Generated schema [" + (counter + 1) + "/" + schemaSources.length + "]:\n "
260                         + new String(schemaSnippetAsBytes));
261             }
262 
263             // Copy the schema source to the schemaSources array.
264             schemaSources[counter] = new StreamSource(new ByteArrayInputStream(schemaSnippetAsBytes), "");
265 
266             // Increase the counter
267             counter++;
268         }
269 
270         try {
271 
272             // All done.
273             final SchemaFactory schemaFactory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
274             schemaFactory.setResourceResolver(resourceResolver);
275             final Schema transientSchema = schemaFactory.newSchema(schemaSources);
276 
277             // All done.
278             return new Tuple<Schema, LSResourceResolver>(transientSchema, resourceResolver);
279 
280         } catch (final SAXException e) {
281             throw new IllegalArgumentException("Could not create Schema from snippets.", e);
282         }
283     }

285     /**
286      * Extracts relevant data from the supplied {@code toWrapAndPackageForTransport} object,
287      * appending the JAXB-transportable objects into the {@code resultingTransportableObjects} List,
288      * and appending the transport type Class names into the {@code resultingTransportTypes} List.
289      *
290      * @param toWrapAndPackageForTransport  The instance from which transport data should be extracted
291      *                                      and appended onto the resultingTransportableObjects and
292      *                                      resultingTransportTypes Lists, respectively.
293      * @param registry                      The TransportTypeConverterRegistry instance used to extract
294      *                                      TransportTypeConverter instances, in turn used to translate
295      *                                      non-JAXB-annotated types to JaxbAnnotatedTypes for transport.
296      * @param resultingTransportableObjects The non-null List holding JAXB-convertible objects.
297      * @param resultingTransportTypes       The non-null SortedSet holding the class names of all
298      *                                      transport types.
299      */
300     public static void extractJaxbTransportData(final Object toWrapAndPackageForTransport,
301                                                 final JaxbConverterRegistry registry,
302                                                 final List<Object> resultingTransportableObjects,
303                                                 final SortedSet<String> resultingTransportTypes) {
304 
305         // Check sanity
306         Validate.notNull(registry, "Cannot handle null TransportTypeConverterRegistry argument.");
307         Validate.notNull(resultingTransportableObjects, "Cannot handle null resultingTransportableObjects argument.");
308         Validate.notNull(resultingTransportTypes, "Cannot handle null resultingTransportTypes argument.");
309 
310         // ### 1) Convert the current type if required.
311         Object added = toWrapAndPackageForTransport;
312         final Class<?> transportType = toWrapAndPackageForTransport == null
313                 ? JaxbAnnotatedNull.class
314                 : registry.getTransportType(toWrapAndPackageForTransport.getClass());
315 
316         final boolean wrapInTransportType = added == null || added.getClass() != transportType;
317         if (wrapInTransportType) {
318             added = registry.packageForTransport(toWrapAndPackageForTransport);
319         }
320 
321         // ### 2) Add the item itself.
322         resultingTransportableObjects.add(added);
323 
324         // ### 3) Add the JAXB-annotated transport classes found within
325         //        the added object, via the use of its TypeExtractor methods.
326         if (added instanceof ClassInformationHolder) {
327             for (String current : ((ClassInformationHolder) added).getClassInformation()) {
328 
329                 if (!resultingTransportTypes.contains(current) && !current.startsWith(JAVALANGCLASS)) {
330                     resultingTransportTypes.add(current);
331                 }
332             }
333         }
334 
335         // ### 4) Add the class of the added object itself. Handle Array types.
336         final Class<?> addedClass = added.getClass();
337         final boolean addedClassIsArray = addedClass.isArray();
338         final String addedClassName = addedClassIsArray
339                 ? addedClass.getComponentType().getName()
340                 : addedClass.getName();
341         if (!addedClassName.startsWith(JAVALANGCLASS)
342                 && !resultingTransportTypes.contains(addedClassName)
343                 && !(addedClassIsArray && addedClass.getComponentType().isPrimitive())) {
344             resultingTransportTypes.add(addedClassName);
345         }
346 
347         // ### 5) Only add internal structure if the current class is not a AbstractJaxbAnnotatedTransportType.
348         //        Acquire internal reflected state as well.
349         if (!wrapInTransportType) {
350 
351             final Set<Class<?>> internalReflectedTypes = new HashSet<Class<?>>();
352             final Set<Class<?>> transportTypes = new HashSet<Class<?>>();
353 
354             addNonTransientInternalFieldTypes(internalReflectedTypes, toWrapAndPackageForTransport);
355             for (Class<?> current : internalReflectedTypes) {
356 
357                 // Don't add a null transport type.
358                 final Class<?> transportTypeOrNull = registry.getTransportType(current);
359                 final Class<?> toAdd = transportTypeOrNull == null ? current : transportTypeOrNull;
360                 addNonTransientInternalClass(transportTypes, toAdd);
361             }
362 
363             // Fire the internally found reflected types through the JaxbConverter
364             for (Class<?> current : transportTypes) {
365                 final String fullyQualifiedClassName = current.getName();
366                 if (!resultingTransportTypes.contains(fullyQualifiedClassName)) {
367                     resultingTransportTypes.add(fullyQualifiedClassName);
368                 }
369             }
370         }
371     }

377     /**
378      * Adds all field types found in the {@code toReflect} object to the provided types Set.
379      * Should any Field in the {@code toReflect} instance be a Collection, Array or Map, any
380      * classes found within the collections are added as well.
381      *
382      * @param types     The resulting types set.
383      * @param toReflect The object from whose internal Fields the type information should be extracted.
384      */
385     @SuppressWarnings("rawtypes")
386     private static void addNonTransientInternalFieldTypes(final Set<Class<?>> types, final Object toReflect) {
387 
388         // Check sanity
389         if (toReflect == null) {
390             return;
391         }
392 
393         // First, add the current Class to the types Set
394         final Class<?> theType = toReflect.getClass();
395         addNonTransientInternalClass(types, theType);
396 
397         // Don't reflect EntityTransporter classes
398         if (theType == EntityTransporter.class) {
399             return;
400         }
401 
402         for (Field current : XmlMarshallableFieldFilter.getMarshallableFields(toReflect)) {
403 
404             // Get the type of the object stored within the current field
405             Object currentValue = get(current, toReflect);
406             if (currentValue != null) {
407 
408                 // Add the class of the currentValue
409                 final Class<?> currentType = currentValue.getClass();
410                 addNonTransientInternalClass(types, currentType);
411 
412                 // Handle Collections, Arrays and Maps, which may contain
413                 // implementation types which should be included.
414 
415                 if (currentType.getClass().isArray()) {
416                     addNonTransientInternalClass(types, currentType.getComponentType());
417                 } else if (Collection.class.isAssignableFrom(currentType)) {
418 
419                     // This is a collection. Dig out the types of all Elements.
420                     for (Object currentElement : Collection.class.cast(get(current, toReflect))) {
421                         addNonTransientInternalFieldTypes(types, currentElement);
422                     }
423                 } else if (Map.class.isAssignableFrom(currentType)) {
424 
425                     // This is a map. Dig out the types of all Keys and Value.
426                     final Map theMap = Map.class.cast(get(current, toReflect));
427                     for (Object currentKey : theMap.keySet()) {
428 
429                         addNonTransientInternalClass(types, currentKey.getClass());
430 
431                         final Object currentMapValue = theMap.get(currentKey);
432                         if (currentMapValue != null) {
433                             addNonTransientInternalClass(types, currentMapValue.getClass());
434                         }
435                     }
436                 }
437             }
438         }
439     }

442      * Retrieves the value of the supplied Field from the given instance.
443      *
444      * @param aField   The Field whose value should be retrieved.
445      * @param anObject The instance whose class contains the supplied Field, and
446      *                 from which the value should be retrieved.
447      * @return the value of the supplied Field from the given instance.
448      */
449     private static Object get(final Field aField, final Object anObject) {
450 
451         aField.setAccessible(true);
452         try {
453             return aField.get(anObject);
454         } catch (Exception e) {
455             return null;
456         }
457     }

459     /**
460      * Adds the given candidate Class to the supplied set of types, given that the candidate
461      * passes some trivial checks.
462      *
463      * @param types     The types set to which the candidate could be added.
464      * @param candidate The type to [possibly] add to the given types Set.
465      */
466     private static void addNonTransientInternalClass(final Set<Class<?>> types, final Class<?> candidate) {
467 
468         if (candidate == null) {
469             return;
470         }
471 
472         // Don't add XmlTransient classes
473         final boolean isXmlTransient = candidate.getAnnotation(XmlTransient.class) != null;
474 
475         // Ignore interfaces
476         final boolean isInterface = candidate.isInterface();
477 
478         // Ignore primitives and Object
479         final boolean isPrimitive = candidate.isPrimitive();
480         final boolean isObject = candidate == Object.class;
481 
482         // Ignore array types
483         final boolean isArray = candidate.isArray();
484 
485         if (!isXmlTransient && !isInterface && !isPrimitive && !isObject && !isArray) {
486             types.add(candidate);
487         }
488     }

490     /**
491      * Creating new JAXBContexts is an expensive operation; caching a few pre-created ones and re-using
492      * them speeds up the process of marshalling and unmarshalling considerably.
493      *
494      * @param classInformation The classes from an EntityTransporter for which a JAXBContext should be acquired.
495      * @return A Tuple holding the best-match cache key for the EntityTransporter, and the corresponding
496      * cached JAXBContext. The JAXBContext value might be {@code null}, indicating that a
497      * cached JAXBContext was not found.
498      */
499     private static Tuple<SortedClassNameSetKey, JAXBContext> getCachedJaxbContext(
500             final SortedSet<String> classInformation) {
501 
502         // Create a sorted clone.
503         final SortedSet<String> copy = new TreeSet<String>(classInformation);
504         copy.add(EntityTransporter.class.getName());
505         final SortedClassNameSetKey key = new SortedClassNameSetKey(copy);
506 
507         JAXBContext jaxbContext = jaxbContextCache.get(key);
508         if (jaxbContext != null) {
509 
510             // Exact existing match. All done.
511             return new Tuple<SortedClassNameSetKey, JAXBContext>(key, jaxbContext);
512         }
513 
514         // Search for matching/compatible SortedClassNameSetKeys.
515         // The first matching one can be used for JAXBContext.
516         Tuple<SortedClassNameSetKey, JAXBContext> toReturn = null;
517         for (Map.Entry<SortedClassNameSetKey, JAXBContext> current : jaxbContextCache.entrySet()) {
518             if (current.getKey().containsAll(copy)) {
519                 toReturn = new Tuple<SortedClassNameSetKey, JAXBContext>(current.getKey(), current.getValue());
520             }
521         }
522 
523         if (toReturn == null) {
524             // No cache entry found. Create one.
525             toReturn = new Tuple<SortedClassNameSetKey, JAXBContext>(key, null);
526         }
527 
528         // All done.
529         return toReturn;
530     }
