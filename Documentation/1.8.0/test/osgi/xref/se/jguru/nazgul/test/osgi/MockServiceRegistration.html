23  package se.jguru.nazgul.test.osgi;
24  
25  import org.apache.commons.lang3.Validate;
26  import org.osgi.framework.ServiceEvent;
27  import org.osgi.framework.ServiceReference;
28  import org.osgi.framework.ServiceRegistration;
29  import se.jguru.nazgul.core.algorithms.event.api.producer.EventConsumerCallback;
30  import se.jguru.nazgul.test.osgi.event.ServiceListenerAdapter;
31  
32  import java.util.Dictionary;
33  import java.util.Enumeration;
34  import java.util.Properties;
35  
36  /**
37   * Mock implementation of an OSGi ServiceRegistration, usable in unit tests.
38   *
39   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
40   */
41  public class MockServiceRegistration<S> implements ServiceRegistration<S> {
42  
43      // Internal state
44      private final MockServiceReference serviceReference;
45      private final MockBundleContext bundleContext;
46      private final Dictionary registrationProperties;
47  
48      /**
49       * Creates a new MockServiceRegistration instance wrapping the provided
50       * reference, context and registration properties.
51       *
52       * @param serviceReference       The ServiceReference to wrap.
53       * @param bundleContext          The BundleContext to wrap.
54       * @param registrationProperties The OSGi service registration properties.
55       */
56      public MockServiceRegistration(final MockServiceReference serviceReference,
57                                     final MockBundleContext bundleContext,
58                                     final Dictionary registrationProperties) {
59  
60          // Check sanity
61          Validate.notNull(serviceReference, "Cannot handle null serviceReference argument.");
62          Validate.notNull(bundleContext, "Cannot handle null bundleContext argument.");
63          Validate.notNull(registrationProperties, "Cannot handle null registrationProperties argument.");
64  
65          // Assign internal state
66          this.serviceReference = serviceReference;
67          this.bundleContext = bundleContext;
68          this.registrationProperties = registrationProperties;
69      }
70  
71      /**
72       * Creates a new MockServiceRegistration instance with an
73       * empty registrationProperties dictionary.
74       *
75       * @param serviceReference The ServiceReference to wrap.
76       * @param bundleContext    The BundleContext to wrap.
77       */
78      public MockServiceRegistration(final MockServiceReference serviceReference,
79                                     final MockBundleContext bundleContext) {
80          this(serviceReference, bundleContext, new Properties());
81      }
82  
83      /**
84       * Returns a {@code ServiceReference} object for a service being
85       * registered.
86       * The {@code ServiceReference} object may be shared with other
87       * bundles.
88       *
89       * @return {@code ServiceReference} object.
90       * @throws IllegalStateException If this
91       *                               {@code ServiceRegistration} object has already been
92       *                               unregistered.
93       */
94      @Override
95      public ServiceReference<S> getReference() {
96          return serviceReference;
97      }
98  
99      /**
100      * Updates the properties associated with a service.
101      * The {@link org.osgi.framework.Constants#OBJECTCLASS} and {@link org.osgi.framework.Constants#SERVICE_ID} keys
102      * cannot be modified by this method. These values are set by the Framework
103      * when the service is registered in the OSGi environment.
104      * The following steps are required to modify service properties:
105      * <ol>
106      * <li>The service's properties are replaced with the provided properties.
107      * <li>A service event of type {@link org.osgi.framework.ServiceEvent#MODIFIED} is fired.
108      * </ol>
109      *
110      * @param properties The properties for this service. See {@link org.osgi.framework.Constants}
111      *                   for a list of standard service property keys. Changes should not
112      *                   be made to this object after calling this method. To update the
113      *                   service's properties this method should be called again.
114      * @throws IllegalStateException    If this {@code ServiceRegistration}
115      *                                  object has already been unregistered.
116      * @throws IllegalArgumentException If {@code properties} contains
117      *                                  case variants of the same key name.
118      */
119     @Override
120     public void setProperties(final Dictionary<String, ?> properties) {
121         for (Enumeration en = properties.keys(); en.hasMoreElements(); ) {
122             final Object currentKey = en.nextElement();
123             registrationProperties.put(currentKey, properties.get(currentKey));
124         }
125 
126         // Fire the modified event
127         bundleContext.fireServiceEvent(new EventConsumerCallback<ServiceListenerAdapter>() {
128             @Override
129             public void onEvent(final ServiceListenerAdapter eventConsumer) {
130                 eventConsumer.serviceChanged(new ServiceEvent(ServiceEvent.MODIFIED, serviceReference));
131             }
132         });
133     }
134 
135     /**
136      * Unregisters a service. Remove a {@code ServiceRegistration} object
137      * from the Framework service registry. All {@code ServiceReference}
138      * objects associated with this {@code ServiceRegistration} object
139      * can no longer be used to interact with the service once unregistration is
140      * complete.
141      * The following steps are required to unregister a service:
142      * <ol>
143      * <li>The service is removed from the Framework service registry so that
144      * it can no longer be obtained.
145      * <li>A service event of type {@link org.osgi.framework.ServiceEvent#UNREGISTERING} is fired
146      * so that bundles using this service can release their use of the service.
147      * Once delivery of the service event is complete, the
148      * {@code ServiceReference} objects for the service may no longer be
149      * used to get a service object for the service.
150      * <li>For each bundle whose use count for this service is greater than
151      * zero: <br>
152      * The bundle's use count for this service is set to zero. <br>
153      * If the service was registered with a {@link org.osgi.framework.ServiceFactory} object, the
154      * {@code ServiceFactory.ungetService} method is called to release
155      * the service object for the bundle.
156      * </ol>
157      *
158      * @throws IllegalStateException If this
159      *                               {@code ServiceRegistration} object has already been
160      *                               unregistered.
161      * @see org.osgi.framework.BundleContext#ungetService
162      * @see org.osgi.framework.ServiceFactory#ungetService
163      */
164     @Override
165     public void unregister() {
166         bundleContext.unregister(serviceReference);
167     }
168 }
