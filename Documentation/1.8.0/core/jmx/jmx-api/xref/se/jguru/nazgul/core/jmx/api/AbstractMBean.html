37  /**
38   * Abstract StandardEmitterMBean implementation adhering to the StandardLifecycle.
39   *
40   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid, jGuru Europe AB</a>
41   */
42  public abstract class AbstractMBean extends StandardEmitterMBean implements LifecycleStateful {
43  
44      // Our log
45      private static final Logger log = LoggerFactory.getLogger(AbstractMBean.class);
46  
47      /**
48       * The name of the state attribute.
49       */
50      public static final String STATE_CHANGE_ATTRIBUTENAME = "state";
51  
52      // Internal state
53      private LifecycleState state = LifecycleState.UNINITIALIZED;
54      private int eventSequenceId = 1;
55      private final Object lock = new Object();

57      /**
58       * <p>Convenience constructor, creating a new AbstractMBean wrapping the supplied data.
59       * All NotificationEmitter operations executed in this AbstractMBean are delegated to
60       * the provided NotificationEmitter object. Thus, all JMX notifications fired by this
61       * AbstractMBean are simply delegated to the supplied NotificationEmitter.</p>
62       * <p>This constructor must be called from a subclass that implements the provided {@code mbeanInterface}.</p>
63       *
64       * @param mbeanInterface a StandardMBean interface.
65       * @param delegate       A non-null NotificationEmitter to which this AbstractMBean will delegate all
66       *                       NotificationEmitter operations.
67       * @throws IllegalArgumentException if the {@code mbeanInterface} does not follow JMX design patterns for
68       *                                  Management Interfaces, or if {@code this} does not implement the specified
69       *                                  interface, or if {@code delegate} is null.
70       * @see javax.management.StandardEmitterMBean#StandardEmitterMBean(Class, boolean,
71       * javax.management.NotificationEmitter)
72       */
73      public AbstractMBean(final Class<?> mbeanInterface,
74                           final NotificationEmitter delegate) {
75          this(mbeanInterface, true, delegate);
76      }
77  
78      /**
79       * <p>Creates an AbstractMBean with the management interface {@code mbeanInterface}, and where notifications are
80       * handled by the given {@code NotificationEmitter}. This constructor can be used to make either Standard MBeans
81       * or MXBeans. The resultant MBean implements the {@code NotificationEmitter} interface by forwarding its methods
82       * to {@code delegate}.</p>
83       * <p>If {@code delegate} is an instance of {@code NotificationBroadcasterSupport} then the MBean's {@link
84       * #sendNotification sendNotification} method will delegate its invocation to {@code delegate.}
85       * {@link javax.management.NotificationBroadcasterSupport#sendNotification(Notification)}.</p>
86       * <p>The array returned by {@link #getNotificationInfo()} on the new MBean is a copy of the array returned by
87       * {@code emitter.}{@link NotificationBroadcaster#getNotificationInfo()} at the time of
88       * construction. If the array returned by {@code emitter.getNotificationInfo()} later changes, that will have no
89       * effect on this object's {@code getNotificationInfo()}.</p>
90       * <p>This constructor must be called from a subclass that implements the given {@code mbeanInterface}.</p>
91       *
92       * @param mbeanInterface a StandardMBean interface.
93       * @param isMXBean       If true, the {@code mbeanInterface} parameter
94       *                       names an MXBean interface and the resultant MBean is an MXBean.
95       * @param delegate       A non-null NotificationEmitter to which this AbstractMBean will delegate all
96       *                       NotificationEmitter operations.
97       * @throws IllegalArgumentException if the {@code mbeanInterface}
98       *                                  does not follow JMX design patterns for Management Interfaces, or
99       *                                  if {@code this} does not implement the specified interface, or
100      *                                  if {@code delegate} is null.
101      */
102     public AbstractMBean(final Class<?> mbeanInterface,
103                          final boolean isMXBean,
104                          final NotificationEmitter delegate) {
105         super(mbeanInterface, isMXBean, delegate);
106     }

109      * {@inheritDoc}
110      */
111     @Override
112     public final LifecycleState getState() {
113         return state;

116     /**
117      * Standard implementation of the MBeanRegistration lifecycle method, delegating execution to (in order):
118      * <ol>
119      * <li>{@code customPreregister()}, where subclasses may manipulate the JMX ObjectName.</li>
120      * <li>{@code super.preRegister()}, performing the actual registration of this AbstractMBean
121      * in the MBeanServer.</li>
122      * </ol>
123      * {@inheritDoc}
124      *
125      * @see #customPreregister(javax.management.MBeanServer, javax.management.ObjectName)
126      */
127     @Override
128     public final ObjectName preRegister(final MBeanServer server, final ObjectName name) throws Exception {
129 
130         // Check sanity
131         if (name == null) {
132             throw new IllegalArgumentException("You should define an ObjectName for the MBean ["
133                     + getClass().getName() + "]. Nameless AbstractMBeans are not permitted.");
134         }
135 
136         // Perform any customizations on the ObjectName to be retrieved.
137         ObjectName objectName = customPreregister(server, name);
138         ObjectName toReturn = null;
139 
140         if (log.isDebugEnabled()) {
141             log.debug("Preregistering object of type [" + getClass().getName() + "] under name ["
142                     + objectName.getCanonicalName() + "]");
143         }
144 
145         try {
146 
147             // Perform normal preregistration.
148             toReturn = super.preRegister(server, objectName);
149 
150             // Preregistration successful; adjust the state.
151             performStateTransition(true, LifecycleState.UNINITIALIZED, LifecycleState.STARTING);
152 
153         } catch (Exception e) {
154 
155             // Complain
156             performStateTransition(false, LifecycleState.UNINITIALIZED, LifecycleState.ERROR);
157 
158             // Re-throw
159             throw e;
160         }
161 
162         if (log.isDebugEnabled()) {
163             log.debug("PreRegister complete for [" + getClass().getName() + "]");
164         }
165 
166         // All done
167         return toReturn;
168     }

170     /**
171      * Delegates to the {@code #customPostregister} method after invoking {@code #postRegister} method
172      * in the superclass.
173      * {@inheritDoc}
174      *
175      * @see #customPostregister()
176      */
177     @Override
178     public final void postRegister(final Boolean successfulRegistration) {
179 
180         // Delegate to superclass
181         super.postRegister(successfulRegistration);
182 
183         // Perform custom postRegistering?
184         if (successfulRegistration) {
185 
186             try {
187                 customPostregister();
188 
189                 // Indicate that we are started
190                 performStateTransition(true, LifecycleState.STARTING, LifecycleState.STARTED);
191 
192             } catch (Exception e) {
193 
194                 log.error("Custom postRegistration failed in [" + getClass().getName() + "]", e);
195                 performStateTransition(false, LifecycleState.STARTING, LifecycleState.ERROR);
196             }
197         } else {
198 
199             // Indicate that registration failed.
200             if (getState() != LifecycleState.ERROR) {
201                 performStateTransition(false, LifecycleState.STARTING, LifecycleState.ERROR);
202             }
203         }
204     }
205 
206     /**
207      * Delegates any custom execution to the {@code #customPreDeregister} method before calling the
208      * {@code #preDeregister} method in the superclass.
209      * {@inheritDoc}
210      *
211      * @see #customPreDeregister()
212      */
213     @Override
214     public final void preDeregister() throws Exception {
215 
216         // Adjust state
217         performStateTransition(true, LifecycleState.STARTED, LifecycleState.STOPPING);
218 
219         // Delegate to custom logic
220         if (getState() != LifecycleState.ERROR) {
221             try {
222                 customPreDeregister();
223             } catch (Exception e) {
224 
225                 log.error("Custom preDeRegistration failed in [" + getClass().getName() + "]", e);
226                 performStateTransition(false, LifecycleState.STARTED, LifecycleState.ERROR);
227             }
228         } else if (log.isDebugEnabled()) {
229             log.debug("" + LifecycleState.ERROR + " state, so customPreDeregister not called.");
230         }
231 
232         // Delegate to superclass
233         try {
234             super.preDeregister();
235         } catch (Exception e) {
236 
237             // Change the state
238             if (getState() != LifecycleState.ERROR) {
239                 performStateTransition(false, LifecycleState.STARTED, LifecycleState.ERROR);
240             }
241 
242             // Re-throw
243             throw e;
244         }
245     }
246 
247     /**
248      * {@inheritDoc}
249      */
250     @Override
251     public final void postDeregister() {
252 
253         // Delegate to the custom postDeregister method.
254         try {
255             customPostDeregister();
256         } catch (Exception e) {
257             log.error("customPostDeregister failed.", e);
258 
259             performStateTransition(false, LifecycleState.STOPPING, LifecycleState.ERROR);
260         }
261 
262         try {
263 
264             super.postDeregister();
265 
266             // Adjust state, if possible
267             final LifecycleState targetState = state == LifecycleState.ERROR
268                     ? LifecycleState.ERROR
269                     : LifecycleState.STOPPED;
270 
271             performStateTransition(false, LifecycleState.STOPPING, targetState);
272 
273         } catch (Exception e) {
274             log.error("super.postDeregister() failed.", e);
275 
276             performStateTransition(false, LifecycleState.STOPPING, LifecycleState.ERROR);
277         }
278     }

281      * Override this method to perform customization of the ObjectName under which this AbstractMBean should
282      * be registered within the supplied MBeanServer. <strong>Note!</strong> This method is invoked from within the
283      * {@code preRegister} method, implying that this method may not register this AbstractMBean in the MBean server.
284      *
285      * @param server     The MBeanServer instance.
286      * @param objectName The object name of the MBean, which is {@code null} if the name parameter to one of the
287      *                   <code>createMBean</code> or <code>registerMBean</code> methods in the {@link javax.management.MBeanServer}
288      *                   interface is {@code null}. In that case, this method must return a non-null ObjectName for
289      *                   the new MBean.
290      * @return A non-null ObjectName under which this AbstractMBean will be registered in the MBeanServer.
291      */
292     protected ObjectName customPreregister(final MBeanServer server, final ObjectName objectName) {
293         return objectName;
294     }

297      * Override this method to perform any customized actions after this AbstractMBean has been

302      * @see #postRegister(Boolean)
303      */
304     protected void customPostregister() {

319      * Override this mthod to perform any customized actions before this AbstractMBean will be de-registered

321      * is called. This method should not throw Exceptions.
322      */
323     protected void customPostDeregister() {
324         // By default, do nothing.

327     /**
328      * Helper method which creates a JMX AttributeChangeNotification instance from the supplied
329      * data, and notifies all registered NotificationListener instances by invoking {@code sendNotification()}
330      * with the newly constructed AttributeChangeNotification as an argument.
331      *
332      * @param attributeName The name of the attribute which changed, such as "state". Cannot be null or empty.
333      * @param eventMessage  A human-readable AttributeChangeNotification eventMessage,
334      *                      such as "LifecycleState changed". Cannot be null or empty.
335      * @param objectType    The attribute Class. Cannot be null.
336      * @param oldValue      The value before the change.
337      * @param newValue      The value after the change.
338      * @param <T>           The attribute Class.
339      */
340     protected final <T> void sendAttributeChangeEvent(final String attributeName,
341                                                       final String eventMessage,
342                                                       final Class<T> objectType,
343                                                       final T oldValue,
344                                                       final T newValue) {
345 
346         // Check sanity
347         Validate.notEmpty(attributeName, "Cannot handle null or empty attributeName argument.");
348         Validate.notEmpty(eventMessage, "Cannot handle null or empty eventMessage argument.");
349         Validate.notNull(objectType, "Cannot handle null or empty objectType argument.");
350 
351         // Notify our listeners about the state change.
352         final Notification notification = new AttributeChangeNotification(
353                 this,
354                 eventSequenceId++,
355                 System.currentTimeMillis(),
356                 eventMessage,
357                 attributeName,
358                 objectType.getName(),
359                 oldValue,
360                 newValue);
361 
362         if (log.isDebugEnabled()) {
363             log.debug("Sending AttributeChangeEvent [" + eventSequenceId + ", " +
364                     "newValue: " + newValue + "]: " + notification);
365         }
366 
367         sendNotification(notification);
368     }

370     /**
371      * Performs a state transition in this AbstractMBean which may be validated (in the sense that the
372      * from state must match the result from {@code getState()} to permit the state transition).
373      * If the stat transition was actually made, an AttributeChangeNotification is sent by this AbstractMBean to
374      * permit any state listeners to react to the event.
375      *
376      * @param validateFromState if {@code true}, the result of {@code getState()} must be equal to from to permit the
377      *                          state transition. Otherwise, the state transition will not be performed.
378      * @param from              The expected state of this AbstractMBean.
379      * @param to                The state to transition to.
380      * @return {@code true} if the transition was successfully made, and {@code false} otherwise.
381      */
382     protected final boolean performStateTransition(final boolean validateFromState,
383                                                    final LifecycleState from,
384                                                    final LifecycleState to) {
385 
386         if (validateFromState && getState() != from) {
387             if (log.isWarnEnabled()) {
388                 log.warn("Aborting illegal state transition [" + getState() + " --> " + to
389                         + "]. (Expected " + from + ") in AbstractMBean [" + getClass().getName() + "]");
390             }
391 
392             // State update not performed.
393             return false;
394         }
395 
396         synchronized (lock) {
397 
398             // Perform the state transition itself.
399             state = to;
400 
401             // Notify our listeners about the state change.
402             sendAttributeChangeEvent(STATE_CHANGE_ATTRIBUTENAME, "LifecycleState changed", LifecycleState.class, from, to);
403         }
404 
405         // All done.
406         return true;
407     }

409     /**
410      * @return An AttributeChangeNotificationFilter which will listen only to state change events emitted by
411      * an AbstractMBean subclass (e.g. AttributeChangeEvent with attributeName set to the value of constant
412      * {@code STATE_CHANGE_ATTRIBUTENAME}).
413      */
414     public static AttributeChangeNotificationFilter getStateTransitionFilter() {
415 
416         final AttributeChangeNotificationFilter toReturn = new AttributeChangeNotificationFilter();
417         toReturn.enableAttribute(AbstractMBean.STATE_CHANGE_ATTRIBUTENAME);
418 
419         // All done.
420         return toReturn;
421     }
