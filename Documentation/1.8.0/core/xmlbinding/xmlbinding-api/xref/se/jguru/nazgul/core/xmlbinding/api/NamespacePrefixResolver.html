22  package se.jguru.nazgul.core.xmlbinding.api;
23  
24  import java.util.Map;
25  import java.util.Set;
26  
27  /**
28   * <p>Specification for how to map XML Namespace URIs to XML Namespace Prefixes.
29   * XML namespaces are provided in the form of an URI-formatted String, such as
30   * {@code http://www.jguru.se/nazgul/core}, and XML prefixes are simply identifiers
31   * for these URIs (such as {@code core}).</p>
32   * <p>The intention for this NamespacePrefixResolver is to inform the XML binder instance
33   * of the relation between the namespace URI and prefix - illustrated below by the mapping
34   * of the {@code http://foo/bar} namespace to the elements ParentElement and ChildElement.
35   * Some XmlBinder implementations [notably JAXB] have rather sketchy implementations of
36   * these mechanics, so we provide this NamespacePrefixResolver specification to provide
37   * a unified specification for relating XML namespace URLs to Prefixes.</p>
38   * <p>Example (desired output):</p>
39   * <pre>
40   *     &lt;?xml version="1.0" encoding="UTF-8" standalone="yes"?&gt;
41   *     &lt;foobar:ParentElement xmlns:xs="http://www.w3.org/2001/XMLSchema"
42   *         <strong>xmlns:foobar="http://foo/bar"</strong>
43   *         xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"&gt;
44   *
45   *         &lt;foobar:ChildElement ... /&gt;
46   *
47   *         ...
48   *     &lt;/foobar:ParentElement&gt;
49   * </pre>
50   * <p>The corresponding code to relate the {@code http://foo/bar} URI to the {@code foobar}
51   * prefix is shown below:</p>
52   * <pre>
53   *     // Acquire the NamespacePrefixResolver in use by the XmlBinder
54   *     XmlBinder binder = ...
55   *     NamespacePrefixResolver resolver = binder.getNamespacePrefixResolver();
56   *
57   *     // Relate the URI to the desired prefix.
58   *     resolver.put("http://foo/bar", "foobar");
59   * </pre>
60   *
61   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
62   */
63  public interface NamespacePrefixResolver {
64  
65      /**
66       * Empty/undeterminate namespace URI.
67       */
68      String EMPTY_NAMESPACE = "";
69  
70      /**
71       * Adds the provided mapping between a single xmlNamespaceUri and corresponding xmlPrefix.
72       *
73       * @param xmlNamespaceUri the unique URI of an XML namespace.
74       * @param xmlPrefix       the unique prefix of the provided XML namespace.
75       * @throws NullPointerException     if any argument was {@code null}.
76       * @throws IllegalArgumentException if the xmlNamespaceUri was already registered to another prefix, or
77       *                                  if any argument was empty.
78       */
79      void put(String xmlNamespaceUri, String xmlPrefix) throws NullPointerException, IllegalArgumentException;
80  
81      /**
82       * Convenience method, adding all provided mappings between xmlNamespaceUris and corresponding xmlPrefixes.
83       *
84       * @param xmlUri2PrefixMap A non-null map relating XML namespace URIs to corresponding prefixes.
85       * @throws NullPointerException     if any argument or element was {@code null}.
86       * @throws IllegalArgumentException if any xmlNamespaceUri was already registered to another prefix, or
87       *                                  if any argument was empty.
88       */
89      void putAll(Map<String, String> xmlUri2PrefixMap) throws NullPointerException, IllegalArgumentException;
90  
91      /**
92       * Retrieves the XML namespace URI for the provided xmlPrefix, or {@code null} if none was found.
93       *
94       * @param xmlPrefix The XML prefix for which to obtain the corresponding XML namespace.
95       * @return the XML namespace URI for the provided xmlPrefix, or {@code null} if none was found.
96       * @throws NullPointerException if the xmlPrefix argument was {@code null}.
97       */
98      String getNamespaceUri(String xmlPrefix) throws NullPointerException;
99  
100     /**
101      * Retrieves the XML prefix for the provided xmlNamespaceUri, or {@code null} if none was found.
102      *
103      * @param xmlNamespaceUri The XML namespace URI for which to obtain the corresponding XML prefix.
104      * @return the XML prefix for the provided xmlNamespaceUri, or {@code null} if none was found.
105      * @throws NullPointerException if the xmlPrefix argument was {@code null}.
106      */
107     String getXmlPrefix(String xmlNamespaceUri) throws NullPointerException;
108 
109     /**
110      * @return A non-modifiable List holding all currently registered XML namespaceURIs.
111      */
112     Set<String> getRegisteredNamespaceURIs();
113 
114     /**
115      * @return A non-modifiable List holding all currently registered XML prefixes.
116      */
117     Set<String> getRegisteredPrefixes();
118 }
