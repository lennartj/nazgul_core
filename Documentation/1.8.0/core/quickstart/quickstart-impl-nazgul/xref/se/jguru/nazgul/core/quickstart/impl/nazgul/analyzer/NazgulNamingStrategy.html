22  package se.jguru.nazgul.core.quickstart.impl.nazgul.analyzer;
23  
24  import org.apache.commons.lang3.Validate;
25  import se.jguru.nazgul.core.quickstart.api.PomType;
26  import se.jguru.nazgul.core.quickstart.api.analyzer.AbstractNamingStrategy;
27  import se.jguru.nazgul.core.quickstart.model.Name;
28  
29  import java.util.Arrays;
30  import java.util.Map;
31  import java.util.TreeMap;
32  import java.util.regex.Pattern;
33  
34  /**
35   * Nazgul-style NamingStrategy implementation, based on the AbstractNamingStrategy.
36   *
37   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
38   */
39  public class NazgulNamingStrategy extends AbstractNamingStrategy {
40  
41      /**
42       * Pattern identifying a single word with ASCII-compatible, lowercase letters and optional digits.
43       */
44      public static final Pattern WORD_PATTERN = Pattern.compile("[a-z][a-z0-9]*", Pattern.CANON_EQ);
45  
46      /**
47       * Pattern identifying several words with ASCII-compatible, lowercase letters and optional digits,
48       * separated by '-' chars.
49       */
50      public static final Pattern WORDS_PATTERN = Pattern.compile("[a-z][a-z0-9]*([.-][a-z0-9]+)*", Pattern.CANON_EQ);
51  
52      /**
53       * Nazgul prefix.
54       */
55      public static final String NAZGUL_PREFIX = "nazgul";
56  
57      // Internal state
58      private String requiredPrefix;
59      private Map<PomType, Pattern> validPatternMap;
60  
61      /**
62       * Default constructor creating a NazgulNamingStrategy which requires that no prefixes should be placed
63       * on folders within the project reactor, and that all artifactIDs should use the prefix {@code NAZGUL_PREFIX}.
64       *
65       * @see se.jguru.nazgul.core.quickstart.model.Name
66       * @see se.jguru.nazgul.core.quickstart.model.Name#parse(String)
67       */
68      public NazgulNamingStrategy() {
69          this(false, NAZGUL_PREFIX);
70      }
71  
72      /**
73       * Compound constructor creating a NazgulNamingStrategy wrapping the supplied data.
74       *
75       * @param prefixIsRequiredOnAllFolders if {@code true}, all folders within the project reactor
76       *                                     are required to sport prefixes.
77       * @param requiredArtifactIdPrefix     A non-empty string which must be used as the prefix of all valid Names.
78       * @see se.jguru.nazgul.core.quickstart.model.Name
79       * @see se.jguru.nazgul.core.quickstart.model.Name#parse(String)
80       */
81      public NazgulNamingStrategy(final boolean prefixIsRequiredOnAllFolders,
82                                  final String requiredArtifactIdPrefix) {
83          super(prefixIsRequiredOnAllFolders);
84  
85          // Assign internal state
86          Validate.notEmpty(requiredArtifactIdPrefix, "Cannot handle null or empty requiredPrefix argument.");
87          this.requiredPrefix = requiredArtifactIdPrefix;
88  
89          validPatternMap = new TreeMap<>();
90          validPatternMap.put(PomType.ROOT_REACTOR, Pattern.compile(AbstractNamingStrategy.REACTOR_SUFFIX));
91          validPatternMap.put(PomType.REACTOR, Pattern.compile(
92                  WORDS_PATTERN + "-" + AbstractNamingStrategy.REACTOR_SUFFIX));
93          validPatternMap.put(PomType.REACTOR, Pattern.compile(
94                  WORDS_PATTERN + "-" + AbstractNamingStrategy.REACTOR_SUFFIX));
95  
96          for (PomType current : Arrays.asList(PomType.PARENT, PomType.API_PARENT,
97                  PomType.MODEL_PARENT, PomType.WAR_PARENT)) {
98              validPatternMap.put(current, Pattern.compile(current.name().toLowerCase().replace('_', '-')));
99          }
100         validPatternMap.put(PomType.OTHER_PARENT, Pattern.compile(WORDS_PATTERN
101                 + "-" + AbstractNamingStrategy.PARENT_SUFFIX));
102     }
103 
104     /**
105      * {@inheritDoc}
106      */
107     @Override
108     public void validate(final Name toValidate, final PomType pomType) throws IllegalArgumentException {
109 
110         // Check sanity
111         Validate.notNull(toValidate, "Cannot handle null Name argument.");
112         Validate.notNull(pomType, "Cannot handle null PomType argument.");
113 
114         // Do we have a required prefix and separator?
115         if (requiredPrefix != null) {
116             if (toValidate.getPrefix() == null) {
117                 throw new IllegalArgumentException("The [" + getClass().getName() + "] NamingStrategy requires the "
118                         + "prefix [" + requiredPrefix + "]");
119             }
120             Validate.matchesPattern(toValidate.getPrefix(), requiredPrefix);
121         }
122         Validate.matchesPattern(toValidate.getSeparator(), "-");
123 
124         // Check the type for compliance.
125         final Pattern pattern = validPatternMap.get(pomType);
126         if (!pattern.matcher(toValidate.getType()).matches()) {
127             throw new IllegalArgumentException("Invalid Name [" + toValidate + "] for PomType [" + pomType + "].");
128         }
129     }
130 }
