22  package se.jguru.nazgul.core.persistence.model;
23  
24  import org.apache.commons.lang3.builder.CompareToBuilder;
25  import org.apache.commons.lang3.builder.EqualsBuilder;
26  import org.apache.commons.lang3.builder.HashCodeBuilder;
27  
28  /**
29   * Trivial utility class for Entities.
30   *
31   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
32   */
33  public final class Entities {
34  
35      // Hide the constructor in utility classes.
36      private Entities() {
37      }
38  
39      /**
40       * Standard fields to disregard in calculations.
41       */
42      public static final String[] STANDARD_DISREGARD_FIELDS = {"version", "id",
43              "pcVersionInit", "pcStateManager", "pcDetachedState"};
44  
45      /**
46       * Standard equality calculations between two objects of a given type.
47       *
48       * @param rhs       The right-hand side comparison.
49       * @param lhs       The left-hand side comparison.
50       * @param stopClass The topmost class to compare.
51       * @param <T>       The type of the comparands.
52       * @return {@code true} if the two comparisons were equal.
53       */
54      public static <T> boolean equals(final T lhs, final T rhs, final Class<? super T> stopClass) {
55          return EqualsBuilder.reflectionEquals(rhs, lhs, false, stopClass, STANDARD_DISREGARD_FIELDS);
56      }
57  
58      /**
59       * Builds a hashCode from the supplied data.
60       *
61       * @param anObject The object for which a HashCode should be constructed.
62       * @param stopType The type up to which we should reflect.
63       * @param <T>      The type of object to acquire a hashcode for.
64       * @return The resulting hash for the supplied instance.
65       */
66      public static <T> int hashCode(final T anObject, final Class<? super T> stopType) {
67          return HashCodeBuilder.reflectionHashCode(17, 37, anObject, false, stopType, STANDARD_DISREGARD_FIELDS);
68      }
69  
70      /**
71       * Performs reflective comparison between the lhs and rhs objects, calculating non-transient properties up to the
72       * supplied stopType.
73       *
74       * @param rhs      The right-hand side comparison object.
75       * @param lhs      The left-hand side comparison object.
76       * @param stopType The type up to which we should reflect.
77       * @param <T>      The type of object to acquire a comparison for.
78       * @return The resulting comparison value.
79       */
80      public static <T> int compare(final T lhs, final T rhs, final Class<? super T> stopType) {
81          return CompareToBuilder.reflectionCompare(lhs, rhs, false, stopType, STANDARD_DISREGARD_FIELDS);
82      }
83  }
