45   * Default implementation of the ConverterRegistry specification.
46   * Not intended for clustered operation.

50  @SuppressWarnings({"rawtypes", "unchecked"})
51  public class DefaultConverterRegistry implements ConverterRegistry {
52  
53      // Our Log
54      private static final Logger log = LoggerFactory.getLogger(DefaultConverterRegistry.class);

56      // Internal state
57      private Map<Class<?>, PrioritizedTypeConverter> sourceTypeToTypeConvertersMap;
58  
59      /**
60       * Default constructor, yielding an empty internal state - i.e. no default
61       * converters added.
62       */
63      public DefaultConverterRegistry() {
64  
65          // Create internal state
66          sourceTypeToTypeConvertersMap = new HashMap<Class<?>, PrioritizedTypeConverter>();
67      }

69      /**
70       * {@inheritDoc}
71       */
72      @Override
73      public void add(final Object... converters) throws IllegalArgumentException {
74  
75          // Check sanity
76          Validate.notEmpty(converters, "Cannot handle null or empty converters argument.");
77          final Map<Object, Tuple<List<Method>, List<Constructor<?>>>> validConverters =
78                  new HashMap<Object, Tuple<List<Method>, List<Constructor<?>>>>();
79  
80          // All converters have annotations?
81          for (Object current : converters) {
82  
83              final Tuple<List<Method>, List<Constructor<?>>> methodsAndConstructors =
84                      ReflectiveConverterFilter.getConverterMethodsAndConstructors(current);
85  
86              if (methodsAndConstructors == null) {
87  
88                  // No converters methods or constructors found within the supplied converter. Complain.
89                  throw new IllegalArgumentException("Found no @Converter-annotated methods within class ["
90                          + current.getClass().getName() + "]. No @Converters added.");
91              }
92  
93              // Map the current converter to its converter methods and constructors, respectively.
94              validConverters.put(current, methodsAndConstructors);
95          }
96  
97          // Map the respective converters to the sourceTypes of their @Converters.
98          final Map<Class<?>, Set<Object>> sourceTypeToConverterInstanceMap = new HashMap<Class<?>, Set<Object>>();
99          for (Map.Entry<Object, Tuple<List<Method>, List<Constructor<?>>>> currentEntry : validConverters.entrySet()) {
100 
101             final Tuple<List<Method>, List<Constructor<?>>> methodsAndConstructors = currentEntry.getValue();
102 
103             // Map the source types of all Methods & Constructors
104             for (Method currentMethod : methodsAndConstructors.getKey()) {
105 
106                 final Class<?> currentType = currentMethod.getParameterTypes()[0];
107                 addCurrentConverter(sourceTypeToConverterInstanceMap, currentEntry.getKey(), currentType);
108             }
109 
110             for (Constructor<?> currentConstructor : methodsAndConstructors.getValue()) {
111 
112                 final Class<?> currentType = currentConstructor.getParameterTypes()[0];
113                 addCurrentConverter(sourceTypeToConverterInstanceMap, currentEntry.getKey(), currentType);
114             }
115         }
116 
117         // Create PrioritizedTypeConverters for all source types.
118         for (Map.Entry<Class<?>, Set<Object>> current : sourceTypeToConverterInstanceMap.entrySet()) {
119 
120             PrioritizedTypeConverter prioritizedTypeConverter = new PrioritizedTypeConverter(
121                     current.getKey(), current.getValue().toArray());
122 
123             // Finally, add the PrioritizedTypeConverter instance.
124             sourceTypeToTypeConvertersMap.put(current.getKey(), prioritizedTypeConverter);
125         }
126     }

138     /**
139      * {@inheritDoc}
140      */
141     @Override
142     public <From, To> To convert(final From source, final Class<To> desiredType)
143             throws IllegalArgumentException {
144 
145         // Find the best PrioritizedTypeConverter instance for the supplied source.
146         final Class<From> sourceType = source == null ? null : (Class<From>) source.getClass();
147         PrioritizedTypeConverter<From> typeConverter = sourceTypeToTypeConvertersMap.get(sourceType);
148 
149         if (typeConverter == null) {
150 
151             // Resort to fuzzy logic to acquire the PrioritizedTypeConverter
152             final SortedMap<Integer, PrioritizedTypeConverter> prioritized = getPrioritizedConverters(sourceType);
153 
154             // Get the least divergent type converter
155             typeConverter = prioritized.get(prioritized.lastKey());
156         }
157 
158         // No TypeConverter found at all?
159         if (typeConverter == null) {
160 
161             // Holler a tad...
162             log.info("TypeConverter not found. [" + desiredType.getSimpleName() + " --> " + typeConverter + "]");
163             return null;
164         }
165 
166         // Dig out the corresponding optimalToType for the acquired TypeConverter.
167         // We may acquire a null response here, if no optimalToType can be found.
168         final Class<To> optimalToType = getOptimalToType(typeConverter, desiredType);
169 
170         To toReturn = null;
171         if (optimalToType != null) {
172 
173             log.debug("Found TypeConverter [" + desiredType.getSimpleName()
174                     + (optimalToType == desiredType ? "" : " (fuzzy: " + optimalToType.getSimpleName() + ")")
175                     + " --> " + typeConverter + "]");
176 
177             // Convert the data
178             toReturn = (To) typeConverter.convert(source, optimalToType);
179         }
180 
181         // All done.
182         return toReturn;
183     }

185     /**
186      * {@inheritDoc}
187      */
188     @Override
189     public <From> Set<Class<?>> getPossibleConversions(final Class<From> sourceType) throws IllegalArgumentException {
190 
191         // Check sanity
192         Validate.notNull(sourceType, "Cannot handle null sourceType argument.");
193 
194         // Acquire the TypeConverter instance
195         final SortedMap<Integer, PrioritizedTypeConverter> converterMap = getPrioritizedConverters(sourceType);
196 
197         if (converterMap.size() == 0) {
198             return new HashSet<Class<?>>();
199         }
200 
201         // Delegate and return.
202         final Set<Class<?>> toReturn = new HashSet<Class<?>>();
203         for (Map.Entry<Integer, PrioritizedTypeConverter> currentEntry : converterMap.entrySet()) {
204 
205             // Acquire the available target types for the current PrioritizedTypeConverter
206             final Set<Class<?>> availableTargetTypes = currentEntry.getValue().getAvailableTargetTypes();
207             for (Class<?> currentTargetType : availableTargetTypes) {
208 
209                 // Don't add twice.
210                 if (!toReturn.contains(currentTargetType)) {
211                     toReturn.add(currentTargetType);
212                 }
213             }
214         }
215 
216         // All done.
217         return toReturn;
218     }

220     /**
221      * {@inheritDoc}
222      */
223     @Override
224     public String toString() {
225 
226         // Create a preamble
227         final StringBuilder builder = new StringBuilder("DefaultConverterRegistry ["
228                 + sourceTypeToTypeConvertersMap.size() + "] source types.\n");
229 
230         final SortedMap<String, Class<?>> sortedClassNames = new TreeMap<String, Class<?>>();
231         for (Class<?> current : sourceTypeToTypeConvertersMap.keySet()) {
232             sortedClassNames.put(current.getSimpleName(), current);
233         }
234 
235         // Delegate detail printout to the PrioritizedTypeConverter.
236         for (Map.Entry<String, Class<?>> current : sortedClassNames.entrySet()) {
237             builder.append(sourceTypeToTypeConvertersMap.get(sortedClassNames.get(current.getKey()))).append("\n");
238         }
239 
240         // All done.
241         return builder.toString();
242     }

248     private <From, To> Class<To> getOptimalToType(final PrioritizedTypeConverter<From> typeConverter,
249                                                   final Class<To> requestedToType) {
250 
251         // Get the available To/target types for the supplied typeConverter.
252         final Set<Class<?>> availableTargetTypes = typeConverter.getAvailableTargetTypes();
253 
254         // First - is the requestedToType available?
255         if (availableTargetTypes.contains(requestedToType)) {
256 
257             // Debug somewhat.
258             log.debug("Exact conversion [" + typeConverter.getSourceType().getSimpleName() + " --> "
259                     + requestedToType.getSimpleName() + "] found in " + typeConverter);
260 
261             return requestedToType;
262         }
263 
264         // Get the best available To/target type, related to the supplied requestedToType.
265         final SortedMap<Integer, List<Class<?>>> prioritizedTypes = createPrioritizedClassMap(
266                 requestedToType, availableTargetTypes);
267 
268         // Unrelated?
269         if (prioritizedTypes.size() == 0) {
270             return null;
271         }
272 
273         // Several optimal types available?
274         final List<Class<?>> bestChoiceToTypes = prioritizedTypes.get(prioritizedTypes.firstKey());
275         final Class<To> optimalToClass = (Class<To>) getOptimalClosestType(bestChoiceToTypes);
276 
277         // Debug somewhat.
278         log.debug("Fuzzy conversion [" + typeConverter.getSourceType().getSimpleName() + " --> "
279                 + optimalToClass.getSimpleName() + " (" + requestedToType.getSimpleName() + ")] found in "
280                 + typeConverter);
281 
282         // All done.
283         return optimalToClass;
284     }

286     /**
287      * Acquires a SortedMap holding all available PrioritizedTypeConverter instances
288      * which can convert from the supplied sourceType.
289      *
290      * @param sourceType The type to convert from.
291      * @param <From>     The type to convert from.
292      * @return a SortedMap holding all available PrioritizedTypeConverter instances
293      * which can convert from the supplied sourceType, mapped to their priority
294      * as defined by the algorithm found in the TypeClosenessTransformer class.
295      */
296     private <From> SortedMap<Integer, PrioritizedTypeConverter> getPrioritizedConverters(
297             final Class<From> sourceType) {
298 
299         // Create the List of converters to return
300         final SortedMap<Integer, PrioritizedTypeConverter> toReturn =
301                 new TreeMap<Integer, PrioritizedTypeConverter>();
302 
303         // Exact match?
304         PrioritizedTypeConverter<From> typeConverter = sourceTypeToTypeConvertersMap.get(sourceType);
305         if (typeConverter != null) {
306 
307             // Exact match found for the type.
308             // Ignore fuzzy logic type resolution.
309             toReturn.put(0, typeConverter);
310 
311         } else {
312 
313             // No exact match found for the type desired.
314             // Perform compound fuzzy logic matching to acquire all matching converters.
315 
316             // Filter out all TypeConverters which can convert the supplied sourceType.
317             final Map<Class<?>, PrioritizedTypeConverter> candidates = CollectionAlgorithms.filter(
318                     sourceTypeToTypeConvertersMap,
319                     new Filter<Tuple<Class<?>, PrioritizedTypeConverter>>() {
320                         @Override
321                         public boolean accept(final Tuple<Class<?>, PrioritizedTypeConverter> candidate) {
322 
323                             // The sourceType can be null, which generates an exception in the isAssignableFrom method.
324                             // Perform normal sanity checking to handle null sourceType values.
325                             if (sourceType != null) {
326 
327                                 // Only accept candidates able to convert from the supplied sourceType/fromType.
328                                 return candidate.getKey().isAssignableFrom(sourceType);
329 
330                             } else {
331 
332                                 // sourceType is null here, so the candidate should only 'accept'
333                                 // here if the internal TypeConverter can handle nulls.
334                                 final PrioritizedTypeConverter converter = candidate.getValue();
335                                 return converter.getSourceType() == Object.class;
336                             }
337                         }
338                     });
339 
340             // Now, prioritize the found sourceTypes
341             final SortedMap<Integer, List<Class<?>>> prioritizedClassMap = createPrioritizedClassMap(
342                     sourceType, candidates.keySet());
343 
344             // Populate the return map
345             for (Map.Entry<Integer, List<Class<?>>> current : prioritizedClassMap.entrySet()) {
346 
347                 // For each priority, simply pick the first type returned.
348                 toReturn.put(current.getKey(), candidates.get(current.getValue().get(0)));
349             }
350         }
351 
352         // Debug somewhat
353         final String sourceTypeSimpleName = sourceType == null ? "<null>" : sourceType.getSimpleName();
354         log.debug("Source type [" + sourceTypeSimpleName + "] yields converters: " + toReturn);
355 
356         // All done.
357         return toReturn;
358     }

360     private void addCurrentConverter(final Map<Class<?>, Set<Object>> sourceTypeToConverterInstanceMap,
361                                      final Object value,
362                                      final Class<?> key) {
363 
364         // Get or create the current converter set
365         Set<Object> currentConverterSet = sourceTypeToConverterInstanceMap.get(key);
366         if (currentConverterSet == null) {
367             currentConverterSet = new HashSet<Object>();
368             sourceTypeToConverterInstanceMap.put(key, currentConverterSet);
369         }
370 
371         // Add if the current value is not present within the currentConverterSet.
372         if (!currentConverterSet.contains(value)) {
373             currentConverterSet.add(value);
374         }
375     }

377     private Class<?> getOptimalClosestType(final List<Class<?>> initialBestChoices) {
378 
379         // Start with the first available best choice type.
380         Class<?> optimalToClass = initialBestChoices.get(0);
381 
382         // Several choices available?
383         if (initialBestChoices.size() > 1) {
384 
385             // If there are class types, use the first one of them.
386             // Otherwise use the first bestChoiceToType found.
387             final List<Class<?>> optimalClasses = CollectionAlgorithms.filter(
388                     initialBestChoices,
389                     new Filter<Class<?>>() {
390                         @Override
391                         public boolean accept(final Class<?> candidate) {
392                             return !candidate.isInterface();
393                         }
394                     });
395 
396             optimalToClass = optimalClasses.size() > 0 ? optimalClasses.get(0) : initialBestChoices.get(0);
397 
398             // Debug somewhat
399             List<String> initialBestChoicesClassNames = new ArrayList<String>();
400             for (Class<?> current : initialBestChoices) {
401                 initialBestChoicesClassNames.add(current.getSimpleName());
402             }
403 
404             log.debug("Optimal type [" + optimalToClass.getSimpleName()
405                     + "] selected from choices " + initialBestChoicesClassNames);
406 
407         } else {
408             log.debug("Single/optimal best choice type: " + initialBestChoices);
409         }
410 
411         // All done.
412         return optimalToClass;
413     }

415     private <To> SortedMap<Integer, List<Class<?>>> createPrioritizedClassMap(
416             final Class<To> requestedToType, final Set<Class<?>> availableTargetTypes) {
417 
418         SortedMap<Integer, List<Class<?>>> prioritizedTypes = new TreeMap<Integer, List<Class<?>>>();
419         for (Class<?> current : availableTargetTypes) {
420 
421             // Find the current priority
422             try {
423                 // The requestedToType can be null; handle this situation.
424                 final int priority = requestedToType == null
425                         ? -1
426                         : TypeExtractor.getRelationDifference(current, requestedToType);
427 
428                 List<Class<?>> currentPriorityTypes = prioritizedTypes.get(priority);
429                 if (currentPriorityTypes == null) {
430                     currentPriorityTypes = new ArrayList<Class<?>>();
431                     prioritizedTypes.put(priority, currentPriorityTypes);
432                 }
433 
434                 // ... and add the current type.
435                 currentPriorityTypes.add(current);
436 
437             } catch (IllegalArgumentException e) {
438                 // Ignore this type.
439             }
440         }
441         return prioritizedTypes;
442     }
