96      @Test
97      public void validateCreatingProjectSkeleton() {
98  
99          // Assemble
100         final File rootDir = FileUtilsTest.createUniqueDirectoryUnderTestData("createProjectData");
101         final TestProjectFactory unitUnderTest = new TestProjectFactory(namingStrategy);
102 
103         final SimpleArtifact reactorParent = new SimpleArtifact(
104                 "se.jguru.nazgul.tools.poms.external",
105                 "nazgul-tools-external-reactor-parent",
106                 "4.0.0");
107 
108         final SimpleArtifact parentParent = new SimpleArtifact(
109                 "se.jguru.nazgul.core.poms.core-parent",
110                 "nazgul-core-parent",
111                 "1.6.1");
112 
113         final Project projectDefinition = unitUnderTest.createProjectDefinition(
114                 "acme", "blah", reactorParent, parentParent);
115 
116         // Act
117         final boolean created = unitUnderTest.createProject(
118                 rootDir,
119                 projectDefinition,
120                 "se.jguru",
121                 "1.0.0-SNAPSHOT",
122                 "1.0.0-SNAPSHOT");
123 
124         // Assert
125         Assert.assertTrue(created);
126 
127         final File reactorRoot = new File(rootDir, projectDefinition.getName());
128         validateDirectory(reactorRoot, PomType.ROOT_REACTOR);
129         validateDirectory(new File(reactorRoot, "poms"), PomType.REACTOR);
130         validateDirectory(new File(reactorRoot, "poms/blah-parent"), PomType.PARENT);
131         validateDirectory(new File(reactorRoot, "poms/blah-api-parent"), PomType.API_PARENT);
132         validateDirectory(new File(reactorRoot, "poms/blah-model-parent"), PomType.MODEL_PARENT);
133         validateDirectory(new File(reactorRoot, "poms/blah-war-parent"), PomType.WAR_PARENT);
134 
135         final List<String> callTrace = unitUnderTest.callTrace;
136         final List<PomType> pomTypeOrderedList = Arrays.asList(PomType.ROOT_REACTOR, PomType.PARENT,
137                 PomType.API_PARENT, PomType.MODEL_PARENT, PomType.WAR_PARENT, PomType.REACTOR);
138 
139         for (int i = 0; i < pomTypeOrderedList.size(); i++) {
140 
141             int index = i * 3 + 2;
142             final String pomTypeName = pomTypeOrderedList.get(i).name();
143             Assert.assertEquals(callTrace.get(index), "getSkeletonPomDirectoryName(blah, " + pomTypeName + ", acme)");
144             Assert.assertEquals(callTrace.get(index + 1), "getTemplateResource(" + pomTypeName + "/pom.xml)");
145             Assert.assertEquals(callTrace.get(index + 2), "getTemplateResourceURL(" + pomTypeName + "/pom.xml)");
146         }
147 
148         // createProjectDefinition(acme,
149         //      blah,
150         //      se.jguru.nazgul.tools.poms.external/nazgul-tools-external-reactor-parent/4.0.0,
151         //      se.jguru.nazgul.core.poms.core-parent/nazgul-core-parent/1.6.1),
152         //
153         // createProject,
154         // getSkeletonPomDirectoryName(blah, ROOT_REACTOR, acme),
155         // getTemplateResource(ROOT_REACTOR/pom.xml),
156         // getTemplateResourceURL(ROOT_REACTOR/pom.xml),
157         // getSkeletonPomDirectoryName(blah, PARENT, acme),
158         // getTemplateResource(PARENT/pom.xml),
159         // getTemplateResourceURL(PARENT/pom.xml),
160         // getSkeletonPomDirectoryName(blah, API_PARENT, acme),
161         // getTemplateResource(API_PARENT/pom.xml),
162         // getTemplateResourceURL(API_PARENT/pom.xml),
163         // getSkeletonPomDirectoryName(blah, MODEL_PARENT, acme),
164         // getTemplateResource(MODEL_PARENT/pom.xml),
165         // getTemplateResourceURL(MODEL_PARENT/pom.xml),
166         // getSkeletonPomDirectoryName(blah, WAR_PARENT, acme),
167         // getTemplateResource(WAR_PARENT/pom.xml),
168         // getTemplateResourceURL(WAR_PARENT/pom.xml),
169         // getSkeletonPomDirectoryName(blah, REACTOR, acme),
170         // getTemplateResource(REACTOR/pom.xml),
171         // getTemplateResourceURL(REACTOR/pom.xml)

174     @Test
175     public void validateGettingDirectoryName() {
176 
177         // Assemble
178         final NamingStrategy noNameRequiredOnDirectories = new TestNamingStrategy(false);
179         final NamingStrategy nameRequiredOnDirectories = new TestNamingStrategy(true);
180 
181         final TestProjectFactory nameFactory = new TestProjectFactory(nameRequiredOnDirectories);
182         final TestProjectFactory noNameFactory = new TestProjectFactory(noNameRequiredOnDirectories);
183 
184         // Act
185         final String gnatFooComponentApi = nameFactory.getSkeletonPomDirectoryName("foo", PomType.COMPONENT_API, "gnat");
186         final String gnatFooParentApi = nameFactory.getSkeletonPomDirectoryName("foo", PomType.API_PARENT, "gnat");
187         final String fooComponentApi = noNameFactory.getSkeletonPomDirectoryName("foo", PomType.COMPONENT_API, "gnat");
188         final String fooParentApi = noNameFactory.getSkeletonPomDirectoryName("foo", PomType.API_PARENT, "gnat");
189 
190         // Assert
191         Assert.assertEquals("gnat-foo-component-api", gnatFooComponentApi);
192         Assert.assertEquals("gnat-foo-api-parent", gnatFooParentApi);
193         Assert.assertEquals("foo-component-api", fooComponentApi);
194         Assert.assertEquals("foo-api-parent", fooParentApi);
195     }

201     private void validateDirectory(final File dir, final PomType pomType) {
202         Assert.assertTrue("Required directory [" + dir.getAbsolutePath() + "] was not found.",
203                 dir.exists() && dir.isDirectory());
204 
205         final File pomFile = new File(dir, "pom.xml");
206         Assert.assertTrue("Required POM file [" + pomFile.getAbsolutePath() + "] was not found.",
207                 pomFile.exists() && pomFile.isFile());
208 
209         // The TestPomFactory should have generated POM file data holding the PomType in text.
210         // Validate that the correct pom is written to the correct location.
211         final Model pomModel = FileUtils.getPomModel(pomFile);
212         Assert.assertEquals(pomType, namingStrategy.getPomType(pomModel));
213     }
