22  package se.jguru.nazgul.core.quickstart.api.generator;
23  
24  import se.jguru.nazgul.core.quickstart.model.Project;
25  import se.jguru.nazgul.core.quickstart.model.SimpleArtifact;
26  
27  import java.io.File;
28  
29  /**
30   * Specification for how to generate a new project structure in an empty or non-existent directory.
31   * A project should contain root resources such as required POMs and directories where maven projects
32   * or other components should be located.
33   *
34   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
35   */
36  public interface ProjectFactory {
37  
38      /**
39       * Creates a Project (definition) by harvesting names from the supplied name,
40       * prefix and parent Artifact definitions.
41       *
42       * @param prefix        The project prefix. Optional, but recommended.
43       * @param name          The project name.
44       * @param reactorParent The parent of the root reactor POM.
45       * @param parentParent  The parent of the topmost parent POM.
46       * @return A Project instance for use with the {@code createProject} method.
47       */
48      Project createProjectDefinition(String prefix,
49                                      String name,
50                                      SimpleArtifact reactorParent,
51                                      SimpleArtifact parentParent);
52  
53      /**
54       * Creates a new Project using the information found within the supplied projectDefinition.
55       *
56       * @param projectParentDir          a directory in which the project should be created.
57       * @param projectDefinition         The entity defining the Project's overall data, and holding naming standards.
58       * @param packagePrefix             A non-null (but optionally empty) package prefix prepended to the groupId of the
59       *                                  reactor parent POM. Typically the reverse DNS of the organisation creating the
60       *                                  project, such as {@code se.jguru}. (For example, the groupId of the topmost
61       *                                  reactor POM created within the projectParentDir is
62       *                                  {@code [packagePrefix].[name]}).
63       * @param reactorParentMavenVersion The non-empty Maven project version of reactor POMs within the project.
64       *                                  Should be identical to the version of the reactor parent POM.
65       * @param topmostParentMavenVersion The non-empty Maven project version of parent POMs within the project (and,
66       *                                  normally, the version of all leaf projects within the Maven project reactor).
67       *                                  Should be identical to the version of the parent parent POM.
68       * @return {@code true} if the project could be properly created, and false otherwise.
69       * @throws java.lang.IllegalArgumentException if projectParentDir was not an existing directory.
70       * @throws java.lang.IllegalStateException    if the projectRootDirectory (typically either on the form
71       *                                            {@code prefix-name} or {@code name}) could not be created
72       *                                            under the projectParentDir.
73       */
74      boolean createProject(File projectParentDir,
75                            Project projectDefinition,
76                            String packagePrefix,
77                            String reactorParentMavenVersion,
78                            String topmostParentMavenVersion)
79              throws IllegalArgumentException, IllegalStateException;
80  }
