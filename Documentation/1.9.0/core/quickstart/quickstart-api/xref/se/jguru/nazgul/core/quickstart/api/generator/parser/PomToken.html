22  package se.jguru.nazgul.core.quickstart.api.generator.parser;
23  
24  import org.apache.commons.lang3.Validate;
25  
26  import java.util.SortedMap;
27  import java.util.TreeMap;
28  
29  /**
30   * Enumeration of a set of standardized tokens normally usable within a Maven POM.
31   * All supplied tokens relate to typical properties used within POMs.
32   *
33   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
34   */
35  public enum PomToken {
36  
37      /**
38       * The groupId for the active project.
39       */
40      GROUPID("groupId"),
41  
42      /**
43       * The artifactId for the active project.
44       */
45      ARTIFACTID("artifactId"),
46  
47      /**
48       * The maven version for the active project.
49       */
50      VERSION("mavenVersion"),
51  
52      /**
53       * The relative path from the VCS project root to the active maven project.
54       */
55      RELATIVE_DIRPATH("relativeDirPath"),
56  
57      /**
58       * The relative path from the VCS project root to the active maven project, with all '/' path
59       * delimiters replaced with '.'.
60       */
61      RELATIVE_PACKAGE("relativePackage"),
62  
63      /**
64       * The groupId of the parent to the active maven project.
65       */
66      PARENT_GROUPID("parentGroupId"),
67  
68      /**
69       * The artifactId of the parent to the active maven project.
70       */
71      PARENT_ARTIFACTID("parentArtifactId"),
72  
73      /**
74       * The maven version of the parent to the active maven project.
75       */
76      PARENT_VERSION("parentMavenVersion"),
77  
78      /**
79       * The relative path between the active maven project and its parent POM.
80       */
81      PARENT_POM_RELATIVE_PATH("parentPomRelativePath"),
82  
83      /**
84       * The fully populated modules XML element, usable only in REACTOR and ROOT_REACTOR poms.
85       */
86      MODULES("modules");
87  
88      // Internal state
89      private String token;
90  
91      PomToken(final String token) {
92          this.token = token;
93      }
94  
95      /**
96       * @return The standard token string to be used within the FactoryParserAgent.
97       */
98      public String getToken() {
99          return token;
100     }
101 
102     /**
103      * Retrieves a SortedMap containing the {@code getToken()} values of each StandardToken as keys and null values.
104      *
105      * @return a SortedMap containing the {@code getToken()} values of each StandardToken as keys and null values.
106      */
107     public static MapBuilder create() {
108 
109         // All done.
110         return new MapBuilder();
111     }
112 
113     /**
114      * Builder class to simplify creating a SortedMap whose keys contains the tokens of each StandardToken.
115      * This Map can be used for token replacements in text template files.
116      */
117     @SuppressWarnings("PMD")
118     static class MapBuilder {
119 
120         // State
121         private SortedMap<String, String> map;
122 
123         MapBuilder() {
124             map = new TreeMap<>();
125             for (PomToken current : PomToken.values()) {
126                 map.put(current.getToken(), null);
127             }
128         }
129 
130         public SortedMap<String, String> build() {
131             return map;
132         }
133 
134         /**
135          * Adds a token/value pair to the MapBuilder map, and returns this builder for
136          * a standard build pattern.
137          *
138          * @param token The non-null StandardToken for which the token/value entry should be added.
139          * @param value The non-null value to add.
140          * @return The {@code map} object.
141          */
142         public MapBuilder addToken(final PomToken token, final String value) {
143 
144             // Check sanity
145             Validate.notNull(token, "Cannot handle null token argument.");
146             Validate.notNull(value, "Cannot handle null value argument.");
147 
148             // Add the token value pair, and return for chaining.
149             map.put(token.getToken(), value);
150             return this;
151         }
152     }
153 }
