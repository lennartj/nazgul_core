43  /**
44   * Builder class to simplify creating a TokenParser tailored for maven POM and project data token substitution.
45   * The returned TokenParsers will use {@code SingleBracketTokenDefinitions} for token definitions, to avoid confusing
46   * the parser tokens with maven's default variable form. (I.e. it will use tokens on the form [token] to
47   * avoid clashing with ${token}).
48   *
49   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
50   */
51  public final class SingleBracketPomTokenParserFactory {
52  
53      // Our log
54      private static final Logger log = LoggerFactory.getLogger(SingleBracketPomTokenParserFactory.class.getName());

56      /**
57       * The key for the token holding the PomType verbatim.
58       */
59      public static final String POMTYPE_KEY = "POMTYPE";
60  
61      /**
62       * The key for the token holding the {@code pomType.name().toLowerCase().replace("_", "-")} value.
63       */
64      public static final String LOWERCASE_POMTYPE_KEY = "pomtype";

66      /**
67       * The name of the directory containing all parent POM projects.
68       */
69      public static final String POMS_DIRECTORY_NAME = "poms";

78      /**
79       * Factory builder method entry method, creating the internal Builder and assigning the supplied tokens.
80       *
81       * @param pomType The POM type for which this SingleBracketPomTokenParserFactory should create a TokenParser.
82       * @param project The non-null Project used by this SingleBracketPomTokenParserFactory to derive tokens for the
83       *                created compound TokenParser.
84       * @return The Builder used to create the TokenParser.
85       */
86      public static DirectoryPrefixEnricher create(final PomType pomType,
87                                                   final Project project) {
88  
89          // Check sanity
90          Validate.notNull(pomType, "Cannot handle null pomType argument.");
91          Validate.notNull(project, "Cannot handle null project argument.");
92  
93          // Derive the project properties.
94          final SortedMap<String, String> pomTokens = new TreeMap<>();
95          pomTokens.put(POMTYPE_KEY, pomType.name());
96          pomTokens.put(LOWERCASE_POMTYPE_KEY, pomType.name().toLowerCase().replace("_", "-"));
97  
98          // All done.
99          return new DirectoryPrefixEnricher(project, pomType, pomTokens, false);
100     }

102     /**
103      * Abstract specification for a BuilderStep class in the controlled Builder pattern.
104      */
105     abstract static class BuilderStep {
106 
107         // Internal state
108         private boolean parentPomType;
109         private boolean requiresProjectSuffix = false;
110         private boolean addProjectNameAsDirectoryPrefix;
111         protected Project project;
112         protected PomType pomType;
113         protected SortedMap<String, String> pomTokens;
114 
115         /**
116          * Creates a new BuilderStep wrapping the supplied data. Each can be {@code null}.
117          *
118          * @param project                         The Project of this BuilderStep.
119          * @param pomType                         The POM type of this BuilderStep.
120          * @param pomTokens                       The pomTokens storage.
121          * @param addProjectNameAsDirectoryPrefix set to {@code true} if directory names and artifactIDs should
122          *                                        add/prepend the project name as prefix
123          *                                        (i.e. yield {@code project.getName()-} prepended to directory names).
124          */
125         protected BuilderStep(final Project project,
126                               final PomType pomType,
127                               final SortedMap<String, String> pomTokens,
128                               final boolean addProjectNameAsDirectoryPrefix) {
129 
130             this.project = project;
131             this.pomType = pomType;
132             this.pomTokens = pomTokens;
133             this.addProjectNameAsDirectoryPrefix = addProjectNameAsDirectoryPrefix;
134 
135             // Check sanity
136             if (pomType != null) {
137 
138                 for (SoftwareComponentPart current : SoftwareComponentPart.values()) {
139                     if (pomType == current.getComponentPomType()) {
140                         requiresProjectSuffix = current.isSuffixRequired();
141                     }
142                 }
143 
144                 parentPomType = pomType.name().contains("PARENT");
145             }
146         }
147 
148         /**
149          * Validates that the given key is present within the pomTokens,
150          * and that it has a non-null value.
151          *
152          * @param key The token key.
153          */
154         protected void validateRequiredKeyValuePair(final String key) {
155 
156             // Check sanity
157             Validate.isTrue(pomTokens.containsKey(key), "Required key [" + key + "] not present in tokenMap.");
158             Validate.notEmpty(pomTokens.get(key),
159                     "Cannot handle null or empty value for key [" + key + "] in tokenMap.");
160         }
161 
162         /**
163          * @return {@code} true if this Builder should have a project suffix, due to the PomType.
164          */
165         protected final boolean isProjectSuffixRequired() {
166             return requiresProjectSuffix;
167         }
168 
169         /**
170          * @return {@code true} if the PomType of this BuilderStep is a Parent POM (as opposed to a Component POM).
171          */
172         protected final boolean isParentPomType() {
173             return parentPomType;
174         }
175 
176         /**
177          * @return {@code true} if this BuilderStep should add prefixes with the Project's name
178          * to local project directory names. This means that SoftwareComponent directories should be
179          * given names on the form {@code foo-finance-api} instead of {@code finance-api},
180          * given that the project's name is {@code foo} and the SoftwareComponent's {@code finance}.
181          */
182         protected boolean isProjectNameAddedToDirectories() {
183             return addProjectNameAsDirectoryPrefix;
184         }
185 
186         /**
187          * Converts the supplied path to a List of segments, assuming that the path uses '/' chars to separate
188          * directory names. Returning a List containing all directory names (i.e. path parts)
189          * in their order of occurrence.
190          *
191          * @param path The path to tokenize, assumed to use '/' as separator chars.
192          * @return A List whose elements contain the directory names within the supplied path.
193          */
194         protected List<String> tokenize(final String path) {
195 
196             final List<String> toReturn = new ArrayList<>();
197             if (path != null) {
198                 final StringTokenizer tok = new StringTokenizer(path, "/", false);
199                 while (tok.hasMoreTokens()) {
200                     toReturn.add(tok.nextToken());
201                 }
202             }
203             return toReturn;
204         }
205     }

265     /**
266      * Class which adds the relative path to a SoftwareComponent, as well as temporary pomTokens
267      * {@code SOFTWARE_COMPONENT_RELATIVE_PATH} and {@code SOFTWARE_COMPONENT_NAME}.
268      * Adds the relative path to the reactor of the SoftwareComponent in which the active Maven project
269      * which should be tokenized by the resulting TokenParser resides. The reactor for a SoftwareComponentPart project
270      * is its parent directory. (I.e. for the {@code finance-api} project residing within the SoftwareComponent
271      * {@code finance}, the relative path should be {@code services/finance} assuming that we would like the
272      * finance SoftwareComponent to reside in the {@code services} VCS path).
273      */
274     public static class RelativePathEnricher extends BuilderStep {
275 
276         /**
277          * The pomTokens key holding the relative path to the SoftwareComponent containing the active Maven project.
278          */
279         public static final String SOFTWARE_COMPONENT_RELATIVE_PATH = "softwareComponentRelativePath";
280 
281         /**
282          * The pomTokens key holding the relative name of the SoftwareComponent containing the active Maven project.
283          */
284         public static final String SOFTWARE_COMPONENT_NAME = "softwareComponentName";
285 
286         /**
287          * The string name of the PomType for the parent of the active Maven project.
288          * (I.e. {@code parentPomType.name()}).
289          */
290         public static final String PARENT_POMTYPE = "parent_pomType";
291 
292         RelativePathEnricher(final Project project,
293                              final PomType pomType,
294                              final SortedMap<String, String> pomTokens,
295                              final boolean addProjectNameAsDirectoryPrefix) {
296             super(project, pomType, pomTokens, addProjectNameAsDirectoryPrefix);
297         }
298 
299         /**
300          * Parent projects are not part of a SoftwareComponent; therefore one should call this method instead of
301          * the {@code inSoftwareComponentWithRelativePath()} method.
302          *
303          * @return The next BuilderStep subtype in the builder chain.
304          */
305         public PrefixEnricher isParentProject() {
306 
307             // Check sanity
308             Validate.isTrue(isParentPomType(),
309                     "isParentProject should only be called for PARENT PomTypes.");
310 
311             // Since all parent poms are located in the [projName]/poms/[xyz-]projName-[something-]parent
312             // directory, we really need not define the software component path at all.
313             //
314             // pomTokens.put(SOFTWARE_COMPONENT_RELATIVE_PATH, "poms");
315 
316             // Add the PomType of the parent, which is non-null for PomTypes other than PomType.PARENT.
317             final PomType parentPomType = getParentPomType(2);
318             if (parentPomType != null) {
319                 pomTokens.put(PARENT_POMTYPE, parentPomType.name());
320 
321                 // Add the relative path to the parent POM directory
322                 final String parentPomDirName = (isProjectNameAddedToDirectories() || isParentPomType()
323                         ? project.getName() + Name.DEFAULT_SEPARATOR
324                         : "") + parentPomType.name().toLowerCase().replace("_", Name.DEFAULT_SEPARATOR);
325                 pomTokens.put(PomToken.PARENT_POM_RELATIVE_PATH.getToken(), "../" + parentPomDirName);
326             } else {
327                 // This should happen if pomType is PomType.PARENT
328                 pomTokens.put(PomToken.PARENT_POM_RELATIVE_PATH.getToken(), "");
329             }
330 
331             // All Done.
332             return new PrefixEnricher(project, pomType, pomTokens, isProjectNameAddedToDirectories());
333         }

335         /**
336          * <p>Adds the relative path to the reactor of the SoftwareComponent in which the active Maven project which
337          * should be tokenized by the resulting TokenParser resides. The reactor for a SoftwareComponentPart project
338          * is its parent directory. (I.e. for the {@code finance-api} project residing within the SoftwareComponent
339          * {@code finance}, the relative path should be {@code services/finance} assuming that we would like the
340          * finance SoftwareComponent to reside in the {@code services} VCS path).
341          * This implies that the directory name of the project itself (such as {@code finance-api}) should not be
342          * provided in the componentRelativePath.</p>
343          * <p>Example:</p>
344          * <pre>
345          *     <code>
346          *         // Assume that we want to create the TokenParser for the project
347          *         //
348          *         // finance-api
349          *         //
350          *         // Then the componentRelativePath provided should be "services/finance"
351          *         // and not "services/finance/finance-api"
352          *         //
353          *         final TokenParser tokenParser = SingleBracketPomTokenParserFactory
354          *         .create(PomType.COMPONENT_API, acmeFooProject, false)
355          *         .inSoftwareComponentWithRelativePath("services/finance")
356          *         .withProjectGroupIdPrefix("org.acme", false)
357          *         .withoutProjectSuffix()
358          *         .build();
359          *     </code>
360          * </pre>
361          *
362          * @param relativePathToSoftwareComponent Cannot be null or empty. The relative directory path from the project VCS
363          *                                        root to the <strong>component directory</strong> containing the Maven
364          *                                        project which should be tokenized by the resulting TokenParser.
365          * @return The next BuilderStep subtype in the builder chain.
366          */
367         public PrefixEnricher inSoftwareComponentWithRelativePath(final String relativePathToSoftwareComponent) {
368 
369             // Delegate if required
370             if (isParentPomType()) {
371                 return isParentProject();
372             }
373 
374             final String invalidRelativePathMsg = "Cannot handle null or empty "
375                     + "projectRelativeDirectoryPath argument for non-RootReactor projects.";
376             final List<String> pathSegments = tokenize(relativePathToSoftwareComponent);
377 
378             // We have a non-parent project, so we might need to calculate the relative path.
379             switch (pomType) {
380 
381                 case ROOT_REACTOR:
382                     pomTokens.put(SOFTWARE_COMPONENT_RELATIVE_PATH, "");
383                     // pomTokens.put(PomToken.PARENT_POM_RELATIVE_PATH.getToken(), "");
384                     // pomTokens.put(SOFTWARE_COMPONENT_NAME, null);
385                     break;
386 
387                 case REACTOR:
388                     Validate.notEmpty(relativePathToSoftwareComponent, invalidRelativePathMsg);
389                     final String pathSegmentSeparator = "/";
390                     String paddedRelativePath = relativePathToSoftwareComponent.endsWith(pathSegmentSeparator)
391                             ? relativePathToSoftwareComponent.substring(0,
392                             relativePathToSoftwareComponent.lastIndexOf(pathSegmentSeparator))
393                             : relativePathToSoftwareComponent;
394                     pomTokens.put(SOFTWARE_COMPONENT_RELATIVE_PATH, paddedRelativePath);
395                     pomTokens.put(SOFTWARE_COMPONENT_NAME, pathSegments.get(pathSegments.size() - 1));
396 
397                     PomType parentPomType = getParentPomType(pathSegments.size());
398                     pomTokens.put(PARENT_POMTYPE, parentPomType.name());
399                     pomTokens.put(PomToken.PARENT_POM_RELATIVE_PATH.getToken(), "");
400                     break;
401 
402                 default:
403                     Validate.notEmpty(relativePathToSoftwareComponent, invalidRelativePathMsg);
404                     pomTokens.put(SOFTWARE_COMPONENT_RELATIVE_PATH, relativePathToSoftwareComponent);
405                     pomTokens.put(SOFTWARE_COMPONENT_NAME, pathSegments.get(pathSegments.size() - 1));
406 
407                     parentPomType = getParentPomType(pathSegments.size());
408                     final String relative = getRelativePathToParentPom(relativePathToSoftwareComponent, parentPomType);
409                     pomTokens.put(PARENT_POMTYPE, parentPomType.name());
410                     pomTokens.put(PomToken.PARENT_POM_RELATIVE_PATH.getToken(), relative);
411                     break;
412             }
413 
414             // All Done.
415             return new PrefixEnricher(project, pomType, pomTokens, isProjectNameAddedToDirectories());
416         }

422         private String getRelativePathToParentPom(final String relativePathToSoftwareComponent,
423                                                   final PomType parentPomType) {
424 
425             final List<String> tokenizedPath = tokenize(relativePathToSoftwareComponent);
426             final int numCdsUpwards = pomType == PomType.REACTOR
427                     ? tokenizedPath.size()
428                     : tokenizedPath.size() + 1;
429 
430             final StringBuilder relativePathBuilder = new StringBuilder();
431             for (int i = 0; i < numCdsUpwards; i++) {
432                 relativePathBuilder.append("../");
433             }
434             relativePathBuilder.append(POMS_DIRECTORY_NAME + "/");
435 
436             // Always prepend the project name to parent POM dirs (i.e. in the "poms/" directory).
437             final String parentPomDirName = project.getName() + Name.DEFAULT_SEPARATOR
438                     + parentPomType.name().replace("_", Name.DEFAULT_SEPARATOR).toLowerCase();
439 
440             relativePathBuilder.append(parentPomDirName);
441             return relativePathBuilder.toString();
442         }

444         private PomType getParentPomType(final int numSegmentsInRelativePathToProjectRoot) {
445 
446             PomType toReturn = null;
447 
448             switch (pomType) {
449                 case REACTOR:
450                     toReturn = numSegmentsInRelativePathToProjectRoot < 2
451                             ? PomType.ROOT_REACTOR
452                             : PomType.REACTOR;
453                     break;
454 
455                 case COMPONENT_MODEL:
456                     toReturn = PomType.MODEL_PARENT;
457                     break;
458 
459                 case MODEL_PARENT:
460                 case COMPONENT_SPI:
461                 case COMPONENT_API:
462                     toReturn = PomType.API_PARENT;
463                     break;
464 
465                 case API_PARENT:
466                 case WAR_PARENT:
467                 case COMPONENT_TEST:
468                 case COMPONENT_IMPLEMENTATION:
469                 case OTHER_PARENT:
470                     toReturn = PomType.PARENT;
471                     break;
472 
473                 default:
474 
475                     if (log.isWarnEnabled()) {
476                         log.warn("Could not determine parent PomType for pomType [" + pomType + "].");
477                     }
478                     toReturn = null;
479                     break;
480             }
481 
482             // All done.
483             return toReturn;
484         }

487     /**
488      * BuilderStep which adds the prefix of Maven GroupIDs for a relative path within a directory.
489      * A "groupIdPrefix" is normally identical to a reverse DNS of the project's organisation,
490      * similar to {@code se.jguru} or {@code org.acme}. Note that the groupIdPrefix should <strong>not</strong>
491      * include the project name (i.e. {@code project.getName()}), as it will be automatically appended by this builder.
492      * Given a SoftwareComponent with the relative path {@code services/finance}, the groupIdPrefix
493      * {@code se.jguru} and the project name {@code foo}, the full groupId of the finance service
494      * SoftwareComponent can be constructed as {@code se.jguru.foo.services.finance}.
495      * This, in turn, implies that the API project in the finance SoftwareComponent will retrieve the groupId
496      * {@code se.jguru.foo.services.finance.api}.
497      */
498     public static class PrefixEnricher extends BuilderStep {
499 
500         /**
501          * The property containing the {@code groupIdPrefix}, a.k.a. reverse organisation DNS,
502          * which should be prepended to the groupId and package properties.
503          * Given a SoftwareComponent with the relative path {@code services/finance}, the groupIdPrefix
504          * {@code se.jguru} and the project name {@code foo}, the full groupId of the finance service
505          * SoftwareComponent can be constructed as {@code se.jguru.foo.services.finance}.
506          * This, in turn, implies that the API project in the finance SoftwareComponent will retrieve the groupId
507          * {@code se.jguru.foo.services.finance.api}.
508          */
509         public static final String REVERSE_DNS_OF_ORGANISATION = "reverseOrganisationDns";
510 
511         PrefixEnricher(final Project project,
512                        final PomType pomType,
513                        final SortedMap<String, String> pomTokens,
514                        final boolean addProjectNameAsDirectoryPrefix) {
515             super(project, pomType, pomTokens, addProjectNameAsDirectoryPrefix);
516         }
517 
518         /**
519          * Adds prefix which should be prepended to the relative path to generate a groupId for the current project.
520          * A "projectGroupIdPrefix" is normally identical to a reverse DNS of the project's organisation,
521          * similar to {@code se.jguru} or {@code org.acme}. Note that the groupIdPrefix should <strong>not</strong>
522          * include the project name (i.e. {@code project.getName()}), as it will be automatically appended by this
523          * builder.
524          *
525          * @param projectGroupIdPrefix A non-null prefix string added to the relative path to generate a groupId for
526          *                             the current project. Empty values are permitted, but null are not. Normally
527          *                             identical to a reverse DNS of the project's organisation, similar to
528          *                             {@code se.jguru} or {@code org.acme}. Note that the groupIdPrefix should
529          *                             <strong>not</strong> include the project name (i.e. {@code project.getName()}),
530          *                             as it will be automatically appended by this builder.
531          * @return The next BuilderStep subtype in the builder chain.
532          */
533         public SuffixEnricher withProjectGroupIdPrefix(final String projectGroupIdPrefix) {
534 
535             // Check sanity
536             Validate.notEmpty(projectGroupIdPrefix, "Cannot handle null or empty projectGroupIdPrefix argument.");
537 
538             final PomType parentPomType = pomTokens.get(RelativePathEnricher.PARENT_POMTYPE) == null
539                     ? null
540                     : PomType.valueOf(pomTokens.get(RelativePathEnricher.PARENT_POMTYPE));
541 
542             // Add the relevant pomTokens
543             pomTokens.put(REVERSE_DNS_OF_ORGANISATION, projectGroupIdPrefix);
544             pomTokens.put(PomToken.PARENT_GROUPID.getToken(), getParentGroupId(projectGroupIdPrefix, parentPomType));
545             pomTokens.put(PomToken.PARENT_ARTIFACTID.getToken(), getParentArtifactId(parentPomType));
546 
547             // All done.
548             return new SuffixEnricher(project, pomType, pomTokens, isProjectNameAddedToDirectories());
549         }

551         //
552         // Private helpers
553         //
554 
555         private String getParentGroupId(final String projectGroupIdPrefix, final PomType parentPomType) {
556 
557             final String delimiter = ".";
558             final String projectPrefix = project.getPrefix() != null ? project.getPrefix() + delimiter : "";
559             StringBuilder groupIdBuilder = new StringBuilder(
560                     projectGroupIdPrefix + delimiter + projectPrefix + project.getName() + delimiter);
561 
562             // Calculate the Maven groupId for the parent POM
563             switch (pomType) {
564                 case ROOT_REACTOR:
565                     groupIdBuilder = new StringBuilder(project.getReactorParent().getGroupId());
566                     break;
567 
568                 case PARENT:
569                     groupIdBuilder = new StringBuilder(project.getParentParent().getGroupId());
570                     break;
571 
572                 case REACTOR:
573 
574                     // Find the relative path of the parent reactor only for the case that the
575                     // parent is not
576                     //
577                     // relativePath: osgi.launcher --> parent GID: [se.jguru.nazgul.core].osgi
578                     if (parentPomType != PomType.ROOT_REACTOR) {
579                         final String relPath = pomTokens.get(RelativePathEnricher.SOFTWARE_COMPONENT_RELATIVE_PATH);
580                         final List<String> segments = tokenize(relPath);
581 
582                         Validate.isTrue(segments.size() > 0, "For a reactor POM not immediately below the root "
583                                 + "reactor, the number of path segments should be at least 1. (Got: "
584                                 + segments.size() + ")");
585

810         /**
811          * Adds a project suffix, which is used to calculate groupId and artifactId for the local
812          * project for ComponentTypes where suffix is required. If the {@code isProjectSuffixRequired()} yields
813          * {@code false}, the projectSuffix value is ignored (implying it may be null or empty in these cases).
814          *
815          * @param projectSuffix The suffix for the project. May be null or empty, depending on the active projectType.
816          * @return The next BuilderStep subtype in the builder chain.
817          * @see se.jguru.nazgul.core.quickstart.api.generator.SoftwareComponentPart#isSuffixRequired()
818          */
819         public MavenVersionEnricher withProjectSuffix(final String projectSuffix) {
820 
821             // Get some token parts
822             final String localProjectDirName = getProjectDirectoryName(projectSuffix);
823             final String relativePackage = getRelativePackage(projectSuffix);
824 
825             String pathToParentDir = pomTokens.get(RelativePathEnricher.SOFTWARE_COMPONENT_RELATIVE_PATH);
826             if (pathToParentDir == null && isParentPomType()) {
827                 pathToParentDir = POMS_DIRECTORY_NAME;
828             }
829             final String relativeDirPath = (pomType == PomType.REACTOR || pomType == PomType.ROOT_REACTOR)
830                     ? pathToParentDir
831                     : pathToParentDir + "/" + localProjectDirName;
832 
833             pomTokens.put(PomToken.RELATIVE_DIRPATH.getToken(), relativeDirPath);
834             pomTokens.put(PomToken.RELATIVE_PACKAGE.getToken(), relativePackage);
835 
836             // Now find the groupId and artifactId for the active project.
837             final boolean nonEmptyProjectPrefix = project.getPrefix() != null && project.getPrefix().length() > 0;
838             final String artifactPrefix = nonEmptyProjectPrefix ? project.getPrefix() + Name.DEFAULT_SEPARATOR : "";
839 
840             // ArtifactID: Calculate for the current project.
841             //
842             // Using the localProjectDirName, add the project name as a prefix if required.
843             // DirName: "finance-api"  --> ArtifactID: "acme-foo-finance-api"
844             // DirName: "foo-finance-api"  --> ArtifactID: "acme-foo-finance-api"
845             //
846             final String slice = artifactPrefix + project.getName() + Name.DEFAULT_SEPARATOR;
847             String rawProjectDirName = localProjectDirName.startsWith(slice)
848                     ? localProjectDirName.substring(slice.length())
849                     : localProjectDirName;
850             String artifactId = artifactPrefix + project.getName() + Name.DEFAULT_SEPARATOR + rawProjectDirName;
851 
852             if (pomType == PomType.REACTOR) {
853                 artifactId = artifactId + "-reactor";
854             } else if (pomType == PomType.ROOT_REACTOR) {
855                 artifactId = artifactPrefix + project.getName() + Name.DEFAULT_SEPARATOR + "reactor";
856             }
857             pomTokens.put(PomToken.ARTIFACTID.getToken(), artifactId);
858 
859             //
860             // GroupID: Calculate for the current project, on the form:
861             //
862             // Component Project [xyz]:
863             //   [reverseDns].[[projectPrefix].][projectName].[relativePathToComponentParent].[componentName].[type],
864             //   i.e:
865             //
866             //   Reverse DNS:    org.acme
867             //   Project Prefix: gnat
868             //   Project Name:   foo
869             //   Relative Path:  messaging
870             //
871             //   a) Project type MODEL           --> groupID: org.acme.gnat.foo.messaging.model
872             //   b) Project type REACTOR         --> groupID: org.acme.gnat.foo.messaging
873             //   c) Project type SPI (jpa)       --> groupID: org.acme.gnat.foo.messaging.spi.jpa
874             //   d) Project type IMPL (pojo)     --> groupID: org.acme.gnat.foo.messaging.impl.pojo
875             //
876             final String groupId = pomTokens.get(PrefixEnricher.REVERSE_DNS_OF_ORGANISATION) + "."
877                     + (nonEmptyProjectPrefix ? project.getPrefix() + "." : "")
878                     + project.getName()
879                     + ("".equals(relativePackage) ? "" : "." + relativePackage);
880             pomTokens.put(PomToken.GROUPID.getToken(), groupId);
881 
882             // All done.
883             return new MavenVersionEnricher(project, pomType, pomTokens, isProjectNameAddedToDirectories());
884         }

910          * @return The name of the directory for the active Maven project. Never null or empty.

912         @SuppressWarnings("all")
913         private String getProjectDirectoryName(final String projectSuffix) {
914 
915             // Check sanity
916             if (pomType == PomType.ROOT_REACTOR) {
917                 // This is a special case
918                 return project.getName();
919             }
920 
921             final boolean addProjectSuffixToDirName = isProjectSuffixRequired()
922                     && projectSuffix != null
923                     && !projectSuffix.isEmpty();
924 
925             final StringBuilder dirNameBuilder = new StringBuilder();
926             if (isProjectNameAddedToDirectories()) {
927 
928                 //
929                 // In this case, the project dirName should be prepended with the
930                 // project prefix (if it exists) and name, i.e:
931                 //
932                 // a) foo-gnat-messaging-model, or
933                 // b) gnat-messaging-model (if no prefix is defined within the project).
934                 //
935                 if (project.getPrefix() != null && !"".equals(project.getPrefix())) {
936                     dirNameBuilder.append(project.getPrefix()).append(Name.DEFAULT_SEPARATOR);
937                 }
938                 dirNameBuilder.append(project.getName()).append(Name.DEFAULT_SEPARATOR);
939             }
940 
941             // For parent POMs, we should simply add the PomType as the name ...
942             if (!isParentPomType()) {
943 
944                 //
945                 // The SoftwareComponent's name should not be null or empty for non-parent projects.
946                 //
947                 final String softwareComponentName = pomTokens.get(RelativePathEnricher.SOFTWARE_COMPONENT_NAME);
948                 Validate.notEmpty(softwareComponentName, "SoftwareComponentName was null or empty for non-parent "
949                         + "pomType [" + pomType + "]. Cannot calculate project directory name. "
950                         + "This should not happen.");
951                 dirNameBuilder.append(softwareComponentName);
952             }
953 
954             // Append the project type to the directory name, minding that reactor POMs are a special case.
955             // Reactor project reside within the software component directory and simply ensure that all
956             // SoftwareComponentPart projects are included within the build reactor.
957             final String lcPomType = pomType.name().toLowerCase().replace("_", Name.DEFAULT_SEPARATOR);
958             if (pomType == PomType.REACTOR) {
959 
960                 // The reactor POM in the poms reactor is not part of a software component.
961                 if (POMS_DIRECTORY_NAME.equals(pomTokens.get(RelativePathEnricher.SOFTWARE_COMPONENT_NAME))) {
962                     return POMS_DIRECTORY_NAME;
963                 }
964 
965                 // Standard component reactor; don't append anything.
966                 // dirNameBuilder.append(Name.DEFAULT_SEPARATOR).append(lcPomType);
967             } else if (isParentPomType()) {
968                 dirNameBuilder.append(lcPomType);
969             } else {
970 
971                 // This is a component PomType. Remove the "component-" part of the lcPomType.
972                 final int stripOffLength = ("component" + Name.DEFAULT_SEPARATOR).length();
973                 if (lcPomType.length() < stripOffLength) {
974                     log.error("Got stripOffLength: [" + stripOffLength + "] and lcPomType.length(): ["
975                             + lcPomType.length() + "] for lcPomType [" + lcPomType + "]");
976                     throw new IllegalStateException("This should not happen...");
977                 }
978 
979                 dirNameBuilder.append(Name.DEFAULT_SEPARATOR).append(lcPomType.substring(stripOffLength));
980             }
981 
982             if (addProjectSuffixToDirName) {
983                 dirNameBuilder.append(Name.DEFAULT_SEPARATOR).append(projectSuffix);
984             }
985 
986             // All done - but shorten the name of the directory somewhat.
987             return dirNameBuilder
988                     .toString()
989                     .replace("implementation", "impl");
990         }

992         private String getRelativePackage(final String localProjectSuffix) {
993 
994             final String prefix = project.getPrefix() != null ? project.getPrefix() + Name.DEFAULT_SEPARATOR : "";
995             final StringBuilder relativePackageBuilder = new StringBuilder();
996 
997             if (isParentPomType()) {
998 
999                 // The relative package should be something like
1000                 // poms.[project.prefix-][project.name]-[pomType],
1001                 //
1002                 // given that the pomType is made lowercase.
1003                 relativePackageBuilder
1004                         .append(POMS_DIRECTORY_NAME)
1005                         .append(".")
1006                         .append(prefix)
1007                         .append(project.getName())
1008                         .append(Name.DEFAULT_SEPARATOR)
1009                         .append(pomType.toString().toLowerCase().replace("_", Name.DEFAULT_SEPARATOR));
1010             } else if (pomType == PomType.ROOT_REACTOR) {
1011                 return "";
1012             } else {
1013 
1014                 // The first part of the relative package should be
1015                 //
1016                 // [relativePathToParent (converted)].[softwareComponent]
1017                 final String relativePathToSoftwareComponent = pomTokens.get(
1018                         RelativePathEnricher.SOFTWARE_COMPONENT_RELATIVE_PATH);
1019                 relativePackageBuilder.append(relativePathToSoftwareComponent.replace("/", "."));
1020 
1021                 if (pomType != PomType.REACTOR) {
1022 
1023                     // A relative path of:
1024                     // 1) "services/finance/finance-api", or
1025                     // 2) "services/finance/foo-finance-api"
1026                     //
1027                     // ... should yield relative package:
1028                     // "services.finance.api"
1029                     //
1030                     // ... where the last package segment is retrieved from the PomType.
1031 
1032                     final String lcPomTypeName = pomType.name().toLowerCase();
1033                     final String pomTypePackage = lcPomTypeName.substring(lcPomTypeName.lastIndexOf("_") + 1);
1034                     relativePackageBuilder.append(".").append(pomTypePackage);

1036                     // Append the project suffix, if required
1037                     if (isProjectSuffixRequired()) {
1038                         relativePackageBuilder.append(".").append(localProjectSuffix);
1039                     }
1040                 }
1041             }
1042 
1043             // All done.
1044             return relativePackageBuilder.toString();
1045         }
