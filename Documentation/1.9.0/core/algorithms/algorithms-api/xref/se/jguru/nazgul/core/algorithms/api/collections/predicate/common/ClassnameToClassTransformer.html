22  package se.jguru.nazgul.core.algorithms.api.collections.predicate.common;
23  
24  import se.jguru.nazgul.core.algorithms.api.collections.predicate.Transformer;
25  
26  /**
27   * Transforms fully qualified class names to Class instances.
28   *
29   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
30   */
31  public class ClassnameToClassTransformer implements Transformer<String, Class<?>> {
32  
33      // Internal state
34      private ClassLoader classLoader;
35  
36      /**
37       * Creates a ClassnameToClassTransformer using the current thread context ClassLoader
38       * to load the classes from given class names. If a contextClassLoader could not be found,
39       * falls back to the class loader of this ClassnameToClassTransformer instance.
40       */
41      public ClassnameToClassTransformer() {
42          this(null);
43      }
44  
45      /**
46       * Creates a ClassnameToClassTransformer using the provided ClassLoader to load the classes
47       * from given class names. If classloader is {@code null}, fallback to default behaviour.
48       *
49       * @param classLoader The ClassLoader which should be used to load the Classes for each
50       *                    given fully qualified class name. If classloader is {@code null}, fallback
51       *                    to default behaviour.
52       * @see #ClassnameToClassTransformer()
53       */
54      public ClassnameToClassTransformer(final ClassLoader classLoader) {
55          this.classLoader = classLoader;
56      }
57  
58      /**
59       * Transforms a fully qualified class name to its corresponding Class instance.
60       *
61       * @param input The fully qualified class name to be transformed.
62       * @return the Class of the provided fully qualified class name.
63       */
64      @Override
65      public Class<?> transform(final String input) {
66  
67          // Do we have a given ClassLoader?
68          ClassLoader loader = (classLoader != null ? classLoader : Thread.currentThread().getContextClassLoader());
69          if (loader == null) {
70              loader = ClassnameToClassTransformer.class.getClassLoader();
71          }
72  
73          try {
74              return loader.loadClass(input);
75          } catch (ClassNotFoundException e) {
76              throw new IllegalArgumentException("Could not find class [" + input + "] in ClassLoader ["
77                      + loader + "]", e);
78          }
79      }
80  }
