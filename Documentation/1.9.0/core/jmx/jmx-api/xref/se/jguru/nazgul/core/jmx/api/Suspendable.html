22  package se.jguru.nazgul.core.jmx.api;
23  
24  import javax.management.MXBean;
25  
26  /**
27   * Specification for a Suspendable LifecycleStateful instance, where operations can
28   * be suspended (put in a wait state) and resumed (resurrected from wait state).
29   *
30   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid, jGuru Europe AB</a>
31   */
32  @MXBean
33  public interface Suspendable extends LifecycleStateful {
34  
35      /**
36       * Performs a graceful and synchronous suspension of this Suspendable, implying that any operations
37       * invoked within this Suspendable are not executed until the resume method has been
38       * called. Calling suspend multiple times should have no effect. When returning from this suspend
39       * method, the LifecycleStateful instance should be properly suspended.
40       *
41       * @return {@code true} if this Suspendable was successfully suspended when this method
42       * has executed, and {@code false} otherwise.
43       * @throws IllegalStateException if this Suspendable was not in state {@code LifecycleStateful.STARTED}.
44       */
45      boolean suspend() throws IllegalStateException;
46  
47      /**
48       * Resumes operation of this suspendable, implying that any operations invoked within this Suspendable are
49       * executed after this resume method has been called. Calling resume multiple times should have no effect.
50       * When returning from this resume method, the LifecycleStateful instance should be properly resumed.
51       *
52       * @return {@code true} if this Suspendable was successfully resumed when this method
53       * has executed, and {@code false} otherwise.
54       * @throws IllegalStateException if this Suspendable was not in state {@code LifecycleStateful.STOPPED}.
55       */
56      boolean resume() throws IllegalStateException;
57  }
